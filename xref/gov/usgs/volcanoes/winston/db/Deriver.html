11  import java.text.ParseException;
12  import java.util.List;
13  import java.util.logging.Level;
14  import java.util.logging.Logger;

16  /**
17   * A class to recalculate stored RSAM values
18   * 
19   * @author Dan Cervelli
20   * @author Tom Parker
21   */
22  public class Deriver
23  {
24  	private static final String DEFAULT_CONFIG_FILENAME = "Deriver.config";
25  	private static final double DEFAULT_CHUNK_SIZE = 600.0;
26  	
27  	private static final boolean DEFAULT_RSAM_ENABLE = true;
28  	private static final int DEFAULT_RSAM_DELTA = 10;
29  	private static final int DEFAULT_RSAM_DURATION = 60;
30  	
31  	private WinstonDatabase winston;
32  	private InputEW input;
33  	private Data data;
34  	
35  	private double startTime;
36  	private double endTime;
37  	
38  	private Logger logger;
39  	
40  	private double chunkSize = 3600;
41  	
42  	private boolean quit = false;
43  	
44  	private boolean rsamEnable = DEFAULT_RSAM_ENABLE;
45  	private int rsamDelta = DEFAULT_RSAM_DELTA;
46  	private int rsamDuration = DEFAULT_RSAM_DURATION;
47  	
48  	private ConfigFile config;
49  	
50  	private List<String> sourceChannels;
51  	
52  	public Deriver()
53  	{
54  		logger = Log.getLogger("gov.usgs.winston");
55  		logger.setLevel(Level.FINE);
56  		config = new ConfigFile(DEFAULT_CONFIG_FILENAME);
57  		processConfigFile();
58  		input = new InputEW(winston);
59  		data = new Data(winston);
60  		deriveAll();
61  	}
62  	
63  	public void processConfigFile()
64  	{
65  		winston = WinstonDatabase.processWinstonConfigFile(config);
66  		
67  		sourceChannels = config.getList("channel");
68  		
69  		String timeRange = config.getString("timeRange");
70  		double[] t;
71  		try {
72  			t = Time.parseTimeRange(timeRange);
73  			startTime = t[0];
74  			endTime = t[1];
75  		} catch (ParseException e) {
76  			System.err.println("Can't parse timeRange");
77  			System.exit(1);
78  		}
79  		
80  		chunkSize = Util.stringToDouble(config.getString("chunkSize"), DEFAULT_CHUNK_SIZE);
81  
82  		rsamEnable = Util.stringToBoolean(config.getString("rsam.enable"), DEFAULT_RSAM_ENABLE);
83  		rsamDelta = Util.stringToInt(config.getString("rsam.delta"), DEFAULT_RSAM_DELTA);
84  		rsamDuration = Util.stringToInt(config.getString("rsam.duration"), DEFAULT_RSAM_DURATION);
85  	}
86  	
87  	public void deriveAll()
88  	{
89  		for (String channel : sourceChannels)
90  		{
91  			logger.info("Working on " + channel);
92  			derive(channel, startTime, endTime);
93  		}
94  	}
95  	
96  	private void derive(String ch, double t1, double t2)
97  	{
98  		input.setRowParameters((int)chunkSize + 65, 60);
99  		
100 		double ct = t1 - chunkSize;
101 		List<TraceBuf> tbs = null;
102 		while (ct < t2)
103 		{
104 			if (quit)
105 				break;
106 			ct += chunkSize;
107 			
108 			double ret = Math.min(ct + chunkSize, t2);
109 			CodeTimer netTimer = new CodeTimer("net");
110 			try{
111 			tbs = data.getTraceBufs(ch, ct, ret, 0);
112 			} catch (UtilException e){
113 			}
114 			netTimer.stop();
115 			
116 			if (tbs != null && tbs.size() > 0)
117 			{
118 				CodeTimer inputTimer = new CodeTimer("input");
119 				input.rederive(tbs, rsamEnable, rsamDelta, rsamDuration);
120 				inputTimer.stop();
121 				
122 				logger.info(String.format("Derived %d TraceBufs in %.3fms, insert completed in %.3fms", 
123 						tbs.size(), netTimer.getRunTimeMillis(), inputTimer.getRunTimeMillis()));
124 			}
125 		}
126 	}
127 	
128 	public static void main(String[] args)
129 	{
130 		new Deriver();
131 	}
132 }
