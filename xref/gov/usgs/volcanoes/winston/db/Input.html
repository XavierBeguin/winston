56        return !Double.isNaN(t1);

59      // TODO: optimize. garbage generator.
60      public void putData(final double j2k, final double[] d) {
61        data.put(j2k, d);
62        if (data.size() > 60) {
63          final int numToDelete = data.size() - 30;
64          final Iterator<Double> it = data.keySet().iterator();
65          for (int i = 0; i < numToDelete; i++) {
66            it.next();
67            it.remove();
68          }
69        }
70      }
71    }

147   public void calculateSpan(final String code) {
148     try {
149       final List<String> dayList = getDayTables(code);
150       double mint = 1E300;
151       double maxt = -1E300;
152       String table = dayList.get(0);
153       ResultSet rs = winston.getStatement().executeQuery("SELECT MIN(st) FROM `" + table + "`");
154       rs.next();
155       mint = Math.min(mint, rs.getDouble(1));
156       table = dayList.get(dayList.size() - 1);
157       rs = winston.getStatement().executeQuery("SELECT MAX(et) FROM `" + table + "`");
158       rs.next();
159       maxt = Math.max(maxt, rs.getDouble(1));
160       setTimeSpan(code, mint, maxt);
161       rs.close();
162     } catch (final Exception e) {
163       LOGGER.error("Could not calculate span: {}", code);
164     }

178   private boolean checkTable(final String code, final String date) {
179     final String table = code + "$$" + date;
180     if (checkTableCache.contains(table)) {
181       return true;
182     }
183 
184     try {
185       final ResultSet rs =
186           winston.getStatement().executeQuery("SELECT COUNT(*) FROM `" + table + "`");
187       final boolean result = rs.next();
188       if (result) {
189         checkTableCache.add(table);
190       }
191       rs.close();
192       return result;
193     } catch (final Exception e) {
194       checkTableCache.clear();
195       return createDayTable(code, date);
196     }
197   }

235   public List<String> getDayTables(final String code) {
236     try {
237       final ArrayList<String> list = new ArrayList<String>(10);
238       if (!winston.useDatabase(code)) {
239         return null;
240       }
241       final ResultSet rs = winston.getStatement().executeQuery("SHOW TABLES");
242       while (rs.next()) {
243         list.add(rs.getString(1));
244       }
245       rs.close();
246       Collections.sort(list);
247       final ArrayList<String> dayList = new ArrayList<String>(list.size() / 2);
248       for (int i = 0; i < list.size(); i++) {
249         final String table = list.get(i);
250         final String day = table.substring(table.indexOf("$$") + 2);
251         if (day.length() == 10 && day.charAt(4) == '_' && day.charAt(7) == '_'
252             && Character.isDigit(day.charAt(0)) && Character.isDigit(day.charAt(9))) {
253           dayList.add(table);
254         }
255       }
256       return dayList;
257     } catch (final Exception e) {
258       LOGGER.error("Could not generate list of tables: {}", code);
259     }
260     return null;
261   }

269   public InputResult inputTraceBuf(final TraceBuf tb) {
270     InputResult result = InputResult.ERROR;
271     try {
272       final String code = tb.toWinstonString();
273       ChannelInputOptimizer opt = channelOptimizers.get(code);
274       if (opt == null) {
275         opt = new ChannelInputOptimizer(code);
276         channelOptimizers.put(code, opt);
277         final double[] span = data.getTimeSpan(code);
278         opt.t1 = span[0];
279         opt.t2 = span[1];
280       }
281 
282       winston.useDatabase(code);
283       final double ts = tb.getStartTimeJ2K();
284       final String date = dateFormat.format(J2kSec.asDate(ts));
285       boolean createdTable = false;
286       if (!tableExists(code, date)) {
287         if (createTable(code, date)) {
288           createdTable = true;
289         } else {
290           throw new Exception("Could not create table.");
291         }
292       }
293 
294       String table = code + "$$" + date;
295 
296       if (writeLocks && !locks.contains(table)) {
297         currentLock = opt;
298         locks.add(table);
299         locks.add(code + "$$H" + date);
300         winston.getStatement()
301             .execute("LOCK TABLES " + table + " WRITE, " + code + "$$H" + date + " WRITE");
302       }
303 
304       final PreparedStatement insert =
305           winston.getPreparedStatement("INSERT IGNORE INTO `" + table + "` VALUES (?,?,?,?,?)");
306       insert.setDouble(1, ts);
307       insert.setDouble(2, tb.getEndTimeJ2K());
308       insert.setDouble(3, tb.samplingRate());
309       insert.setString(4, tb.dataType());
310       final byte[] stripped = Arrays.trimToCapacity(tb.bytes, tb.bytes.length - 1);
311       final byte[] compressed = Zip.compress(stripped);
312       insert.setBytes(5, compressed);
313       insert.executeUpdate();
314 
315       opt.t1 = Math.min(opt.t1, tb.getStartTimeJ2K());
316       opt.t2 = Math.max(opt.t2, tb.getEndTimeJ2K());
317 
318       if (!writeLocks) {
319         setTimeSpan(code, opt.t1, opt.t2);
320         // setTimeSpan(code, opt.t1, opt.t2);
321       }
322 
323       /// ----- now insert/update helicorder/rsam data
324       table = code + "$$H" + date;
325       final double fst = Math.floor(tb.getStartTimeJ2K());
326       final double cet = Math.ceil(tb.getEndTimeJ2K());
327       final double[][] heliList = new double[((int) Math.round(cet - fst)) + 1][];
328       int j = 0;
329       for (int i = (int) Math.round(fst); i <= (int) Math.round(cet); i++) {
330         double[] d = opt.getData(i);
331         if (d == null) {
332           final boolean filled = false;
333           /*
334            * // needed for static importer
335            * if (fillHeli)
336            * {
337            * ResultSet rs = winston.executeQuery("SELECT j2ksec, smin, smax, rcnt, rsam FROM " +
338            * table + " WHERE j2ksec=" + i);
339            * if (rs.next())
340            * {
341            * filled = true;
342            * d = new double[] { rs.getDouble(1), rs.getDouble(2), rs.getDouble(3), rs.getDouble(4),
343            * rs.getDouble(5) };
344            * }
345            * }
346            */
347 
348           if (!filled) {
349             d = new double[] {i, Integer.MAX_VALUE, Integer.MIN_VALUE, 0, 0};
350             winston.getStatement().execute("INSERT IGNORE INTO `" + table + "` VALUES (" + i + ","
351                 + Integer.MAX_VALUE + "," + Integer.MIN_VALUE + ",0,0)");
352           }
353         }
354         heliList[j++] = d;
355       }
356       double st = tb.getStartTimeJ2K();
357       final double dt = 1 / tb.samplingRate();
358       for (int i = 0; i < tb.numSamples(); i++) {
359         j = (int) (Math.floor(st) - fst);
360         final double[] d = heliList[j];
361         final int sample = tb.samples()[i];
362         d[1] = Math.min(d[1], sample);
363         d[2] = Math.max(d[2], sample);
364         d[4] = (d[4] * d[3] + Math.abs(sample)) / (d[3] + 1);
365         d[3] = d[3] + 1;
366         st += dt;
367       }
368 
369       for (int i = 0; i < heliList.length; i++) {
370         final int j2k = (int) Math.round(fst + i);
371         winston.getStatement()
372             .execute("UPDATE `" + table + "` SET smin=" + (int) heliList[i][1] + ", smax="
373                 + (int) heliList[i][2] + ", rcnt=" + (int) heliList[i][3] + ", rsam="
374                 + heliList[i][4] + " WHERE j2ksec=" + j2k);
375         opt.putData(j2k, heliList[i]);
376       }
377       result = createdTable ? InputResult.SUCCESS_CREATED_TABLE : InputResult.SUCCESS;
378     } catch (final Exception e) {
379       LOGGER.error("Could not insert TraceBuf: {}. ({})", tb, e.getLocalizedMessage());
380     }
381     return result;
382   }

527   /**
528    * Purge tables.
529    *
530    * @param channel the channel
531    * @param days the number of days
532    * @param admin the Admin or null to create one if needed.
533    * @return the Admin or null if none.
534    */
535   public Admin purgeTables(final String channel, final int days, Admin admin) {
536     if (days <= 0) {
537       return admin;
538     }
539 
540     final SimpleDateFormat df = new SimpleDateFormat("yyyy_MM_dd");
541     df.setTimeZone(TimeZone.getTimeZone("GMT"));
542     final Date now = new Date(CurrentTime.getInstance().now());
543     final Date then = new Date(now.getTime() - (days * 86400000L));
544 
545     LOGGER.info("Purging '{}' tables before: {}", channel, df.format(then));
546 
547     final List<String> list = getDayTables(channel);
548     if (list == null || list.size() == 0) {
549       LOGGER.info("No day tables found");
550       return admin;
551     }
552 
553     boolean deleted = false;
554     boolean setTime = false;
555     for (final String table : list) {
556       final String ss[] = table.split("\\$\\$");
557 
558       if (df.format(then).compareTo(ss[1]) > 0) {
559         try {
560           if (!winston.useDatabase(channel)) {
561             return admin;
562           }
563 
564           checkTableCache.remove(table);
565 
566           winston.getStatement().execute("DROP TABLE `" + table + "`");
567           winston.getStatement().execute("DROP TABLE `" + ss[0] + "$$H" + ss[1] + "`");
568           deleted = true;
569           LOGGER.info("Deleted table: {}", table);
570         } catch (final Exception e) {
571           LOGGER.error("Could not drop old table: {}.  Are permissions set properly?", channel);
572         }
573       } else {
574         if (deleted) {
575           try {
576             final String nextLowestTable = table;
577             final ResultSet rs = winston.getStatement()
578                 .executeQuery("SELECT MIN(st) FROM `" + nextLowestTable + "`");
579             rs.next();
580             final double t1 = rs.getDouble(1);
581             setTimeSpan(channel, t1, Double.NaN);
582             rs.close();
583             setTime = true;
584           } catch (final Exception e) {
585             LOGGER.error("Could not update span after dropping table: ", channel);
586           }
587         }
588         break;
589       }
590     }
591     if (deleted && !setTime) {
592       // must have deleted all of the tables, just delete the channel entirely
593       LOGGER.info("Permanently deleting channel: {}", channel);
594       if (admin == null) {
595         admin = new Admin(winston);
596       }
597       admin.deleteChannel(channel);
598     }
599     return admin;
600   }

617       return;
618     }
619     try {
620       final ChannelInputOptimizer opt = channelOptimizers.get(code);
621 
622       if (!Double.isNaN(st)) {
623         if (opt != null) {
624           opt.t1 = st;
625         }
626         winston.getStatement().execute("UPDATE `" + winston.databasePrefix
627             + "_ROOT`.channels SET st=" + st + " WHERE code='" + code + "'");
628       }
629       if (!Double.isNaN(et)) {
630         if (opt != null) {
631           opt.t2 = et;
632         }
633         winston.getStatement().execute("UPDATE `" + winston.databasePrefix
634             + "_ROOT`.channels SET et=" + et + " WHERE code='" + code + "'");
635       }
636     } catch (final Exception e) {
637       LOGGER.error("Could not set time span for channel: {}. ({})", code, e.getLocalizedMessage());
638     }
639   }

649   private boolean tableExists(final String code, final String date) {
650     final String table = code + "$$" + date;
651     if (checkTableCache.contains(table)) {
652       return true;
653     }
654 
655     try {
656       final ResultSet rs =
657           winston.getStatement().executeQuery("SELECT COUNT(*) FROM `" + table + "`");
658       final boolean result = rs.next();
659       if (result) {
660         checkTableCache.add(table);
661       }
662       rs.close();
663       return result;
664     } catch (final Exception e) {
665     }
666     // checkTableCache.clear();
667     // return createDayTable(code, date);
668     return false;
669   }

671   public void unlockTables() {
672     if (currentLock == null) {
673       return;
674     }
675     try {
676       winston.getStatement().execute("UNLOCK TABLES");
677       setTimeSpan(currentLock.code, currentLock.t1, currentLock.t2);
678       currentLock = null;
679       writeLocks = false;
680       locks.clear();
681     } catch (final SQLException e) {
682       LOGGER.error("Exception while unlocking tables. ({})", e.getLocalizedMessage());
683     }
684   }
