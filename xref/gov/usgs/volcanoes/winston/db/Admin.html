115 	 * Remove channels from the database
116 	 * @param chx the channel expression
117 	 * @param delay the delay in milliseconds between channels
118 	 */
119 	public void deleteChannels(String chx, long delay)
120 	{
121 		String ch = chx;
122 		try
123 		{
124 			List<String> channelList = channels.getChannelCodes(chx);
125 			if (channelList == null || channelList.size() == 0)
126 			{
127 				winston.getLogger().info("deleteChannels: no channels found (" + chx + ")");
128 			}
129 			else
130 			{
131 				// delete each channel
132 				for (int i = 0; i < channelList.size(); i++)
133 				{
134 					ch = channelList.get(i);
135 					if (i != 0 && delay != 0)
136 						Thread.sleep(delay);
137 					doDeleteChannel(ch);
138 					winston.getLogger().info("Deleted channel: " + ch);
139 				}
140 				ch = chx;
141 			}
142 		}
143 		catch (Exception e)
144 		{
145 			winston.getLogger().log(Level.SEVERE, "Error during deleteChannels(" + ch + ")", e);
146 		}
147 	}

149 	/**
150 	 * Determines if the channel expression is valid. The channel
151 	 * expression used for <code>deleteChannels</code> is restricted
152 	 * and should be verified.
153 	 * @param chx the channel expression.
154 	 * @return true if valid, false otherwise.
155 	 * @see #deleteChannels(String, long)
156 	 */
157 	public boolean isChannelExpressionValid(String chx)
158 	{
159 		boolean valid = true;
160 		int index;
161 		// ensure the station and network do not have wild card
162 		String[] ca = chx.split("\\$");
163 		switch (ca.length)
164 		{
165 		case 3: // station$channel$network
166 			if (ca[0].indexOf(WILDCARD) >= 0
167 					|| ((index = ca[2].indexOf(WILDCARD)) >= 0 && index <= 1))
168 			{
169 				valid = true;
170 			}
171 			break;
172 		case 4: // station$channel$network$location
173 			if (ca[0].indexOf(WILDCARD) >= 0
174 					|| ca[2].indexOf(WILDCARD) >= 0)
175 			{
176 				valid = true;
177 			}
178 			break;
179 		default:
180 			valid = true;
181 			break;
182 		}
183 		return valid;
184 	}

231 	 * Leaving as-is for old importer.
232 	 * @param day
233 	 * @param ch
234 	 * @return true if successful, false otherwise
235 	 */
236 	public boolean repairChannel(String day, String ch)
237 	{
238 		try
239 		{
240 			winston.useDatabase(ch);
241 			boolean fix = false;
242 			Statement st = winston.getStatement();
243 			
244 			winston.getLogger().info("Checking: " + ch + " " + day);
245 			ResultSet rs = st.executeQuery("CHECK TABLE `" + ch + "$$" + day + "` FAST QUICK");
246 			rs.next();
247 			String s = rs.getString("Msg_text");
248 			if (s.endsWith("doesn't exist"))
249 			{
250 				winston.getLogger().info(ch + " wave table doesn't exist.");
251 				return true;
252 			}
253 			if (!s.equals("Table is already up to date"))
254 				fix = true;
255 			
256 			rs = st.executeQuery("CHECK TABLE `" + ch + "$$H" + day + "` FAST QUICK");
257 			rs.next();
258 			s = rs.getString("Msg_text");
259 			if (s.endsWith("doesn't exist"))
260 			{
261 				winston.getLogger().info(ch + " helicorder table doesn't exist.");
262 				return true;
263 			}
264 			// TODO: check table existence
265 			if (!rs.getString("Msg_text").equals("Table is already up to date"))
266 				fix = true;
267 			
268 			if (fix)
269 			{
270 				winston.getLogger().info("Repairing: " + ch);
271 //				winston.getStatement().execute("REPAIR TABLE " + ch + "$$" + day + " QUICK");
272 //				winston.getStatement().execute("REPAIR TABLE " + ch + "$$H" + day + " QUICK");
273 				winston.getStatement().execute("REPAIR TABLE `" + ch + "$$" + day + "`");
274 				winston.getStatement().execute("REPAIR TABLE `" + ch + "$$H" + day + "`");
275 			}
276 			return true;
277 		}
278 		catch (Exception e)
279 		{
280 			winston.getLogger().log(Level.SEVERE, "Failed to repair: " + ch);
281 		}
282 		return false;
283 	}

307 	 * @param database name of database
308 	 * @param table name of table
309 	 * @return true if OK, false otherwise
310 	 * @throws SQLException
311 	 */
312 	public boolean checkTable(String database, String table) throws SQLException
313 	{
314 		ResultSet rs = winston.getStatement().executeQuery("CHECK TABLE `" + table + "` FAST QUICK");
315 		rs.next();
316 		String s = rs.getString("Msg_text");
317 		if (s.endsWith("doesn't exist"))
318 		{
319 			winston.getLogger().info(table + " doesn't exist.");
320 			return false;
321 		}
322 		return s.equals("Table is already up to date");
323 	}

326 	 * Attempts to repair a table.
327 	 * @param database the database
328 	 * @param table the table name
329 	 * @return flag indicating a whether table is healthy
330 	 */
331 	public boolean repairTable(String database, String table)
332 	{
333 		try
334 		{
335 			winston.useDatabase(database);
336 			winston.getLogger().info("Checking table: " + table);
337 			boolean ct = checkTable(database, table);
338 			if (ct == true)
339 				return true;
340 			
341 			winston.getLogger().info("Repairing table: " + table);
342 			winston.getStatement().execute("REPAIR TABLE `" + table + "` QUICK");
343 			ct = checkTable(database, table);
344 			if (ct == true)
345 				return true;
346 
347 			winston.getLogger().info("Still broken, attempting further repair: " + table);
348 			winston.getStatement().execute("REPAIR TABLE `" + table + "` QUICK USE_FRM");
349 			return checkTable(database, table);
350 		}
351 		catch (Exception e)
352 		{
353 			winston.getLogger().log(Level.SEVERE, "Failed to repair: " + table);
354 		}
355 		return false;
356 	}

379 		System.out.println(

382 "Information about connecting to the Winston database must be present\n" +
383 "in Winston.config in the current directory.\n\n" +
384 "Usage:\n" +
385 "  java gov.usgs.winston.db.Admin [options] command [command arguments]\n" +
386 "\nValid options:\n" +
387 "  --delay seconds                 the delay between each channel for commands\n" +
388 "                                  for multiple channels\n" +
389 "\nValid commands:\n" +
390 "  --list                          lists all channels\n" +
391 "  --list times                    lists all channels with time span\n" +
392 "  --delete channel                delete the specified channel\n" +

417 	/** 
418 	 * Main method
419 	 * @param args command-line args
420 	 */
421 	public static void main(String[] args)
422 	{
423 		if (args.length == 0)
424 		{
425 			printUsage(-1, null);
426 		}
427 		
428 		int argIndex = 0;
429 		String cmd = null;
430 		String arg;
431 		int status = 0;
432 		final ConfigFile cf = new ConfigFile("Winston.config");
433 		final String driver = cf.getString("winston.driver");
434 		final String url = cf.getString("winston.url");
435 		final String db = cf.getString("winston.prefix");
436 		final Admin admin = new Admin(driver, url, db);
437 		long delay = 0;
438 		
439 		while (argIndex < args.length && status == 0)
440 		{
441 			arg = args[argIndex++];
442 			// first check for options
443 			if (arg.equals("--delay")) 
444 			{
445 				try
446 				{
447 					arg = args[argIndex++];
448 					// convert seconds to milliseconds
449 					delay = Integer.parseInt(arg) * 1000L;
450 				}
451 				catch (Exception e)
452 				{
453 					status = 2;
454 				}
455 			}
456 			// then check the command and command arguments
457 			else if (cmd != null || (cmd = arg).length() == 0)
458 			{
459 				// already had a command or empty command
460 				status = 2;
461 			}
462 			else if (cmd.equals("--span"))
463 			{
464 				admin.calculateTableSpans();
465 			}
466 			else if (cmd.equals("--list"))
467 			{
468 				boolean times = false;
469 				if (argIndex < args.length)
470 				{
471 					arg = args[argIndex++];
472 					if (arg.equals("times"))
473 						times = true;
474 					else
475 						status = 2;
476 				}
477 				if (status == 0)
478 					admin.listChannels(times);
479 			}
480 			else if (cmd.equals("--delete") || cmd.equals("--deletex"))
481 			{
482 				if (argIndex < args.length)
483 				{
484 					arg = args[argIndex++];
485 					if (cmd.equals("--deletex"))
486 					{
487 						if (admin.isChannelExpressionValid(arg))
488 						{
489 							admin.deleteChannels(arg, delay);
490 						}
491 						else
492 						{
493 							status = 3;
494 							System.err.println(
495 									"channel expresion may not have wild card for network or station ("
496 									+ arg + ")");
497 						}
498 					}
499 					else
500 					{
501 						admin.deleteChannel(arg);
502 					}
503 				}
504 				else
505 					status = 2;
506 			}
507 			else if (cmd.equals("--purge") || cmd.equals("--purgex"))
508 			{
509 				String channel = null;
510 				int days = 0;
511 				try
512 				{
513 					arg = args[argIndex++];
514 					channel = arg;
515 					arg = args[argIndex++];
516 					days = Integer.parseInt(arg);
517 					if (days <= 0)
518 					{
519 						status = 2;
520 					}
521 				}
522 				catch (Exception e)
523 				{
524 					status = 2;
525 				}
526 
527 				if (status == 0)
528 				{
529 					if (cmd.equals("--purgex"))
530 						admin.purgeChannels(channel, days, delay);
531 					else
532 						admin.purge(channel, days);
533 				}
534 			}
535 			else if (cmd.equals("--repair"))
536 			{
537 				if (argIndex < args.length)
538 				{
539 					final String day = args[argIndex++];
540 					final String ch;
541 					if (argIndex < args.length)
542 						ch = args[argIndex++];
543 					else
544 						ch = null;
545 					admin.repair(day, ch);
546 				}
547 				else
548 					status = 2;
549 			}
550 			//		else if (cmd.equals("--deletewinston"))
551 			//		{
552 			//			admin.deleteWinston();
553 			//		}
554 			else
555 			{
556 				System.err.println("Invalid argument(" + argIndex + "): '" + arg + "'");
557 				status = 1;
558 			}
559 		}
560 		
561 		if (status != 0)
562 		{
563 			printUsage(status, cmd);
564 		}
565 	}
