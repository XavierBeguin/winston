79  	public List<String> getWaveServerMenu(boolean scnl, double embargo, double span, double maxDays)
80  	{
81  		if (!winston.checkConnect())
82  			return null;
83  
84  		List<Channel> sts = channels.getChannels();
85  		if (sts == null)
86  			return null;
87  
88  		List<String> list = new ArrayList<String>(sts.size());
89  		for (Channel st : sts)
90  		{
91  			String[] ss = st.getCode().split("\\$");
92  			double[] ts = { st.getMinTime(), st.getMaxTime() };
93  
94  
95  			if (maxDays > 0)
96  				ts[0] = Math.max(ts[0], Util.nowJ2K() - (maxDays * ONE_DAY));
97  			
98  			if (ts != null && ts[0] < ts[1])
99  			{
100 				
101 				if (!scnl && ss.length == 3)
102 					list.add(" " + st.getSID() + " " + ss[0] + " " + ss[1] + " " + ss[2] + " " + decimalFormat.format(Util.j2KToEW(ts[0])) + " " + decimalFormat.format(Util.j2KToEW(ts[1])) + " s4 ");
103 				else if (scnl)
104 				{
105 					String loc = (ss.length == 4 ? ss[3] : "--");
106 					String line = " " + st.getSID() + " " + ss[0] + " " + ss[1] + " " + ss[2] + " " + loc + " " + decimalFormat.format(Util.j2KToEW(ts[0])) + " " + decimalFormat.format(Util.j2KToEW(ts[1])) + " s4 ";
107 					list.add(line);
108 				}
109 			}
110 		}
111 		return list;
112 	}

124 			if (rs.next())

175 	public Object[] getWaveServerRaw(String s, String c, String n, String l, double t1, double t2)
176 	{
177 		String lc = "";
178 		if (l != null && !l.equals("--"))
179 			lc = "$" + l;
180 		String code = s + "$" + c + "$" + n + lc;
181 		if (!winston.checkConnect() || !winston.useDatabase(code))
182 			return null;
183 		List<byte[]> bufs = null;
184 		try{
185 			bufs = data.getTraceBufBytes(code, t1, t2, 0);
186 		} catch (UtilException e){
187 		}
188 		if (bufs == null || bufs.size() == 0)
189 			return null;
190 
191 		try
192 		{
193 			int sid = channels.getChannelID(code);
194 			TraceBuf tb0 = new TraceBuf(bufs.get(0));
195 			TraceBuf tbN = new TraceBuf(bufs.get(bufs.size() - 1));
196 			int total = 0;
197 			for (byte[] buf : bufs)
198 				total += buf.length;
199 			
200 			String lr = "";
201 			if (l != null)
202 				lr = " " + l;
203 			String hdr = sid + " " + s + " " + c + " " + n + lr + " F " + tb0.dataType() + " " + tb0.getStartTime() + " " + tbN.getEndTime() + " " + total;
204 			return new Object[] { hdr, new Integer(total), bufs };
205 		}
206 		catch (Exception e)
207 		{
208 			winston.getLogger().log(Level.SEVERE, "Could not get raw wave.", e);
209 		}
210 		return null;
211 	}
