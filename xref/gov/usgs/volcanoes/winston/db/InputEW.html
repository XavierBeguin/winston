119   private boolean createDayTable(final String code, final String date) {
120     try {
121       final double prevDayJ2k = J2kSec.fromDate(dateFormat.parse(date)) - (24 * 3600);
122       final String prevDate = dateFormat.format(J2kSec.asDate(prevDayJ2k));
123 
124       final String waveTable = code + "$$" + date;
125       final String heliTable = code + "$$H" + date;
126       final String waveTableLast = code + "$$" + prevDate;
127       final String heliTableLast = code + "$$H" + prevDate;
128       final String waveTableall = code + "$$" + "past2days";
129       final String heliTableall = code + "$$H" + "past2days";
130 
131       winston.getStatement()
132           .execute("CREATE TABLE `" + waveTable + "` (st DOUBLE PRIMARY KEY, et DOUBLE, sr DOUBLE, "
133               + "datatype CHAR(3), tracebuf BLOB) " + winston.tableEngine);
134 
135       System.out
136           .println("CREATE TABLE `" + waveTable + "` (st DOUBLE PRIMARY KEY, et DOUBLE, sr DOUBLE, "
137               + "datatype CHAR(3), tracebuf BLOB) " + winston.tableEngine);
138 
139       winston.getStatement().execute(
140           "CREATE TABLE `" + heliTable + "` (j2ksec DOUBLE PRIMARY KEY, smin INT, smax INT, "
141               + "rcnt INT, rsam DOUBLE) " + winston.tableEngine);
142 
143       if (enableValarmView) {
144         // if there is data from the previous day, we want to union it
145         // into our
146         // view, otherwise, setup views into current days data
147         LOGGER.info("Creating VIEWs for VAlarm: {}", heliTableall);
148 
149         String sql =
150             "CREATE or REPLACE VIEW `" + waveTableall + "` AS SELECT * FROM `" + waveTable + "`";
151         if (tableExists(waveTableLast)) {
152           sql += " UNION ALL select * from `" + waveTableLast + "`";
153         }
154         winston.getStatement().execute(sql);
155 
156         sql = "CREATE or REPLACE VIEW `" + heliTableall + "` AS SELECT * FROM `" + heliTable + "`";
157         if (tableExists(heliTableLast)) {
158           sql += " UNION ALL select * from `" + heliTableLast + "`";
159         }
160         winston.getStatement().execute(sql);
161       }
162 
163       return true;
164     } catch (final Exception ex) {
165       LOGGER.error("Could not create day table: '{}${}'. ({})", code, date, ex);
166     }
167     return false;
168   }

170   private List<String> getDayTables(final String code) {
171     final ArrayList<String> list = new ArrayList<String>(10);
172     try {
173       final ResultSet rs = winston.getStatement().executeQuery("SHOW TABLES");
174       while (rs.next()) {
175         list.add(rs.getString(1));
176       }
177       rs.close();
178 
179       Collections.sort(list);
180 
181       final ArrayList<String> dayList = new ArrayList<String>(list.size() / 2);
182       for (final String table : list) {
183         final String day = table.substring(table.indexOf("$$") + 2);
184         if (day.length() == 10 && day.charAt(4) == '_' && day.charAt(7) == '_'
185             && Character.isDigit(day.charAt(0)) && Character.isDigit(day.charAt(9))) {
186           dayList.add(table);
187         }
188       }
189       return dayList;
190     } catch (final Exception e) {
191       LOGGER.error("Could not get list of tables for channel: {}. ({})", code, e);
192     }
193     return null;
194   }

202   /**
203    * Gets a helicorder row. This function MUST be called before
204    * updateHelicorderRow because it is responsible for creating the blank row
205    * if no existing data can be found.
206    *
207    * @param table
208    * @param j2ksec
209    * @return
210    */
211   private double[] getHelicorderRow(final String channel, final double j2ksec,
212       final boolean useDB) {
213     SortedMap<Double, double[]> rows = channelHelicorderRows.get(channel);
214     if (rows == null) {
215       rows = new TreeMap<Double, double[]>();
216       channelHelicorderRows.put(channel, rows);
217     }
218     double[] d = rows.get(j2ksec);
219     if (d != null) {
220       return d;
221     }
222 
223     final String date = dateFormat.format(J2kSec.asDate(j2ksec));
224     final String table = channel + "$$H" + date;
225 
226     if (useDB) {
227       try {
228         final ResultSet rs = winston.getStatement().executeQuery(
229             "SELECT j2ksec, smin, smax, rcnt, rsam FROM `" + table + "` WHERE j2ksec=" + j2ksec);
230         if (rs.next()) {
231           d = new double[] {rs.getDouble(1), rs.getDouble(2), rs.getDouble(3), rs.getDouble(4),
232               rs.getDouble(5), 0, 0, 0};
233         }
234         rs.close();
235       } catch (final Exception e) {
236         LOGGER.warn("Could not get helicorder row: {}", e.getMessage());
237       }
238     }
239 
240     if (d == null) {
241       d = new double[] {j2ksec, Integer.MAX_VALUE, Integer.MIN_VALUE, 0, 0, 0, 0, 0};
242     }
243 
244     rows.put(j2ksec, d);
245     if (rows.size() > maxRows) {
246       for (int i = 0; i < numRowsToDelete; i++) {
247         rows.remove(rows.firstKey());
248       }
249     }
250     return d;
251   }

253   private PreparedStatement getInputStatement(final String table, final TraceBuf tb) {
254     if (tb == null) {
255       LOGGER.error("null tb");
256       return null;
257     }
258 
259     final PreparedStatement insert =
260         winston.getPreparedStatement("INSERT INTO `" + table + "` VALUES (?,?,?,?,?);");
261     if (insert == null) {
262       LOGGER.error("Call to getPreparedStatement returned null.");
263       return null;
264     }
265 
266     try {
267       insert.setDouble(1, tb.getStartTimeJ2K());
268       insert.setDouble(2, tb.getEndTimeJ2K());
269       insert.setDouble(3, tb.samplingRate());
270       insert.setString(4, tb.dataType());
271       final byte[] compressed =
272           Zip.compress(tb.bytes, Deflater.BEST_SPEED, 0, tb.bytes.length - 1);
273       insert.setBytes(5, compressed);
274     } catch (final SQLException e) {
275       LOGGER.error("Could not create prepared statement: {}.({})", tb, e);
276     }
277 
278     return insert;
279   }

281   private double getRSAMMu(final String channel, final double j2ksec, int delta,
282       final int duration) {
283     final SortedMap<Double, double[]> rows = channelHelicorderRows.get(channel);
284     if (rows == null) {
285       return Double.NaN;
286     }
287 
288     delta = -delta;
289     double sampleSum = 0;
290     double sampleCount = 0;
291     for (int k = delta - duration; k < delta; k++) {
292       final double[] hr = rows.get(j2ksec + k);
293       if (hr != null) {
294         sampleSum += hr[HeliFields.MEAN] * hr[HeliFields.RCNT];
295         sampleCount += hr[HeliFields.RCNT];
296       }
297     }
298 
299     if (sampleCount != 0) {
300       return sampleSum / sampleCount;
301     } else {
302       return 0;
303     }
304   }

307     winston.useRootDatabase();
308     if (!channelSid.containsKey(c)) {
309       final ResultSet rs = winston.getStatement().executeQuery(
310           "SELECT sid FROM `" + winston.databasePrefix + "_ROOT`.channels WHERE code='" + c + "'");
311       if (rs.next()) {
312         final int sid = rs.getInt(1);
313         channelSid.put(c, sid);
314       }
315     }
316

350   /**
351    * Inserts multiple metadata updates
352    *
353    * This function returns either null if the input is null or of size 0, if
354    * the Winston connection can not be established, or the channelmetadata
355    * database can not be used (again, bad database). Otherwise it returns a
356    * list of InputResults for each entry.
357    *
358    * @param channel
359    * @param entries
360    * @return the result or null (see above)
361    */
362   public void inputMetadata(final String channel, final Map<String, String> m) {
363 
364     if (!winston.checkConnect() || !winston.useRootDatabase()) {
365       LOGGER.error("Can't update metadata: Can't connect to Winston");
366     } else {
367       try {
368         final int sid = getSid(channel);
369 
370         final PreparedStatement ps = winston.getPreparedStatement(
371             "REPLACE INTO channelmetadata (sid, name, value) VALUES (?,?,?);");
372         final Iterator<String> it = m.keySet().iterator();
373         while (it.hasNext()) {
374           final String name = it.next();
375           ps.setInt(1, sid);
376           ps.setString(2, name);
377           ps.setString(3, m.get(name));
378 
379           ps.executeUpdate();
380           LOGGER.error("Metadata updated for {}: {}={}", channel, name, m.get(name));
381         }
382 
383         m.clear();
384       } catch (final Exception e) {
385         LOGGER.error("Can't update metadata: {}", e.getMessage());
386         e.printStackTrace();
387       }
388     }
389   }

391   /**
392    * Inserts multiple TraceBufs for a single channel into the database. This
393    * uses optimizations that require that all of the TraceBufs be from the
394    * same channel. Using this function with different channels will produce
395    * unstable results.
396    *
397    * This function returns either null if the input is null or of size 0, if
398    * the Winston connection can not be established, if the station time span
399    * can be calculated (bad database), or the channel's database can not be
400    * used (again, bad database). Otherwise it returns a list of InputResults
401    * for each TraceBuf and one that specifies the result of the time span
402    * update.
403    *
404    * @param tbs
405    *          the list of TraceBufs to insert
406    * @return the result or null (see above)
407    */
408   public List<InputResult> inputTraceBufs(final List<TraceBuf> tbs, final boolean computeRsam,
409       final int delta, final int duration) {
410     if (tbs == null || tbs.size() == 0) {
411       return getError(InputResult.Code.ERROR_INPUT);
412     }
413 
414     if (!winston.checkConnect()) {
415       return getError(InputResult.Code.ERROR_NO_WINSTON);
416     }
417 
418     final String channel = tbs.get(0).toWinstonString();
419     final double[] span = getTimeSpan(channel);
420     if (span == null) {
421       return getError(InputResult.Code.ERROR_TIME_SPAN);
422     }
423 
424     final double stBefore = span[0];
425 
426     if (!winston.useDatabase(channel)) {
427       return getError(InputResult.Code.ERROR_DATABASE);
428     }
429 
430     final ArrayList<InputResult> results = new ArrayList<InputResult>(tbs.size() + 1);
431 
432     final SortedSet<Double> modifiedHeliRows = new TreeSet<Double>();
433 
434     final Iterator<TraceBuf> it = tbs.iterator();
435     while (it.hasNext()) {
436       boolean tableCreated = false;
437       final TraceBuf tb = it.next();
438       final InputResult result = new InputResult(InputResult.Code.NO_CODE, tb);
439       if (tb == null) {
440         result.code = InputResult.Code.ERROR_NULL_TRACEBUF;
441       } else if (!tb.toWinstonString().equals(channel)) {
442         result.code = InputResult.Code.ERROR_CHANNEL;
443       }
444 
445       if (result.code != InputResult.Code.NO_CODE) {
446         continue;
447       }
448 
449       final double ts = tb.getStartTimeJ2K();
450       final String date = dateFormat.format(J2kSec.asDate(ts));
451       final String endDate = dateFormat.format(J2kSec.asDate(tb.getEndTimeJ2K() + 1));
452       final String table = channel + "$$" + date;
453 
454       try {
455         if (!tableExists(channel, date)) {
456           createDayTable(channel, date);
457           tableCreated = true;
458         }
459         if (!tableExists(channel, endDate)) {
460           createDayTable(channel, endDate);
461           tableCreated = true;
462         }
463 
464         final PreparedStatement insert = getInputStatement(table, tb);
465 
466         try {
467           insert.executeUpdate();
468         } catch (final SQLException ex) {
469           if (ex.getMessage().startsWith("Duplicate entry")) {
470             result.code = InputResult.Code.ERROR_DUPLICATE;
471           } else {
472             throw ex;
473           }
474         }
475 
476         span[0] = Math.min(span[0], tb.getStartTimeJ2K());
477         span[1] = Math.max(span[1], tb.getEndTimeJ2K());
478 
479         if (tb.samplingRate() > 2 && result.code != InputResult.Code.ERROR_DUPLICATE) {
480           updateHelicorderData(modifiedHeliRows, channel, date, tb, computeRsam, delta, duration,
481               true);
482         }
483       } catch (final SQLException ex) {
484         result.code = InputResult.Code.ERROR_DATABASE;
485         LOGGER.error("Could not insert trace buf: {}", ex);
486       }
487 
488       if (result.code == InputResult.Code.NO_CODE) {
489         if (tableCreated) {
490           result.code = InputResult.Code.SUCCESS_CREATED_TABLE;
491         } else {
492           result.code = InputResult.Code.SUCCESS;
493         }
494       }
495 
496       results.add(result);
497       tableCreated = false;
498     }
499 
500     final InputResult heliResult = new InputResult(InputResult.Code.SUCCESS_HELICORDER, null);
501     final double failed = writeHelicorderData(channel, modifiedHeliRows);
502     if (!Double.isNaN(failed)) {
503       heliResult.code = InputResult.Code.ERROR_HELICORDER;
504       heliResult.failedHeliJ2K = failed;
505     }
506 
507     results.add(heliResult);
508 
509     final InputResult spanResult = new InputResult(InputResult.Code.ERROR_TIME_SPAN, null);
510     try {
511       if (span[0] == stBefore) {
512         setTimeSpan(channel, Double.NaN, span[1]);
513       } else {
514         setTimeSpan(channel, span[0], span[1]);
515       }
516       spanResult.code = InputResult.Code.SUCCESS_TIME_SPAN;
517     } catch (final SQLException ex) {
518       LOGGER.error("Could not set time span for channel: {}. ({})", channel, ex);
519     }
520     results.add(spanResult);
521 
522     return results;
523   }

525   public void purgeTables(final String channel, final int days) {
526     if (days <= 0) {
527       return;
528     }
529 
530     if (!winston.checkConnect()) {
531       return;
532     }
533 
534     if (!winston.useDatabase(channel)) {
535       return;
536     }
537 
538     final List<String> list = getDayTables(channel);
539     if (list == null) {
540       return;
541     }
542 
543     final Date now = new Date(CurrentTime.getInstance().now());
544     final Date then = new Date(now.getTime() - (days * 86400000L));
545     final String thenString = Time.format(WinstonDatabase.WINSTON_TABLE_DATE_FORMAT, then);
546 
547     LOGGER.info("Purging '{}' tables before: {}", channel, thenString);
548 
549     boolean deleted = false;
550     boolean setTime = false;
551     for (final String table : list) {
552       final String ss[] = table.split("\\$\\$");
553 
554       if (thenString.compareTo(ss[1]) > 0) {
555         try {
556           checkTableCache.remove(table);
557           winston.getStatement().execute("DROP TABLE `" + table + "`");
558           winston.getStatement().execute("DROP TABLE `" + ss[0] + "$$H" + ss[1] + "`");
559 
560           deleted = true;
561           LOGGER.info("Deleted table: {}", table);
562         } catch (final Exception e) {
563           LOGGER.error("Could not drop old table: {}.  Are permissions set properly?", channel);
564         }
565       } else {
566         if (deleted) {
567           try {
568             final String nextLowestTable = table;
569             final ResultSet rs = winston.getStatement()
570                 .executeQuery("SELECT MIN(st) FROM `" + nextLowestTable + "`");
571             rs.next();
572             final double t1 = rs.getDouble(1);
573             setTimeSpan(channel, t1, Double.NaN);
574             rs.close();
575             setTime = true;
576           } catch (final Exception e) {
577             LOGGER.error("Could not update span after dropping table: {}", channel);
578           }
579         }
580         break;
581       }
582     }
583     if (deleted && !setTime) {
584       // must have deleted all of the tables, just delete the channel
585       // entirely
586       LOGGER.info("Permanently deleting channel: " + channel);
587       new Admin(winston).deleteChannel(channel);
588     }
589   }

591   public boolean rederive(final List<TraceBuf> tbs, final boolean computeRsam, final int delta,
592       final int duration) {
593     if (tbs == null || tbs.size() == 0) {
594       return false;
595     }
596 
597     if (!winston.checkConnect()) {
598       return false;
599     }
600 
601     final String channel = tbs.get(0).toWinstonString();
602 
603     if (!winston.useDatabase(channel)) {
604       return false;
605     }
606 
607     final SortedSet<Double> modifiedHeliRows = new TreeSet<Double>();
608     final Iterator<TraceBuf> it = tbs.iterator();
609     while (it.hasNext()) {
610       final TraceBuf tb = it.next();
611       if (tb == null) {
612         continue;
613       }
614 
615       if (!tb.toWinstonString().equals(channel)) {
616         continue;
617       }
618 
619       final double ts = tb.getStartTimeJ2K();
620       final String date = dateFormat.format(J2kSec.asDate(ts));
621 
622       updateHelicorderData(modifiedHeliRows, channel, date, tb, computeRsam, delta, duration,
623           false);
624     }
625 
626     return Double.isNaN(writeHelicorderData(channel, modifiedHeliRows));
627   }

664       }
665       winston.getStatement().execute("UPDATE `" + winston.databasePrefix + "_ROOT`.channels SET et="
666           + et + " WHERE code='" + channel + "'");
667     }

689   private boolean tableExists(final String table) {
690 
691     if (checkTableCache.contains(table)) {
692       return true;
693     }
694 
695     try {
696       final ResultSet rs = winston.getStatement().executeQuery("SHOW TABLES LIKE '" + table + "'");
697       final boolean result = rs.next();
698       if (result) {
699         checkTableCache.add(table);
700         rs.close();
701       }
702       return result;
703     } catch (final Exception e) {
704     }
705     return false;
706   }

721   /**
722    * Updates a helicorder row.
723    *
724    * @param channel
725    * @param date
726    * @param tb
727    * @throws SQLException
728    */
729   private void updateHelicorderData(final Set<Double> modifiedRows, final String channel,
730       final String date, final TraceBuf tb, final boolean computeRsam, final int delta,
731       final int duration, final boolean useDB) // throws SQLException
732   {
733     // TODO: optimize/simplify
734     final double fst = Math.floor(tb.getStartTimeJ2K());
735     final double cet = Math.ceil(tb.getEndTimeJ2K());
736     final double[][] heliList = new double[((int) Math.round(cet - fst)) + 1][];
737     int j = 0;
738     for (int i = (int) Math.round(fst); i <= (int) Math.round(cet); i++) {
739       modifiedRows.add((double) i);
740       heliList[j] = getHelicorderRow(channel, i, useDB);
741       if (computeRsam) {
742         final double mu = getRSAMMu(channel, i, delta, duration);
743         heliList[j][HeliFields.MU] = mu;
744       }
745       j++;
746     }
747     double st = tb.getStartTimeJ2K();
748     final double dt = 1 / tb.samplingRate();
749 
750     for (int i = 0; i < tb.numSamples(); i++) {
751       j = (int) (Math.floor(st) - fst);
752       final double[] d = heliList[j];
753       final int sample = tb.samples()[i];
754       d[HeliFields.SMIN] = Math.min(d[HeliFields.SMIN], sample);
755       d[HeliFields.SMAX] = Math.max(d[HeliFields.SMAX], sample);
756       if (computeRsam) {
757         // original RSAM
758         d[HeliFields.RSAM] =
759             (d[HeliFields.RSAM] * d[HeliFields.RCNT] + Math.abs(sample)) / (d[HeliFields.RCNT] + 1);
760         // average of samples
761         d[HeliFields.MEAN] =
762             (d[HeliFields.MEAN] * d[HeliFields.RCNT] + sample) / (d[HeliFields.RCNT] + 1);
763         d[HeliFields.WEIGHTED_RSAM] =
764             (d[HeliFields.WEIGHTED_RSAM] * d[HeliFields.RCNT] + Math.abs(sample - d[HeliFields.MU]))
765                 / (d[HeliFields.RCNT] + 1);
766         d[HeliFields.RCNT]++;
767       }
768       st += dt;
769     }
770   }

772   /**
773    * Writes helicorder data to the database. In case of failure, returns the
774    * j2k that failed. Returns NaN on success.
775    *
776    * @param channel
777    * @param modifiedRows
778    * @return
779    */
780   private double writeHelicorderData(final String channel, final Set<Double> modifiedRows) // throws
781   // SQLException
782   {
783     for (final double j2k : modifiedRows) {
784       final String date = dateFormat.format(J2kSec.asDate(j2k));
785       final String table = channel + "$$H" + date;
786 
787       final double[] row = getHelicorderRow(channel, j2k, false);
788       final String sql = String.format(
789           "INSERT INTO `%s` (j2ksec, smin, smax, rcnt, rsam) "
790               + "VALUES (%f,%d,%d,%d,%f) ON DUPLICATE KEY UPDATE "
791               + "smin=VALUES(smin), smax=VALUES(smax), rcnt=VALUES(rcnt), rsam=VALUES(rsam)",
792           table, j2k, (int) row[HeliFields.SMIN], (int) row[HeliFields.SMAX],
793           (int) row[HeliFields.RCNT], row[HeliFields.WEIGHTED_RSAM]);
794       try {
795         winston.getStatement().execute(sql);
796       } catch (final SQLException ex) {
797         LOGGER.warn("Could not write helicorder row: {}", ex.getMessage());
798         LOGGER.warn("SQL: {}", sql);
799         return j2k;
800       }
801     }
802 
803     return Double.NaN;
804   }
