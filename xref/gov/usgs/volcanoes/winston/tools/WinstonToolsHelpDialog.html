18  public class WinstonToolsHelpDialog extends JDialog {
19    private static final long serialVersionUID = 1L;
20  
21    JButton dismiss;
22    JPanel helpPanel;
23  
24  
25    public WinstonToolsHelpDialog(final WinstonToolsPanel p) {
26      super();
27  
28      final JScrollPane scrollPane = new JScrollPane(getHelpPane(p.getHelpFileName()));
29      scrollPane.setVerticalScrollBarPolicy(ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS);
30      scrollPane.setPreferredSize(new Dimension(500, 500));
31  
32      this.add(scrollPane, BorderLayout.NORTH);
33      dismiss = new JButton("Close");
34      dismiss.addActionListener(new ActionListener() {
35        public void actionPerformed(final ActionEvent e) {
36          dispose();
37        }
38      });
39  
40      this.add(dismiss, BorderLayout.SOUTH);
41      pack();
42    }
43  
44    private JEditorPane getHelpPane(final String s) {
45      JEditorPane editorPane;
46      // s = "file:gov/usgs/winston/tools/" + s + "Help.html";
47      try {
48        editorPane = new JEditorPane(s);
49      } catch (final IOException e) {
50        editorPane = new JEditorPane();
51        try {
52          editorPane.getDocument().insertString(0, "No help available for " + s, null);
53          // editorPane.getDocument().insertString(0, e.getMessage() + s, null);
54        } catch (final BadLocationException e1) {
55        }
56      }
57  
58      return editorPane;
59    }
