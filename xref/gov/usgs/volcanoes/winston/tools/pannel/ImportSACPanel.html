26  public class ImportSACPanel extends WinstonToolsRunnablePanel {
27  
28    private static final long serialVersionUID = 1L;
29    private ImportSAC is;
30  
31    private ScnlPanel scnlP;
32    private FilePanel fileP;
33    private JTextField rsamDeltaF;
34    private JTextField rsamDurationF;
35    private JButton importB;
36  
37    public ImportSACPanel() {
38      super("Import SAC");
39    }
40  
41    @Override
42    protected void createUI() {
43      this.setBorder(BorderFactory.createTitledBorder(BorderFactory.createLineBorder(Color.black),
44          "Import SAC File"));
45  
46      final FormLayout layout = new FormLayout("right:max(40dlu;p), 4dlu, left:p", "");
47  
48      final DefaultFormBuilder builder = new DefaultFormBuilder(layout);
49      builder.setDefaultDialogBorder();
50  
51      builder.append("SCNL", scnlP);
52      builder.nextLine();
53      builder.appendUnrelatedComponentsGapRow();
54      builder.nextLine();
55      builder.append("file", fileP);
56      builder.nextLine();
57      builder.appendUnrelatedComponentsGapRow();
58      builder.nextLine();
59      builder.append("RSAM Delta", rsamDeltaF);
60      builder.nextLine();
61      builder.append("RSAM Duration", rsamDurationF);
62      builder.nextLine();
63      builder.appendUnrelatedComponentsGapRow();
64      builder.nextLine();
65      builder.append("", importB);
66      this.add(builder.getPanel(), BorderLayout.CENTER);
67    }
68  
69    @Override
70    protected void createFields() {
71      scnlP = new ScnlPanel();
72  
73      fileP = new FilePanel(FilePanel.Type.OPEN);
74      rsamDeltaF = new JTextField(5);
75      rsamDeltaF.setText("10");
76      rsamDurationF = new JTextField(5);
77      rsamDurationF.setText("60");
78      importB = new JButton("import");
79      importB.addActionListener(new ActionListener() {
80        public void actionPerformed(final ActionEvent e) {
81          start();
82        }
83      });
84    }
85  
86    @Override
87    protected void go() {
88  
89      final ArrayList<String> as = new ArrayList<String>();
90      as.add("-rd");
91      as.add(rsamDeltaF.getText());
92      as.add("-rl");
93      as.add(rsamDurationF.getText());
94      as.add(fileP.getFileName());
95      final Set<String> kvs = new HashSet<String>();
96      kvs.add("-rd");
97      kvs.add("-rl");
98      kvs.add("-c");
99  
100     final Set<String> flags = new HashSet<String>();
101     final Arguments args = new Arguments(as.toArray(new String[0]), flags, kvs);
102     is = new ImportSAC();
103     is.processArguments(args);
104     is.setChannel(scnlP.getSCNL('$'));
105     final List<String> files = args.unused();
106     StaticImporter.process(files, is);
107   }
108 
109   @Override
110   public boolean needsWinston() {
111     return true;
112   }

114 }
