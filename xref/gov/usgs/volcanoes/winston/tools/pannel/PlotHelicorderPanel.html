42    public class FileTypeActionListener implements ActionListener {
43      public void actionPerformed(final ActionEvent e) {
44        String sn = filePanel.getFileName();
45        final int i = sn.lastIndexOf('.');
46        if (i != -1) {
47          sn = sn.substring(0, i);
48        }
49  
50        sn = sn + "." + e.getActionCommand();
51        filePanel.setFileName(sn);
52      }
53    }
54    public class TimeRangeDocumentListener implements DocumentListener {
55  
56      JTextField f;
57  
58      public TimeRangeDocumentListener(final JTextField f) {
59        this.f = f;
60      }
61  
62      public void changedUpdate(final DocumentEvent e) {}
63  
64      public void insertUpdate(final DocumentEvent e) {
65        f.setBackground(validateTime() ? Color.white : RED);
66      }
67  
68      public void removeUpdate(final DocumentEvent e) {
69        f.setBackground(validateTime() ? Color.white : RED);
70      }
71  
72      private boolean validateTime() {
73        try {
74          Time.parseTimeRange(f.getText());
75          return true;
76        } catch (final ParseException ex) {
77          return false;
78        }
79      }
80    }

145   @Override
146   protected void createFields() {

278   @Override
279   protected void go() {
280     plotB.setEnabled(false);
281 
282     final String server = waveServerF.getText().trim();
283     final int port = Integer.parseInt(portF.getText().trim());
284     final String scnl = scnlPanel.getSCNLasSCNL('_');
285 
286     final String timeZone = (String) timeZones.getSelectedItem();
287     final SimpleDateFormat dateFormat = new SimpleDateFormat(Time.INPUT_TIME_FORMAT);
288     long date = 0;
289     try {
290       date = dateFormat.parse(end.getText()).getTime();
291     } catch (final ParseException e) {
292       e.printStackTrace();
293     }
294 
295     // TODO fix error handling
296     final ArrayList<String> args = new ArrayList<String>();
297 
298     args.add("-wws");
299     args.add(server + ":" + port);
300     args.add("-s");
301     args.add(scnl);
302     args.add("-e");
303     args.add(end.getText());
304     args.add("-m");
305     args.add(chunkList.getSelectedItem().toString());
306     args.add("-h");
307     args.add(spanList.getSelectedItem().toString());
308     args.add("-tz");
309     args.add(timeZone);
310     args.add("-to");
311     args.add("" + TimeZone.getTimeZone(timeZone).getOffset(date));
312     args.add("-x");
313     args.add("1000");
314     args.add("-y");
315     args.add("1000");
316     args.add("-lm");
317     args.add("70");
318     args.add("-rm");
319     args.add("70");
320     args.add("-o");
321     args.add(filePanel.getFileName());
322     if (!clipValue.getText().equalsIgnoreCase("auto")) {
323       args.add("-c");
324       args.add(clipValue.getText());
325     }
326     args.add("-b");
327     args.add(null);
328     args.add("-r");
329     args.add("" + (showClip.isSelected() ? 1 : 0));
330     args.add("-ft");
331     args.add(fileTypeGroup.getSelection().getActionCommand());
332 
333     new PlotHelicorder(args.toArray(new String[0]));
334     System.out.println("Done.");
335 
336     plotB.setEnabled(true);
337   }
