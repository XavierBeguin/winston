216 	protected void go() {
217 		plotB.setEnabled(false);
218 
219 		String server = waveServerF.getText().trim();
220 		int port = Integer.parseInt(portF.getText().trim());
221 		String scnl = scnlPanel.getSCNLasSCNL('_');
222 		
223 		String timeZone = (String)timeZones.getSelectedItem();
224 		SimpleDateFormat dateFormat = new SimpleDateFormat(gov.usgs.util.Time.INPUT_TIME_FORMAT);
225 		Date date = null;
226 		try {
227 			date = dateFormat.parse(end.getText());
228 		} catch (ParseException e) {
229 			e.printStackTrace();
230 		}
231 		
232 		// TODO fix error handling
233 		ArrayList<String> args = new ArrayList<String>();
234 
235 		args.add("-wws");
236 		args.add(server + ":" + port);
237 		args.add("-s");
238 		args.add(scnl);
239 		args.add("-e");
240 		args.add(end.getText());
241 		args.add("-m");
242 		args.add(chunkList.getSelectedItem().toString());
243 		args.add("-h");
244 		args.add(spanList.getSelectedItem().toString());
245 		args.add("-tz");
246 		args.add(timeZone);
247 		args.add("-to");
248 		args.add("" + TimeZone.getTimeZone(timeZone).getOffset(date.getTime()));
249 		args.add("-x");
250 		args.add("1000");
251 		args.add("-y");
252 		args.add("1000");
253 		args.add("-lm");
254 		args.add("70");
255 		args.add("-rm");
256 		args.add("70");
257 		args.add("-o");
258 		args.add(filePanel.getFileName());
259 		if (!clipValue.getText().equalsIgnoreCase("auto"))
260 		{
261 			args.add("-c");
262 			args.add(clipValue.getText());
263 		}
264 		args.add("-b");
265 		args.add(null);
266 		args.add("-r");
267 		args.add("" + (showClip.isSelected() ? 1 : 0));
268 		args.add("-ft");
269 		args.add(fileTypeGroup.getSelection().getActionCommand());
270 
271 		new PlotHelicorder(args.toArray(new String[0]));
272 		System.out.println("Done.");
273 
274 		plotB.setEnabled(true);
275 	}

299 	public class FileTypeActionListener implements ActionListener {
300 		public void actionPerformed(ActionEvent e) {
301 			String sn = filePanel.getFileName();
302 			int i = sn.lastIndexOf('.');
303 			if (i != -1)
304 				sn = sn.substring(0, i);
305 
306 			sn = sn + "." + e.getActionCommand();
307 			filePanel.setFileName(sn);
308 		}
309 	}

311 	public class TimeRangeDocumentListener implements DocumentListener {
312 
313 		JTextField f;
314 
315 		public TimeRangeDocumentListener(JTextField f) {
316 			this.f = f;
317 		}
318 
319 		public void insertUpdate(DocumentEvent e) {
320 			f.setBackground(validateTime() ? Color.white : RED);
321 		}
322 
323 		public void removeUpdate(DocumentEvent e) {
324 			f.setBackground(validateTime() ? Color.white : RED);
325 		}
326 
327 		public void changedUpdate(DocumentEvent e) {
328 		}
329 
330 		private boolean validateTime() {
331 			try {
332 				Time.parseTimeRange(f.getText());
333 				return true;
334 			} catch (ParseException ex) {
335 				return false;
336 			}
337 		}
