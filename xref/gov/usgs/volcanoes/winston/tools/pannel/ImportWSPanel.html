156   @Override
157   protected void go() {
158     String s;
159 
160     importB.setEnabled(false);
161     ws.setQuit(false);
162     final ConfigFile config = new ConfigFile("Winston.config");
163     config.put("createDatabase", "true");
164 
165     s = waveServer.getText().trim() + ":" + port.getText().trim();
166     config.put("waveServer", s);
167 
168     s = (createChannels.isSelected()) ? "true" : "flase";
169     config.put("createChannels", s);
170 
171     config.put("channel", scnlPanel.getSCNL());
172 
173     s = getTimeRange();
174     config.put("timeRange", s);
175 
176     config.put("chunkSize", chunkSize.getText());
177     config.put("chunkDelay", chunkDelay.getText());
178 
179     // config.put("rsam.delta", rsamDelta.getText());
180     // config.put("rsam.duration", rsamDuration.getText());
181 
182     ws.setConfig(config);
183     ws.processConfig();
184     ws.createJobs();
185     ws.startImport();
186     importB.setEnabled(true);
187   }

201   private String getTimeRange() {
202     String s;
203     if (explicitB.isSelected())
204       s = start.getText() + "," + end.getText();
205     else {
206       s = ((TimeRangeOption) rangeList.getSelectedItem()).getValue();
207       s += "," + gov.usgs.util.Time.format(gov.usgs.util.Time.INPUT_TIME_FORMAT, new Date());
208     }
209 
210     return s;
211   }

247   public class TimeRangeDocumentListener implements DocumentListener {
248 
249     JTextField f;
250 
251     public TimeRangeDocumentListener(final JTextField f) {
252       this.f = f;
253     }
254 
255     public void insertUpdate(final DocumentEvent e) {
256       f.setBackground(validateTime() ? Color.white : RED);
257     }
258 
259     public void removeUpdate(final DocumentEvent e) {
260       f.setBackground(validateTime() ? Color.white : RED);
261     }
262 
263     public void changedUpdate(final DocumentEvent e) {}
264 
265     private boolean validateTime() {
266       try {
267         Time.parseTimeRange(f.getText());
268         return true;
269       } catch (final ParseException ex) {
270         return false;
271       }
272     }
