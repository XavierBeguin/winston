16  import gov.usgs.volcanoes.winston.in.metadata.AbstractMetadataImporter;
17  import gov.usgs.volcanoes.winston.in.metadata.ImportDataless;

22  public class ImportInstrumentLocations extends WinstonToolsRunnablePanel {
23  
24  	private static final long serialVersionUID = 1L;
25  	private static final String CONFIG_FILE = "Winston.config";
26  
27  	private FilePanel fileP;
28  	private JButton importB;
29  	private JComboBox fileType;
30  
31  	private enum FileType {
32  		DATALESS_SEED("Dataless SEED", new ImportDataless(CONFIG_FILE)), 
33  		HYPOINVERSE("Hypoinverse", new ImportHypoinverse(CONFIG_FILE));
34  
35  		public final String text;
36  		public final AbstractMetadataImporter importer;
37  
38  		private FileType(String text, AbstractMetadataImporter importer) {
39  			this.text = text;
40  			this.importer = importer;
41  		}
42  
43  		public String toString() {
44  			return text;
45  		}
46  		public AbstractMetadataImporter getImporter() {
47  			return importer;
48  		}
49  
50  	}
51  
52  	public ImportInstrumentLocations() {
53  		super("Instrument Locations");
54  	}
55  
56  	protected void createUI() {
57  		this.setBorder(BorderFactory.createTitledBorder(BorderFactory.createLineBorder(Color.black),
58  				"Update Instrument Locations"));
59  
60  		FormLayout layout = new FormLayout("right:max(40dlu;p), 4dlu, left:p", "");
61  
62  		DefaultFormBuilder builder = new DefaultFormBuilder(layout);
63  		builder.setDefaultDialogBorder();
64  
65  		builder.append("Source file", fileP);
66  		builder.nextLine();
67  		builder.append("File type", fileType);
68  		builder.nextLine();
69  		builder.appendUnrelatedComponentsGapRow();
70  		builder.nextLine();
71  		builder.append("", importB);
72  		this.add(builder.getPanel(), BorderLayout.CENTER);
73  	}
74  
75  	protected void createFields() {
76  		fileP = new FilePanel(FilePanel.Type.OPEN);
77  
78  		fileType = new JComboBox(FileType.values());
79  		fileType.setSelectedItem(FileType.HYPOINVERSE);;
80  		
81  		importB = new JButton("import metadata");
82  		importB.addActionListener(new ActionListener() {
83  			public void actionPerformed(ActionEvent e) {
84  				start();
85  			}
86  		});
87  
88  	}
89  
90  	protected void go() {
91  		AbstractMetadataImporter importer = ((FileType) fileType.getSelectedItem()).getImporter();
92  		importer.updateInstruments(fileP.getFileName());
93  	}
94  
95  	public boolean needsWinston() {
96  		return true;
97  	}
98  
99  }
