122   @Override
123   protected void go() {
124     exportB.setEnabled(false);
125 
126     final String server = waveServerF.getText().trim();
127     final int port = Integer.parseInt(portF.getText().trim());
128     final String scnl = scnlPanel.getSCNL('$');
129     final String file = filePanel.getFileName();
130 
131     double cs;
132     try {
133       cs = Double.parseDouble(chunkSize.getText());
134     } catch (final Exception e) {
135       cs = 0;
136     }
137 
138     double wt;
139     try {
140       wt = Double.parseDouble(waitTime.getText());
141     } catch (final Exception e) {
142       wt = 0;
143     }
144 
145     // TODO fix error handling
146     double[] d;
147 
148     try {
149       d = Time.parseTimeRange(start.getText() + "," + end.getText());
150       System.out.println("Starting export...");
151       if (cs == 0)
152         WWSClient.outputSac(server, port, d[0], d[1], scnl, file);
153       else
154         WWSClient.outputSac(server, port, d[0], d[1], scnl, file, cs, wt);
155     } catch (final ParseException e) {
156     }
157 
158     System.out.println("Done.");
159 
160     exportB.setEnabled(true);
161   }

187   public class TimeRangeDocumentListener implements DocumentListener {
188 
189     JTextField f;
190 
191     public TimeRangeDocumentListener(final JTextField f) {
192       this.f = f;
193     }
194 
195     public void insertUpdate(final DocumentEvent e) {
196       f.setBackground(validateTime() ? Color.white : RED);
197     }
198 
199     public void removeUpdate(final DocumentEvent e) {
200       f.setBackground(validateTime() ? Color.white : RED);
201     }
202 
203     public void changedUpdate(final DocumentEvent e) {}
204 
205     private boolean validateTime() {
206       try {
207         Time.parseTimeRange(f.getText());
208         return true;
209       } catch (final ParseException ex) {
210         return false;
211       }
212     }
213   }
