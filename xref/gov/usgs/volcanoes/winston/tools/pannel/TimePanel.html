94  	public class StandardTimeBox extends JTextField implements Observer 
95  	{
96  
97  		private static final long serialVersionUID = 1L;
98  
99  		public StandardTimeBox() {
100 			super(20);
101 			time.addObserver(this);
102 			this.setToolTipText(gov.usgs.util.Time.STANDARD_TIME_FORMAT);
103 			addActionListener(new ActionListener() {
104 				public void actionPerformed(ActionEvent evt) {
105 					updateTime(getText());
106 				}
107 			});
108 			addFocusListener(new FocusListener(){
109 				public void focusGained(FocusEvent e) {}
110 				public void focusLost(FocusEvent e) {
111 					updateTime(getText());
112 				}});
113 
114 		}
115 
116 		public final void updateTime(String s)
117 		{
118 			double d = gov.usgs.util.Time.parse(gov.usgs.util.Time.STANDARD_TIME_FORMAT, getText());
119 			if (d != 0)
120 				time.setDate(Util.j2KToDate(d));
121 			else
122 				this.setBackground(RED);
123 		}
124 		
125 		public void update(Observable o, Object arg) {
126 			this.setText(""+((Time) o).getStandardTime());
127 			this.setBackground(WHITE);
128 		}
129 	}

167 	public class EwBox extends JTextField implements Observer 
168 	{
169 
170 		private static final long serialVersionUID = 1L;
171 
172 		public EwBox() {
173 			super(20);
174 			time.addObserver(this);
175 			this.setToolTipText("seconds past 1970-01-01 00:00:00 UTC");
176 		
177 			
178 			addActionListener(new ActionListener() {
179 				public void actionPerformed(ActionEvent evt) {
180 					updateTime(getText());
181 				}
182 			});
183 			addFocusListener(new FocusListener(){
184 				public void focusGained(FocusEvent e) {}
185 				public void focusLost(FocusEvent e) {
186 					updateTime(getText());
187 				}});
188 
189 		}
190 		
191 		public final void updateTime(String s)
192 		{
193 			try
194 			{
195 				time.setDate(Util.j2KToDate(Util.ewToJ2K(Double.parseDouble(s))));
196 			} catch (NumberFormatException ex) {
197 				this.setBackground(RED);
198 			}
199 		}
200 
201 		public void update(Observable o, Object arg) {
202 			double d = ((Time) o).getEW();
203 			this.setText(timeFormatter.format(d));
204 			this.setBackground(WHITE);
205 		}
206 	}

250 	public class Time extends Observable {
251 
252 		Date date;
253 		
254 		public Time() {
255 			setDate();
256 		}
257 		
258 		
259 		public void setDate() {
260 			setDate(new Date());
261 		}
262 		
263 		public void setDate(Date d) {
264 			date = d;
265 			setChanged();
266 			notifyObservers();
267 		}
268 		
269 		public double getJ2k()
270 		{
271 			return Util.dateToJ2K(date);
272 		}
273 		
274 		public double getEW()
275 		{
276 			return Util.j2KToEW(Util.dateToJ2K(date));
277 		}
278 
279 		public String getInputTime()
280 		{
281 			return gov.usgs.util.Time.format(gov.usgs.util.Time.INPUT_TIME_FORMAT, date);
282 		}
283 		
284 		public String getStandardTime()
285 		{
286 			return gov.usgs.util.Time.format(gov.usgs.util.Time.STANDARD_TIME_FORMAT, date);
287 		}
