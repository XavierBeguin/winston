3   import java.awt.Color;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   
7   import javax.swing.BorderFactory;
8   import javax.swing.BoxLayout;
9   import javax.swing.JButton;
10  import javax.swing.JTextArea;
11  import javax.swing.JTextField;
12  
13  import com.jgoodies.forms.builder.DefaultFormBuilder;
14  import com.jgoodies.forms.layout.FormLayout;
15  
16  import gov.usgs.volcanoes.winston.db.Upgrade;
17  import gov.usgs.volcanoes.winston.tools.WinstonToolsRunnablePanel;
18  
19  public class AdminPanel extends WinstonToolsRunnablePanel {
20  
21    private static final long serialVersionUID = 1L;
22    Upgrade ug;
23    JTextField currentVersion;
24    JTextArea upgradeDescription;
25    JButton upgradeB;
26  
27    public AdminPanel() {
28      super("Winston Admin");
29    }
30  
31    @Override
32    protected void createUI() {
33  
34  
35      this.setBorder(BorderFactory.createTitledBorder(BorderFactory.createLineBorder(Color.black),
36          "Winston Admin"));
37      setLayout(new BoxLayout(this, BoxLayout.PAGE_AXIS));
38  
39      final FormLayout layout = new FormLayout("left:p, 4dlu, fill:p:G", "");
40  
41      final DefaultFormBuilder builder = new DefaultFormBuilder(layout);
42      builder.setDefaultDialogBorder();
43  
44      createFields();
45      builder.append("Current schema", currentVersion);
46      builder.nextLine();
47      builder.append("Available upgrade", upgradeDescription);
48      builder.nextLine();
49      builder.appendUnrelatedComponentsGapRow();
50      builder.nextLine();
51      builder.append("", upgradeB);
52      add(builder.getPanel());
53    }
54  
55    @Override
56    protected void createFields() {
57      ug = new Upgrade();
58  
59      currentVersion = new JTextField(ug.getCurrentVersion());
60  
61      upgradeDescription = new JTextArea(ug.getUpgraderDescription());
62      upgradeDescription.setLineWrap(true);
63      upgradeDescription.setWrapStyleWord(true);
64  
65      upgradeB = new JButton("upgrade");
66      upgradeB.setEnabled(ug.upgradeAvailable());
67      upgradeB.addActionListener(new ActionListener() {
68        public void actionPerformed(final ActionEvent e) {
69          start();
70        }
71      });
72    }
73  
74    @Override
75    protected void go() {
76      ug.doUpgrade();
77      removeAll();
78      createUI();
79    }
80  
81    @Override
82    public boolean needsWinston() {
83      return true;
84    }
85  }
