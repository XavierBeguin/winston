1   package gov.usgs.volcanoes.winston.tools;
2   
3   import javax.swing.JPanel;
4   import javax.swing.JTextField;
5   
6   public class ScnlPanel extends JPanel {
7   
8     private static final long serialVersionUID = 1L;
9   
10    private final JTextField sta;
11    private final JTextField comp;
12    private final JTextField net;
13    private final JTextField loc;
14  
15    public ScnlPanel() {
16      super();
17  
18      sta = new JTextField(5);
19      add(sta);
20      comp = new JTextField(3);
21      add(comp);
22      net = new JTextField(2);
23      add(net);
24      loc = new JTextField(2);
25      add(loc);
26    }
27  
28    public void setSCNL(final String s) {
29      final String[] scnl = s.split("\\s");
30      sta.setText(scnl[0]);
31      comp.setText(scnl[1]);
32      net.setText(scnl[2]);
33      loc.setText(scnl[3]);
34    }
35  
36    public String getSCNL(final char c) {
37      final String s = sta.getText() + c + comp.getText() + c + net.getText();
38      final String l = loc.getText();
39      if (l.equals("") || l.equals("--"))
40        return s;
41      else
42        return s + c + l;
43    }
44  
45    public String getSCNLasSCNL(final char c) {
46      final String s = sta.getText() + c + comp.getText() + c + net.getText();
47      String l = loc.getText();
48      if (l.equals(""))
49        l = "--";
50  
51      return s + c + l;
52    }
53  
54    public String getSCNL() {
55      return getSCNL(' ');
56    }
57  }
