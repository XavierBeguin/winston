38    private static final Logger LOGGER = LoggerFactory.getLogger(HttpCommandFactory.class);
39  
40    private Class<? extends HttpBaseCommand> clazz;
41    private String commandName;
42  
43    private HttpCommandFactory(Class<? extends HttpBaseCommand> clazz, String commandName) {
44      this.clazz = clazz;
45      this.commandName = commandName;
46    }
47  
48    public Class<? extends HttpBaseCommand> getCommandClass() {
49      return clazz;
50    }
51    
52    public String commandName() {
53      return commandName;
54    }
55    public static HttpBaseCommand get(WinstonDatabasePool databasePool, String command)
56        throws InstantiationException, IllegalAccessException, MalformedCommandException {
57      int cmdEnd = command.indexOf('?');
58      if (cmdEnd != -1) {
59        command = command.substring(0, cmdEnd);
60      }
61      for (HttpCommandFactory cmd : HttpCommandFactory.values()) {
62        if (cmd.toString().toLowerCase().equals(command)) {
63          HttpBaseCommand baseCommand = cmd.clazz.newInstance();
64          baseCommand.databasePool(databasePool);
65          return baseCommand;
66        }
67      }
68      throw new MalformedCommandException();
69    }
70  
71    public static List<String> getNames() {
72      List names = new ArrayList();
73      for (HttpCommandFactory command : HttpCommandFactory.values()) {
74        names.add(command.commandName());
75      }
76      return names;
77    }
