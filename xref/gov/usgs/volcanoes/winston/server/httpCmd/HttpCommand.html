7   package gov.usgs.volcanoes.winston.server.httpCmd;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  import gov.usgs.volcanoes.winston.server.WinstonDatabasePool;
13  
14  public enum HttpCommand {
15  
16    MENU("/menu", HttpMenuCommand.class);
17  
18    private static final Logger LOGGER = LoggerFactory.getLogger(HttpCommand.class);
19  
20    private String command;
21    private Class<? extends HttpBaseCommand> clazz;
22  
23    private HttpCommand(String command, Class<? extends HttpBaseCommand> clazz) {
24      this.command = command;
25      this.clazz = clazz;
26    }
27  
28    public static HttpBaseCommand get(WinstonDatabasePool databasePool, String command)
29        throws InstantiationException, IllegalAccessException {
30      int cmdEnd = command.indexOf('?');
31      if (cmdEnd != -1) {
32        command = command.substring(0, cmdEnd);
33      }
34      for (HttpCommand cmd : HttpCommand.values()) {
35        if (cmd.command.equals(command)) {
36          HttpBaseCommand baseCommand = cmd.clazz.newInstance();
37          baseCommand.databasePool(databasePool);
38          return baseCommand;
39        }
40      }
41      throw new RuntimeException("Unknown HTTP command " + command);
42    }
43  }
