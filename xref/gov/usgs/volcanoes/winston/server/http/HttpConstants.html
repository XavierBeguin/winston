139     map.put("DEFAULT_TC", HELI_TIME_CHUNK);
140     map.put("DEFAULT_BR", HELI_BAR_RANGE);
141     map.put("DEFAULT_SC", HELI_SHOW_CLIP);
142     map.put("DEFAULT_FC", HELI_FORCE_CENTER);
143     map.put("DEFAULT_LB", HELI_LABEL);
144     map.put("DEFAULT_CV", HELI_CLIP_VALUE);
145     map.put("DEFAULT_TZ", TIME_ZONE);
146     
147     map.put("RSAM_HEIGHT", RSAM_HEIGHT);
148     map.put("RSAM_WIDTH", RSAM_WIDTH);
149     map.put("RSAM_DOWN_SAMPLE", RSAM_DOWN_SAMPLE);
150     map.put("RSAM_DOWN_SAMPLE_PERIOD", RSAM_DOWN_SAMPLE_PERIOD);
151     map.put("RSAM_RSAM_PERIOD", RSAM_RSAM_PERIOD);
152     map.put("RSAM_DETREND", RSAM_DETREND);

161     map.put("GAPS_START_TIME", GAPS_START_TIME);
162     map.put("GAPS_END_TIME", GAPS_END_TIME);
163     map.put("GAPS_MINIMUM_DURATION", GAPS_MINIMUM_DURATION);
