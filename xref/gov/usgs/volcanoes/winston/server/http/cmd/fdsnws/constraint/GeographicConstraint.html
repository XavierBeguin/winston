7   public abstract class GeographicConstraint extends FdsnConstraint {
8   
9     public static GeographicConstraint build (Map<String, String> arguments) throws FdsnException {
10      final String latitude = getArg(arguments, "latitude", "lat");
11      final String longitude = getArg(arguments, "longitude", "lon");
12      final String minRadius = arguments.get("minradius");
13      final String maxRadius = arguments.get("maxradius");
14      
15      if (!(latitude == null && longitude == null && minRadius == null && maxRadius == null)) {
16        return new GeographicCircleConstraint(latitude, longitude, minRadius, maxRadius);
17      } else {
18        final String minLatitude = getArg(arguments, "minlatitude", "minLat");
19        final String maxLatitude = getArg(arguments, "maxlatitude", "maxLat");
20        final String minLongitude = getArg(arguments, "minlongitude", "minLon");
21        final String maxLongitude = getArg(arguments, "maxlongitude", "maxLon");
22  
23        return new GeographicSquareConstraint(minLatitude, maxLatitude, minLongitude, maxLongitude);
24      }
25    }
