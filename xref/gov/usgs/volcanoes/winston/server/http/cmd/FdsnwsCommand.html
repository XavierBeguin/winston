36  /**
37   * Return the wave server menu. Similar to earthworm getmenu command.
38   *
39   * @author Tom Parker
40   *
41   */
42  public final class FdsnwsCommand extends HttpBaseCommand {
43  
44    private static final Logger LOGGER = LoggerFactory.getLogger(FdsnwsCommand.class);
45  
46    public static final String VERSION = "1.1.2";
47   
48    /**
49     * Constructor.
50     */
51    public FdsnwsCommand() {
52      super();
53    }
54  
55  
56    /**
57     * Execute the command.
58     * 
59     * @param ctx handler context
60     * @param request HTTP request
61     * @throws UtilException when things go wrong
62     */
63    public void doCommand(ChannelHandlerContext ctx, FullHttpRequest request) throws UtilException {
64      String[] splits = request.getUri().split("/");
65      
66      if (splits.length == 5) {
67        dispatch(splits[2], ctx, request);
68      } else if (splits.length == 4) {
69        sendUsage(splits[2], ctx, request);
70      } else {
71        ErrorResponse error = new ErrorResponse(ctx);
72        error.request(request);
73        error.version(VERSION);
74        error.status(HttpResponseStatus.BAD_REQUEST);
75        error.shortDescription("Bad request");
76        error.detailedDescription("Request cannot be parsed.");
77        error.sendError();
78      }
79    }
80  
81    private void sendUsage(String service, ChannelHandlerContext ctx, FullHttpRequest request) {
82      Map<String, Object> root = new HashMap<String, Object>();
83      root.put("service", service);
84      root.put("baseUrl", "http://" + ctx.channel().localAddress().toString().substring(1) + "/");
85      root.put("UrlBuilderTemplate", service + "_UrlBuilder");
86      root.put("InterfaceDescriptionTemplate", service + "_InterfaceDescription");
87      root.put("versionString", Version.VERSION_STRING);
88  
89  
90      try {
91        HttpTemplateConfiguration cfg = HttpTemplateConfiguration.getInstance();
92        Template template = cfg.getTemplate("fdsnws/usage.ftl");
93        
94        Writer sw = new StringWriter();
95        template.process(root, sw);
96        String html = sw.toString();
97        sw.close();
98  
99        FullHttpResponse response = new DefaultFullHttpResponse(request.getProtocolVersion(),
100           HttpResponseStatus.OK, Unpooled.copiedBuffer(html, Charset.forName("UTF-8")));
101       response.headers().set(HttpHeaders.Names.CONTENT_LENGTH, html.length());
102       response.headers().set(HttpHeaders.Names.CONTENT_TYPE, "text/html; charset=UTF-8");
103 
104       if (HttpHeaders.isKeepAlive(request)) {
105         response.headers().set(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.KEEP_ALIVE);
106       }
107       ctx.writeAndFlush(response);
108     } catch (IOException e) {
109       LOGGER.error(e.getLocalizedMessage());
110     } catch (TemplateException e) {
111       LOGGER.error(e.getLocalizedMessage());
112     }
113 
114   }
115   
116   private void dispatch(String service, ChannelHandlerContext ctx, FullHttpRequest request) throws UtilException {
117     ErrorResponse error;
118     switch (service) {
119       case "dataselect":
120         try {
121           DataselectService.dispatch(databasePool, ctx, request);
122         } catch (FdsnException e) {
123           throw new UtilException(e.getLocalizedMessage());
124         }
125         break;
126       case "station":
127         try {
128           StationService.dispatch(databasePool, ctx, request);
129         } catch (FdsnException e) {
130           throw new UtilException(e.getLocalizedMessage());
131         }
132         break;
133       case "event":
134         error = new ErrorResponse(ctx);
135         error.request(request);
136         error.version(VERSION);
137         error.status(HttpResponseStatus.NOT_IMPLEMENTED);
138         error.shortDescription("Not implemented");
139         error.detailedDescription("Winston does not implement the event service");
140         error.sendError();
141         break;
142       default:
143         error = new ErrorResponse(ctx);
144         error.request(request);
145         error.version(VERSION);
146         error.status(HttpResponseStatus.BAD_REQUEST);
147         error.shortDescription("Bad request");
148         error.detailedDescription("Service cannot be parsed.");
149         error.sendError();
150     }
151   }
