8   import java.text.ParseException;
9   import java.util.Map;

11  import gov.usgs.earthworm.message.TraceBuf;
12  import gov.usgs.volcanoes.core.time.J2kSec;

14  import gov.usgs.volcanoes.winston.server.http.cmd.fdsnws.FdsnException;

16  /**
17   * Constrain results to a time window.
18   * 
19   * @author Tom Parker
20   *
21   */
22  public class TimeWindowConstraint extends TimeConstraint {
23  
24    private double startBefore;
25    private double startAfter;
26    private double endBefore;
27    private double endAfter;
28  
29    /**
30     * Constructor.
31     * 
32     * @param startBefore latest start
33     * @param startAfter earliest start
34     * @param endBefore latest end
35     * @param endAfter earliest end
36     * @throws FdsnException when things go wrong
37     */
38    public TimeWindowConstraint(final String startBefore, final String startAfter,
39        final String endBefore, final String endAfter) throws FdsnException {
40      super();
41      try {
42        this.startBefore = dateStringToDouble(startBefore, FAR_IN_FUTURE);
43        this.startAfter = dateStringToDouble(startAfter, FAR_IN_PAST);
44        this.endBefore = dateStringToDouble(endBefore, FAR_IN_FUTURE);
45        this.endAfter = dateStringToDouble(endAfter, FAR_IN_PAST);
46      } catch (final ParseException e) {
47        throw new FdsnException(400, "Can't parse time constraint: " + this);
48      }
49  
50    }
51  
52    public boolean matches(final Channel chan) {
53      final double start = chan.getMinTime();
54      if (start > startBefore || start < startAfter)
55        return false;
56  
57      final double end = chan.getMaxTime();
58      if (end > endBefore || end < endAfter)
59        return false;
60  
61      return true;
62    }
63  
64    @Override
65    public String toString() {
66      return "FdsnTimeWindowConstraint: " + J2kSec.toDateString(startBefore) + " : "
67          + J2kSec.toDateString(startAfter) + " : " + J2kSec.toDateString(endBefore) + " : "
68          + J2kSec.toDateString(endAfter);
69    }
70  
71    public static TimeWindowConstraint build(Map<String, String> arguments) throws FdsnException {
72      final String startBefore = arguments.get("startbefore");
73      final String startAfter = arguments.get("startafter");
74      final String endBefore = arguments.get("endbefore");
75      final String endAfter = arguments.get("andafter");
76      return new TimeWindowConstraint(startBefore, startAfter, endBefore, endAfter);
77    }
78  
79  
80  }
