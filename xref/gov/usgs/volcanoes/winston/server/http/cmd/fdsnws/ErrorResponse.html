6   package gov.usgs.volcanoes.winston.server.http.cmd.fdsnws;
7   
8   import java.nio.charset.Charset;
9   import java.util.Date;
10  
11  import io.netty.buffer.Unpooled;
12  import io.netty.channel.ChannelHandlerContext;
13  import io.netty.handler.codec.http.DefaultFullHttpResponse;
14  import io.netty.handler.codec.http.FullHttpRequest;
15  import io.netty.handler.codec.http.FullHttpResponse;
16  import io.netty.handler.codec.http.HttpHeaders;
17  import io.netty.handler.codec.http.HttpResponseStatus;
18  
19  /**
20   * Return a properly formatted error to the client.
21   * 
22   * @author Tom Parker
23   *
24   */
25  public class ErrorResponse {
26  
27    private final ChannelHandlerContext ctx;
28    private FullHttpRequest request;
29    private HttpResponseStatus status;
30    private String shortDescription;
31    private String detailedDescription;
32    private String version;
33  
34    /**
35     * Constructor.
36     * 
37     * @param ctx handler context
38     */
39    public ErrorResponse(ChannelHandlerContext ctx) {
40      this.ctx = ctx;
41    }
42  
43    /**
44     * Request mutator.
45     * 
46     * @param request http request
47     * @return this for the fluent fans
48     */
49    public ErrorResponse request(FullHttpRequest request) {
50      this.request = request;
51      return this;
52    }
53  
54    /**
55     * status mutator.
56     * 
57     * @param status response status
58     * @return this for the fluent fans
59     */
60    public ErrorResponse status(HttpResponseStatus status) {
61      this.status = status;
62      return this;
63    }
64  
65    /**
66     * error mutator.
67     * 
68     * @param shortDescription short description
69     * @return this for the fluent fans
70     */
71    public ErrorResponse shortDescription(String shortDescription) {
72      this.shortDescription = shortDescription;
73      return this;
74    }
75  
76    /**
77     * detailed description mutator.
78     * 
79     * @param detailedDescription detailed description
80     * @return this for the fluent fans
81     */
82    public ErrorResponse detailedDescription(String detailedDescription) {
83      this.detailedDescription = detailedDescription;
84      return this;
85    }
86  
87    /**
88     * version mutator.
89     * 
90     * @param version version
91     * @return this for the fluent fans
92     */
93    public ErrorResponse version(String version) {
94      this.version = version;
95      return this;
96    }
97  
98    /**
99     * send the error, thorwing IllegalState when not properly initalized.
100    */
101   public void sendError() {
102     if (request == null || status == null || shortDescription == null || detailedDescription == null
103         || version == null || status == null)
104       throw new IllegalStateException();
105 
106     StringBuffer sb = new StringBuffer();
107     sb.append("Error ").append(status.code()).append(": ").append(shortDescription);
108     sb.append("\n\n");
109     sb.append(detailedDescription);
110     sb.append("\n\n");
111     sb.append("Request:\n").append(request.getUri());
112     sb.append("\n\n");
113     sb.append("Request Submitted:\n").append(FdsnwsDate.toString(new Date()));
114     sb.append("\n\n");
115     sb.append("Service version:\n").append(version);
116     String html = sb.toString();
117 
118     FullHttpResponse response = new DefaultFullHttpResponse(request.getProtocolVersion(), status,
119         Unpooled.copiedBuffer(html, Charset.forName("UTF-8")));
120     response.headers().set(HttpHeaders.Names.CONTENT_LENGTH, html.length());
121     response.headers().set(HttpHeaders.Names.CONTENT_TYPE, "text/plain; charset=UTF-8");
122 
123     if (HttpHeaders.isKeepAlive(request)) {
124       response.headers().set(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.KEEP_ALIVE);
125     }
126     ctx.writeAndFlush(response);
127   }
128 }
