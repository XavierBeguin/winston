64    private static final String VERSION = "1.1.2";
65    private static final String SERVICE = "station";
66    private static final String DEFAULT_LEVEL = "station";
67  
68    private static final Logger LOGGER = LoggerFactory.getLogger(StationService.class);

71      version = VERSION;

75    /**
76     * Fulfill the request.
77     * 
78     * @param databasePool winston database pool
79     * @param ctx handler context
80     * @param request the request
81     * @throws UtilException when things go wrong
82     * @throws FdsnException when FDSN-WS spec violated
83     */
84    public static void dispatch(WinstonDatabasePool databasePool, ChannelHandlerContext ctx,
85        FullHttpRequest request) throws UtilException, FdsnException {
86      String method = request.getUri().split("/")[4];
87      int queryStringIdx = method.indexOf('?');
88      if (queryStringIdx != -1) {
89        method = method.substring(0, queryStringIdx);
90      }
91  
92      switch (method) {
93        case "version":
94          sendVersion(ctx, request);
95          break;
96        case "application.wadl":
97          sendWadl(ctx, request);
98          break;
99        case "query":
100         sendQueryResponse(databasePool, ctx, request);
101         break;
102       default:
103         ErrorResponse error = new ErrorResponse(ctx);
104         error.request(request);
105         error.version(VERSION);
106         error.status(HttpResponseStatus.BAD_REQUEST);
107         error.shortDescription("Bad request");
108         error.detailedDescription("Method cannot be parsed.");
109         error.sendError();
110     }
111   }

113   private static void sendQueryResponse(WinstonDatabasePool databasePool, ChannelHandlerContext ctx,
114       FullHttpRequest request) throws UtilException, FdsnException {
115     Map<String, String> arguments = parseRequest(request);
116     final String level = StringUtils.stringToString(arguments.get("level"), DEFAULT_LEVEL);
117 
118     List<FdsnConstraint> constraints = buildConstraints(arguments);
119     LOGGER.debug("got constraints");
120 
121     List<Channel> channels = getChannels(databasePool);
122     if (channels == null) {
123       ErrorResponse error = new ErrorResponse(ctx);
124       error.request(request);
125       error.version(VERSION);
126       error.status(HttpResponseStatus.NOT_FOUND);
127       error.shortDescription("No data");
128       error.detailedDescription("No matching data found.");
129       error.sendError();
130       return;
131     }
132 
133     final DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
134     try {
135       DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
136       final Document doc = dBuilder.newDocument();
137       Element rootElement = createRootElement(doc);
138 
139       Element networkElement = null;
140       Element stationElement = null;
141       double stationStart = Double.MAX_VALUE;
142       double stationEnd = -Double.MAX_VALUE;
143       String net = null;
144       String sta = null;
145 
146       for (final Channel c : channels) {
147         if (pruneChannel(constraints, c)) {
148           continue;
149         }
150 
151         if (!c.network.equals(net)) {
152           net = c.network;
153           networkElement = doc.createElement("Network");
154           networkElement.setAttribute("code", net);
155           rootElement.appendChild(networkElement);
156         }
157 
158         if ("network".equals(level))
159           continue;
160 
161         if (!c.station.equals(sta)) {
162           if (stationElement != null) {
163             final String s = FdsnwsDate.toString(J2kSec.asDate(stationStart));
164             stationElement.setAttribute("startDate", s);
165             stationStart = Double.MAX_VALUE;
166 
167             final String e = FdsnwsDate.toString(J2kSec.asDate(stationEnd));
168             stationElement.setAttribute("endDate", e);
169             stationEnd = -Double.MAX_VALUE;
170 
171             final Element creationElement = doc.createElement("CreationDate");
172             creationElement
173                 .appendChild(doc.createTextNode(FdsnwsDate.toString(J2kSec.asDate(stationStart))));
174           }
175 
176           sta = c.station;
177           stationElement = createStationElement(c, doc);
178           networkElement.appendChild(stationElement);
179         }
180 
181         stationStart = Math.min(stationStart, c.getMinTime());
182         stationEnd = Math.max(stationEnd, c.getMaxTime());
183 
184         if (!"station".equals(level))
185           stationElement.appendChild(createChannelElement(c, doc));
186       }
187 
188 
189       final String responseXml = serialize(doc);
190       FullHttpResponse response = new DefaultFullHttpResponse(request.getProtocolVersion(),
191           HttpResponseStatus.OK, Unpooled.copiedBuffer(responseXml, Charset.forName("UTF-8")));
192       response.headers().set(HttpHeaders.Names.CONTENT_LENGTH, responseXml.length());
193       response.headers().set(HttpHeaders.Names.CONTENT_TYPE, "application/xml; charset=UTF-8");
194 
195       if (HttpHeaders.isKeepAlive(request)) {
196         response.headers().set(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.KEEP_ALIVE);
197       }
198       ctx.writeAndFlush(response);
199     } catch (final ParserConfigurationException e) {
200       e.printStackTrace();
201     }
202 
203   }

206    private static Element createStationElement(final Channel c, final Document doc) {
207     final Element station = doc.createElement("Station");
208     station.setAttribute("code", c.station);
209 
210     Element e;
211     e = doc.createElement("Latitude");
212     e.appendChild(doc.createTextNode("" + c.getInstrument().getLatitude()));
213     station.appendChild(e);
214 
215     e = doc.createElement("Longitude");
216     e.appendChild(doc.createTextNode("" + c.getInstrument().getLongitude()));
217     station.appendChild(e);
218 
219     e = doc.createElement("Elevation");
220     e.appendChild(doc.createTextNode("" + c.getInstrument().getHeight()));
221     station.appendChild(e);
222 
223     e = doc.createElement("Site");
224     final Element n = doc.createElement("Name");
225     n.appendChild(doc.createTextNode(c.getInstrument().getDescription()));
226     e.appendChild(n);
227     station.appendChild(e);
228 
229     return station;
230   }

232   private static Element createChannelElement(final Channel c, final Document doc) {
233     final Element channelElement = doc.createElement("Channel");
234     final String loc = c.location.equals("--") ? "  " : c.location;
235     channelElement.setAttribute("locationCode", loc);
236     channelElement.setAttribute("code", c.channel);
237 
238     Element e;
239     e = doc.createElement("Latitude");
240     e.appendChild(doc.createTextNode("" + c.getInstrument().getLatitude()));
241     channelElement.appendChild(e);
242 
243     e = doc.createElement("Longitude");
244     e.appendChild(doc.createTextNode("" + c.getInstrument().getLongitude()));
245     channelElement.appendChild(e);
246 
247     e = doc.createElement("Elevation");
248     e.appendChild(doc.createTextNode("" + c.getInstrument().getHeight()));
249     channelElement.appendChild(e);
250 
251     e = doc.createElement("Depth");
252     e.setAttribute("Supported", "no");
253     e.appendChild(doc.createTextNode("" + 0));
254     channelElement.appendChild(e);
255 
256     return channelElement;
257   }

259   private static boolean pruneChannel(List<FdsnConstraint> constraints, final Channel c) {
260     boolean prune = false;
261     Iterator<FdsnConstraint> it = constraints.iterator();
262     while (!prune && it.hasNext()) {
263       if (!it.next().matches(c))
264         prune = true;
265     }
266     return prune;
267   }

269   protected static List<FdsnConstraint> buildConstraints(Map<String, String> arguments)
270       throws FdsnException {
271     List<FdsnConstraint> constraints = new ArrayList<FdsnConstraint>();
272     constraints.add(ChannelConstraint.build(arguments));
273     constraints.add(TimeWindowConstraint.build(arguments));
274     constraints.add(GeographicConstraint.build(arguments));
275     constraints.addAll(ChannelConstraint.buildMulti(arguments));
276     
277     return constraints;
278   }
