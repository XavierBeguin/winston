65  /**
66   * Return the wave server menu. Similar to earthworm getmenu command.
67   *
68   * @author Tom Parker
69   *
70   */
71  public final class StatusCommand extends HttpBaseCommand {
72  
73    private static final Logger LOGGER = LoggerFactory.getLogger(StatusCommand.class);
74  
75    private static final AttributeKey<ConnectionStatistics> connectionStatsKey;
76  
77    static {
78      connectionStatsKey = AttributeKey.valueOf("connectionStatistics");
79    }
80  
81    public StatusCommand() {
82      super();
83    }
84  
85  
86    public void doCommand(ChannelHandlerContext ctx, FullHttpRequest request) throws UtilException {
87      StringBuffer error = new StringBuffer();
88  
89       final DecimalFormat formatter = new DecimalFormat("#.##");
90      final double now = J2kSec.fromEpoch(System.currentTimeMillis());
91  
92      // get and sort menu
93      List<Channel> sts;
94      try {
95        sts = databasePool.doCommand(new WinstonConsumer<List<Channel>>() {
96  
97          public List<Channel> execute(WinstonDatabase winston) throws UtilException {
98            return new Channels(winston).getChannelsByLastInsert();
99          }
100 
101       });
102     } catch (Exception e) {
103       throw new UtilException(e.getMessage());
104     }
105 
106     for (int i = 0; i < sts.size(); i++) {
107       final Channel chan = sts.get(i);
108     }
109 
110     final double medianDataAge = now - sts.get(sts.size() / 2).getMaxTime();
111 
112     final Map<String, Integer> oneMinChannels = new HashMap<String, Integer>();
113     final Map<String, Integer> fiveMinChannels = new HashMap<String, Integer>();
114     final Map<String, Integer> oneHourChannels = new HashMap<String, Integer>();
115     final Map<String, Integer> oneDayChannels = new HashMap<String, Integer>();
116     final Map<String, Integer> oneMonthChannels = new HashMap<String, Integer>();
117     final Map<String, Integer> ancientChannels = new HashMap<String, Integer>();
118 
119     for (final Channel chan : sts) {
120       final double age = now - chan.getMaxTime();
121       final String code = chan.getCode().replace('$', '_');
122       if (age < 60)
123         oneMinChannels.put(code, (int) age);
124       else if (age <= 60 * 5)
125         fiveMinChannels.put(code, (int) age);
126       else if (age <= 60 * 60)
127         oneHourChannels.put(code, (int) age);
128       else if (age <= 60 * 60 * 24)
129         oneDayChannels.put(code, (int) age);
130       else if (age <= 60 * 60 * 24 * 7 * 4)
131         oneMonthChannels.put(code, (int) age);
132       else
133         ancientChannels.put(code, (int) age);
134     }
135 
136     Map<String, Object> root = new HashMap<String, Object>();
137     root.put("oneMinChannels", oneMinChannels);
138     root.put("fiveMinChannels", fiveMinChannels);
139     root.put("oneHourChannels", oneHourChannels);
140     root.put("oneDayChannels", oneDayChannels);
141     root.put("oneMonthChannels", oneMonthChannels);
142     root.put("ancientChannels", ancientChannels);
143     
144     root.put("channelCount", sts.size());
145     root.put("medianDataAge", formatter.format(medianDataAge));
146     
147     ConnectionStatistics connectionStatistics = ctx.channel().attr(connectionStatsKey).get();
148     root.put("connectionCount", connectionStatistics.getOpen());
149     
150     final Channel chan = sts.get(0);
151     root.put("mostRecentChan", chan.getCode().replace('$', '_'));
152     root.put("mostRecentTime", formatter.format(now - chan.getMaxTime()));
153 
154     try {
155       HttpTemplateConfiguration cfg = HttpTemplateConfiguration.getInstance();
156       Template template = cfg.getTemplate("status.ftl");
157       Writer sw = new StringWriter();
158       template.process(root, sw);
159       String html = sw.toString();
160       sw.close();
161 
162       FullHttpResponse response = new DefaultFullHttpResponse(request.getProtocolVersion(),
163           HttpResponseStatus.OK, Unpooled.copiedBuffer(html, Charset.forName("UTF-8")));
164       response.headers().set(HttpHeaders.Names.CONTENT_LENGTH, html.length());
165       response.headers().set(HttpHeaders.Names.CONTENT_TYPE, "text/html; charset=UTF-8");
166 
167       if (HttpHeaders.isKeepAlive(request)) {
168         response.headers().set(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.KEEP_ALIVE);
169       }
170       ctx.writeAndFlush(response);
171     } catch (IOException e) {
172       LOGGER.error(e.getLocalizedMessage());
173     } catch (TemplateException e) {
174       LOGGER.error(e.getLocalizedMessage());
175     }
176 
177   }
