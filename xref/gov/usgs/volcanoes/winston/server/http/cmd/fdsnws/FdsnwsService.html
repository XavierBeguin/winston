65    protected static void sendVersion(ChannelHandlerContext ctx, FullHttpRequest request) {
66      FullHttpResponse response = new DefaultFullHttpResponse(request.getProtocolVersion(),
67          HttpResponseStatus.OK, Unpooled.copiedBuffer(version, Charset.forName("UTF-8")));
68      response.headers().set(HttpHeaders.Names.CONTENT_LENGTH, version.length());
69      response.headers().set(HttpHeaders.Names.CONTENT_TYPE, "text/plain; charset=UTF-8");
70  
71      if (HttpHeaders.isKeepAlive(request)) {
72        response.headers().set(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.KEEP_ALIVE);
73      }
74      ctx.writeAndFlush(response);
75    }
76  
77    protected static void sendWadl(ChannelHandlerContext ctx, FullHttpRequest request) {
78      Map<String, Object> root = new HashMap<String, Object>();
79      root.put("host", ctx.channel().localAddress().toString().substring(1));
80  
81  
82      try {
83        HttpTemplateConfiguration cfg = HttpTemplateConfiguration.getInstance();
84        Template template = cfg.getTemplate("fdsnws/" + service + "_application.ftl");
85  
86        Writer sw = new StringWriter();
87        template.process(root, sw);
88        String xml = sw.toString();
89        sw.close();
90  
91        FullHttpResponse response = new DefaultFullHttpResponse(request.getProtocolVersion(),
92            HttpResponseStatus.OK, Unpooled.copiedBuffer(xml, Charset.forName("UTF-8")));
93        response.headers().set(HttpHeaders.Names.CONTENT_LENGTH, xml.length());
94        response.headers().set(HttpHeaders.Names.CONTENT_TYPE, "application/xml; charset=UTF-8");
95  
96        if (HttpHeaders.isKeepAlive(request)) {
97          response.headers().set(HttpHeaders.Names.CONNECTION, HttpHeaders.Values.KEEP_ALIVE);
98        }
99        ctx.writeAndFlush(response);
100     } catch (IOException e) {
101       LOGGER.error(e.getLocalizedMessage());
102     } catch (TemplateException e) {
103       LOGGER.error(e.getLocalizedMessage());
104     }
105   }

108   protected static Element createRootElement(final Document doc) {
109     final Element rootElement = doc.createElement("FDSNStationXML");
110     rootElement.setAttribute("schemaVersion", "1.0");
111     rootElement.setAttribute("xsi:schemaLocation",
112         "http://www.fdsn.org/xml/station/fdsn-station-1.0.xsd");
113     rootElement.setAttribute("xmlns:xsi", "http://www.w3.org/2001/XMLSchema-instance");
114     rootElement.setAttribute("xmlns", "http://www.fdsn.org/xml/station/1");
115     doc.appendChild(rootElement);
116 
117     final Element source = doc.createElement("source");
118     source.appendChild(doc.createTextNode("Winston"));
119     rootElement.appendChild(source);
120 
121     final Element module = doc.createElement("module");
122     module.appendChild(doc.createTextNode("Winston " + Version.VERSION_STRING));
123     rootElement.appendChild(module);
124 
125     final Element created = doc.createElement("Created");
126     created.appendChild(doc.createTextNode(FdsnwsDate.toString(new Date())));
127     rootElement.appendChild(created);
128     
129     return rootElement;
130   }

132   protected static String serialize(final Document doc) {
133     final DOMImplementation impl = doc.getImplementation();
134     final DOMImplementationLS implLS = (DOMImplementationLS) impl.getFeature("LS", "3.0");
135     final LSSerializer lsSerializer = implLS.createLSSerializer();
136     lsSerializer.getDomConfig().setParameter("format-pretty-print", true);
137 
138     final LSOutput lsOutput = implLS.createLSOutput();
139     lsOutput.setEncoding("UTF-8");
140     final Writer stringWriter = new StringWriter();
141     lsOutput.setCharacterStream(stringWriter);
142     lsSerializer.write(doc, lsOutput);
143     return stringWriter.toString();
144   }

146   protected static String getArg(Map<String, String> arguments, final String s1, final String s2) {
147     String arg = arguments.get(s1);
148     if (arg == null)
149       arg = arguments.get(s2);
150 
151     return arg;
152   }

154   protected static Map<String, String> parseRequest(FullHttpRequest request) {
155     Map<String, String> arguments = new HashMap<String, String>();
156 
157 
158     if (request.getMethod() == HttpMethod.GET) {
159       QueryStringDecoder decoder = new QueryStringDecoder(request.getUri());
160 
161       for (String name : decoder.parameters().keySet()) {
162         arguments.put(name, decoder.parameters().get(name).get(0));
163         LOGGER.info("{} : {}", name, decoder.parameters().get(name).get(0));
164       }
165     } else if (request.getMethod() == HttpMethod.POST) {
166       String[] lines = request.content().toString(CharsetUtil.UTF_8).split("\n");
167       StringBuffer chans = new StringBuffer();
168       for (String list : lines) {
169         int idx = list.indexOf('=');
170         if (idx != -1) {
171           arguments.put(list.substring(0, idx), list.substring(idx, list.length()));
172         } else {
173           chans.append(list);
174         }
175       }
176       if (chans.length() > 0) {
177         arguments.put("chans", chans.toString());
178       }
179     }
180 
181     return arguments;
182   }

185   protected static List<Channel> getChannels(WinstonDatabasePool databasePool)
186       throws UtilException {
187     List<Channel> channels;
188     try {
189       channels = databasePool.doCommand(new WinstonConsumer<List<Channel>>() {
190 
191         public List<Channel> execute(WinstonDatabase winston) throws UtilException {
192           return new Channels(winston).getChannels();
193         }
194 
195       });
196     } catch (Exception e) {
197       throw new UtilException(e.getMessage());
198     }
199     return channels;
200   }
