3   import java.util.Collections;
4   import java.util.HashMap;
5   import java.util.Map;
6   
7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   
10  public class MimeType {
11    private static final Logger LOGGER = LoggerFactory.getLogger(MimeType.class);
12      private static final Map<String, String> mimeTypes;
13      
14      static {
15          Map<String, String> map = new HashMap<String, String>();
16          map.put("png", "image/png");
17          map.put("jpg", "image/jpeg");
18          map.put("jpeg", "image/jpeg");
19          map.put("gif", "image/gif");
20          map.put("html", "text/html");
21          map.put("ico", "image/x-icon");
22          map.put("css", "text/css");
23          map.put("js", "application/javascript");
24          mimeTypes = Collections.unmodifiableMap(map);
25      }
26      
27      public static String guessMimeType(String fileName) {
28        LOGGER.info("guessing mime-type for {}", fileName);
29          int index = fileName.indexOf('.');
30          String extension = null;
31          
32          if (index == -1)
33              extension = fileName;
34          else if (fileName.length() + 2 > index)
35              extension = fileName.substring(index+1);
36  
37          return mimeTypes.get(extension);
38      }
39  }
