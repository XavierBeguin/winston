31    /**
32     * Launch the WWS.
33     *
34     * @param args
35     *          command line arguments
36     * @throws Exception when command line cannot be parsed
37     */
38    public static void main(final String[] args) throws Exception {
39      final WWSArgs config = new WWSArgs(args);
40      final WWS wws = new WWS(config.configFileName);
41  
42      wws.launch();
43  
44      final BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
45      boolean acceptCommands = !(config.isNoInput);
46      if (acceptCommands) {
47        wws.logger.info("Enter ? for console commands.");
48      }
49  
50      while (acceptCommands) {
51        String s = in.readLine();
52        if (s != null) {
53          s = s.toLowerCase().trim();
54          if (s.equals("q")) {
55            acceptCommands = false;
56            System.exit(0);
57          } else if (s.startsWith("c")) {
58            wws.printConnections(s);
59          } else if (s.startsWith("m")) {
60            wws.printCommands(s);
61          } else if (s.equals("d")) {
62            wws.dropConnections(wws.idleTime);
63          } else if (s.startsWith("t")) {
64            wws.toggleTrace(s);
65          } else if (s.equals("0")) {
66            LogManager.getRootLogger().setLevel(Level.ERROR);
67          } else if (s.equals("1")) {
68            LogManager.getRootLogger().setLevel(Level.WARN);
69          } else if (s.equals("2")) {
70            LogManager.getRootLogger().setLevel(Level.INFO);
71          } else if (s.equals("3")) {
72            LogManager.getRootLogger().setLevel(Level.DEBUG);
73          } else if (s.equals("?")) {
74            WWS.printKeys();
75          } else {
76            WWS.printKeys();
77          }
78        }
79      }
80    }

82    public static void printKeys() {
83      final StringBuffer sb = new StringBuffer();
84      sb.append(Version.VERSION_STRING + "\n");
85      sb.append("Keys:\n");
86      sb.append(" 0-3: logging level\n");
87      sb.append("        d: drop idle connections\n");
88      sb.append("        q: quit\n");
89      sb.append("        ?: display keys\n");
90      sb.append("        m: print running commands\n");
91      sb.append(" t<index>: toggle tracing of a connection\n");
92      sb.append("        c: print connections sorted by bytes transmited\n");
93      sb.append("       cA: print connections sorted by address\n");
94      sb.append("       cC: print connections sorted by connect time\n");
95      sb.append("       cL: print connections sorted by last request time\n");
96      sb.append("       cR: print connections sorted by bytes received\n");
97      sb.append("       cT: print connections sorted by bytes transmited\n");
98      sb.append("       cI: print connections sorted by index\n");
99      sb.append("         : append '-' to sort in descending order.\n");
100 
101     System.out.println(sb);
102   }

214   /**
215    * Processes the configuration file (default 'WWS.config'). See the default
216    * file for documentation on the different options.
217    */
218   public void processConfigFile() {
219     final ConfigFile cf = new ConfigFile(configFilename);
220     if (!cf.wasSuccessfullyRead()) {
221       fatalError(configFilename + ": could not read config file.");
222     }
223 
224     final String a = cf.getString("wws.addr");
225     if (a != null) {
226       try {
227         serverIP = InetAddress.getByName(a);
228       } catch (final UnknownHostException e) {
229         logger.info("unknown host " + a);
230       }
231       logger.info("config: wws.addr=" + serverIP.getCanonicalHostName() + ".");
232     }
233 
234     final int p = StringUtils.stringToInt(cf.getString("wws.port"), -1);
235     if (p < 0 || p > 65535) {
236       fatalError(configFilename + ": bad or missing 'wws.port' setting.");
237     }
238     serverPort = p;
239     logger.info("config: wws.port=" + serverPort + ".");
240 
241     final boolean k = StringUtils.stringToBoolean(cf.getString("wws.keepalive"), false);
242     keepalive = k;
243     logger.info("config: wws.keepalive=" + keepalive + ".");
244 
245     final int h = StringUtils.stringToInt(cf.getString("wws.handlers"), -1);
246     if (h < 1 || h > 128) {
247       fatalError(configFilename + ": bad or missing 'wws.handlers' setting.");
248     }
249     handlers = h;
250     logger.info("config: wws.handlers=" + handlers + ".");
251 
252     final int m = StringUtils.stringToInt(cf.getString("wws.maxConnections"), -1);
253     if (m < 0) {
254       fatalError(configFilename + ": bad or missing 'wws.maxConnections' setting.");
255     }
256 
257     connections.setMaxConnections(m);
258     logger.info("config: wws.maxConnections=" + connections.getMaxConnections() + ".");
259 
260     final long i = 1000 * StringUtils.stringToInt(cf.getString("wws.idleTime"), 7200);
261     if (i < 0) {
262       fatalError(configFilename + ": bad or missing 'wws.idleTime' setting.");
263     }
264 
265     idleTime = i;
266     logger.info("config: wws.idleTime=" + idleTime / 1000 + ".");
267 
268     embargo = 0; // TODO: implement
269 
270     maxDays = StringUtils.stringToInt(cf.getString("wws.maxDays"), 0);
271     logger.info("config: wws.maxDays=" + maxDays + ".");
272 
273     if (cf.getString("wws.allowHttp") == null) {
274       fatalError(configFilename + ": missing 'wws.allowHttp' setting.");
275     } else {
276       allowHttp = StringUtils.stringToBoolean(cf.getString("wws.allowHttp"));
277     }
278     logger.info("config: wws.allowHttp=" + allowHttp + ".");
279 
280     httpMaxSize = StringUtils.stringToInt(cf.getString("wws.httpMaxSize"), 10000000);
281     logger.info("config: wws.httpMaxSize=" + httpMaxSize + ".");
282 
283     slowCommandTime = StringUtils.stringToInt(cf.getString("wws.slowCommandTime"), 1500);
284     logger.info("config: wws.slowCommandTime=" + slowCommandTime + ".");
285 
286     httpRefreshInterval = StringUtils.stringToInt(cf.getString("wws.httpRefreshInterval"), 0);
287     logger.info("config: wws.httpRefreshInterval=" + httpRefreshInterval + ".");
288 
289     winstonDriver = cf.getString("winston.driver");
290     winstonURL = cf.getString("winston.url");
291     winstonPrefix = cf.getString("winston.prefix");
292     winstonStatementCacheCap =
293         StringUtils.stringToInt(cf.getString("winston.statementCacheCap"), 100);
294   }
