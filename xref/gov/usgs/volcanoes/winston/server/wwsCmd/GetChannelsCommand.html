31  /**
32   * Return Channel details.
33   * 
34   * request = /^GETCHANNELS:? GC( METADATA)?$/
35   *
36   * @author Dan Cervelli
37   * @author Tom Parker
38   */
39  public class GetChannelsCommand extends WwsBaseCommand {
40    private static final int PROTOCOL_VERSION = 3;
41  
42    private static final Logger LOGGER = LoggerFactory.getLogger(GetChannelsCommand.class);
43  
44    public GetChannelsCommand() {
45      super();
46    }
47  
48    public void doCommand(ChannelHandlerContext ctx, WwsCommandString cmd)
49        throws MalformedCommandException {
50  
51      if (!cmd.isLegal(2) && !cmd.isLegal(3)) {
52        throw new MalformedCommandException();
53      }
54  
55      boolean metadata = false;
56      if ("METADATA".equals(cmd.getString(2))) {
57        metadata = true;
58      }
59  
60      WinstonDatabase winston = null;
61      List<Channel> chs = null;
62      try {
63        winston = databasePool.borrowObject();
64        if (!winston.checkConnect()) {
65          LOGGER.error("WinstonDatabase unable to connect to MySQL.");
66        } else {
67          Channels channels = new Channels(winston);
68          channels.setAparentRetention(maxDays * ONE_DAY_S); 
69          chs = channels.getChannels(metadata);
70          LOGGER.info("got {} channels", chs.size());
71        }
72      } catch (Exception e) {
73        LOGGER.error("Unable to fulfill command.", e);
74      } finally {
75        if (winston != null) {
76          databasePool.returnObject(winston);
77        }
78      }
79  
80      final StringBuilder sb = new StringBuilder(chs.size() * 60);
81      sb.append(String.format("%s %d\n", cmd.getID(), chs.size()));
82      for (final Channel ch : chs) {
83        if (metadata)
84          sb.append(ch.toMetadataString() + "\n");
85        else
86          sb.append(ch.toPV2String() + "\n");
87      }
88      LOGGER.info("maxDays = {}", maxDays);
89      ctx.writeAndFlush(sb.toString());
90    }
91  }
