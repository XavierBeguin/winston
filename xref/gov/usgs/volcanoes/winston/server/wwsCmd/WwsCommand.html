7   package gov.usgs.volcanoes.winston.server.wwsCmd;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  import gov.usgs.volcanoes.winston.server.WinstonDatabasePool;
13  
14  /**
15   * 
16   * @author Tom Parker
17   *
18   */
19  public enum WwsCommand {
20    MENU("MENU", MenuCommand.class);
21  
22    private static final Logger LOGGER = LoggerFactory.getLogger(WwsCommand.class);
23  
24    private String command;
25    private Class<? extends WwsBaseCommand> clazz;
26  
27    private WwsCommand(String command, Class<? extends WwsBaseCommand> clazz) {
28      this.command = command;
29      this.clazz = clazz;
30    }
31  
32    /**
33     * 
34     * @param winstonDatabasePool 
35     * @param command
36     * @return
37     * @throws InstantiationException
38     * @throws IllegalAccessException
39     * @throws UnsupportedCommandException 
40     */
41    public static WwsBaseCommand get(WinstonDatabasePool databasePool, WwsCommandString command)
42        throws InstantiationException, IllegalAccessException, UnsupportedCommandException {
43      for (WwsCommand cmd : WwsCommand.values()) {
44        if (cmd.command.equals(command.getCommand())) {
45          WwsBaseCommand baseCommand = cmd.clazz.newInstance();
46          baseCommand.databasePool(databasePool);
47          return baseCommand;
48        }
49      }
50      throw new UnsupportedCommandException("Unknown WWS command " + command.getCommand());
51    }
52  }
