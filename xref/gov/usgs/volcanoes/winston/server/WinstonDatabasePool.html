7   package gov.usgs.volcanoes.winston.server;
8   
9   import org.apache.commons.pool2.impl.GenericObjectPool;
10  import org.apache.commons.pool2.impl.GenericObjectPoolConfig;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  import gov.usgs.volcanoes.core.configfile.ConfigFile;
15  import gov.usgs.volcanoes.core.util.UtilException;
16  import gov.usgs.volcanoes.winston.db.WinstonDatabase;
17  import gov.usgs.volcanoes.winston.server.wws.WinstonConsumer;
18  
19  public final class WinstonDatabasePool extends GenericObjectPool<WinstonDatabase> {
20  
21    private static final Logger LOGGER = LoggerFactory.getLogger(WinstonDatabasePool.class);
22    public WinstonDatabasePool(ConfigFile configFile, GenericObjectPoolConfig poolConfig) {
23      super(new WinstonDatabaseFactory(configFile), poolConfig);
24    }
25    
26    public <T> T doCommand(WinstonConsumer<T> consumer) throws Exception {
27      WinstonDatabase winston = null;
28      try {
29        winston = borrowObject();
30        if (!winston.checkConnect()) {
31          throw new UtilException("Unable to connect to MySQL.");
32        } else {
33          return consumer.execute(winston);
34        }
35      } finally {
36        if (winston != null) {
37          returnObject(winston);
38        }
39      }
40    }
41  }
