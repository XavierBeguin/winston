24  /**
25   * Statistics of connections to a wave server.
26   * 
27   * @author Tom Parker
28   *
29   */
30  public class ConnectionStatistics {
31  
32    private static final Logger LOGGER = LoggerFactory.getLogger(ConnectionStatistics.class);
33  
34    private static enum SortOrder {
35      ADDRESS, CONNECT_TIME, LAST_REQUEST_TIME, RX_BYTES, TX_BYTES, INDEX;
36  
37      public static SortOrder parse(char c) {
38        switch (Character.toUpperCase(c)) {
39          default:
40          case 'A':
41            return ADDRESS;
42          case 'L':
43            return LAST_REQUEST_TIME;
44          case 'C':
45            return CONNECT_TIME;
46          case 'R':
47            return RX_BYTES;
48          case 'T':
49            return TX_BYTES;
50          case 'I':
51            return INDEX;
52        }
53      }
54    }
55  
56    private final AtomicLong connectionCount;
57    private final AtomicLong wwsCount;
58    private final AtomicLong httpCount;
59    private final AtomicLong openCount;
60    private Map<InetSocketAddress, Connection> connectionMap;
61  
62    private class Connection {
63      private final String address;
64      private final ChannelTrafficShapingHandler trafficShapingHandler;
65      private final AtomicLong connectTime;
66      private AtomicLong lastTime;
67  
68      public Connection(String address, ChannelTrafficShapingHandler trafficShapingHandler) {
69        this.address = address;
70        this.trafficShapingHandler = trafficShapingHandler;
71        connectTime = new AtomicLong(System.currentTimeMillis());
72        lastTime = new AtomicLong(connectTime.get());
73      }
74  
75      public String address() {
76        return address;
77      }
78  
79      public long connectTime() {
80        return connectTime.get();
81      }
82  
83      public long lastTime() {
84        return lastTime.get();
85      }
86  
87      public long cumulativeReadBytes() {
88        return trafficShapingHandler.trafficCounter().cumulativeReadBytes();
89      }
90  
91  
92      public long cumulativeWrittenBytes() {
93        return trafficShapingHandler.trafficCounter().cumulativeWrittenBytes();
94      }
95    }
96  
97    public ConnectionStatistics() {
98      LOGGER.debug("Creating new connection stats");
99      connectionCount = new AtomicLong(0);
100     wwsCount = new AtomicLong(0);
101     httpCount = new AtomicLong(0);
102     openCount = new AtomicLong(0);
103     connectionMap = Collections.synchronizedMap(new HashMap<InetSocketAddress, Connection>());
104 
105   }
106 
107   public long getCount() {
108     return connectionCount.get();
109   }
110 
111   public void incrHttpCount(SocketAddress socketAddress) {
112     httpCount.incrementAndGet();
113     connectionMap.get(socketAddress).lastTime.set(System.currentTimeMillis());
114   }
115 
116   public long getHttpCount() {
117     return httpCount.get();
118   }
119 
120   public void incrWwsCount(SocketAddress socketAddress) {
121     wwsCount.incrementAndGet();
122     connectionMap.get(socketAddress).lastTime.set(System.currentTimeMillis());
123   }
124 
125   public long getWwsCount() {
126     return wwsCount.get();
127   }
128 
129   public void incrOpenCount() {
130     openCount.incrementAndGet();
131     connectionCount.incrementAndGet();
132   }
133 
134   public void decrOpenCount() {
135     openCount.decrementAndGet();
136   }
137 
138   public long getOpen() {
139     return openCount.get();
140   }
141 
142   public void mapChannel(InetSocketAddress remoteAddress,
143       ChannelTrafficShapingHandler trafficCounter2) {
144     LOGGER.debug("mapping " + remoteAddress);
145     connectionMap.put(remoteAddress, new Connection(remoteAddress.toString(), trafficCounter2));
146   }
147 
148   public void unmapChannel(InetSocketAddress remoteAddress) {
149     connectionMap.remove(remoteAddress);
150   }

152   public String printConnections(String s) {
153     StringBuffer sb = new StringBuffer();
154     sb.append("------- Connections --------\n");
155     sb.append(
156         "[A]ddress                 [C]onnect time         [L]ast time            [R]X bytes  [T]X bytes  [I]ndex\n");
157 
158     char col = 'T';
159     if (s.length() > 1)
160       col = s.charAt(1);
161     boolean desc = s.endsWith("-");
162 
163     List<Connection> connections = new ArrayList<Connection>(connectionMap.size());
164     connections.addAll(connectionMap.values());
165     Collections.sort(connections, getComparator(SortOrder.parse(col), desc));
166 
167     for (Connection connection : connections) {
168       sb.append(String.format("%-25s %-22s %-22s %-11d %-11d\n", connection.address(),
169           Time.format(Time.STANDARD_TIME_FORMAT, connection.connectTime()),
170           Time.format(Time.STANDARD_TIME_FORMAT, connection.lastTime()),
171           connection.cumulativeReadBytes(), connection.cumulativeWrittenBytes()));
172 
173     }
174     sb.append(
175         "[A]ddress                 [C]onnect time         [L]ast time            [R]X bytes  [T]X bytes  [I]ndex\n");
176     sb.append("\n\n");
177     sb.append("Total Connections : ").append(connectionCount).append('\n');
178     sb.append("Open Connections  : ").append(openCount).append('\n');
179     sb.append("WWS Commands      : ").append(wwsCount).append('\n');
180     sb.append("HTTP commands     : ").append(httpCount).append('\n');
181     return sb.toString();
182   }

184   public static Comparator<Connection> getComparator(final SortOrder order, final boolean desc) {
185     return new Comparator<Connection>() {
186       public int compare(Connection cs1, Connection cs2) {
187         int cmp = 0;
188         switch (order) {
189           default:
190           case ADDRESS:
191             cmp = cs1.address.compareTo(cs2.address);
192             break;
193           case LAST_REQUEST_TIME:
194             cmp = (int) (cs1.lastTime() - cs2.lastTime());
195             break;
196           case CONNECT_TIME:
197             cmp = (int) (cs1.connectTime() - cs2.connectTime());
198             break;
199           case RX_BYTES:
200             cmp = (int) (cs1.cumulativeReadBytes() - cs2.cumulativeReadBytes());
201             break;
202           case TX_BYTES:
203             cmp = (int) (cs1.cumulativeWrittenBytes() - cs2.cumulativeWrittenBytes());
204             break;
205         }
206         if (desc)
207           cmp = -cmp;
208 
209         return cmp;
210       }
211     };
212   }
