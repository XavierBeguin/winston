66    @Override
67    protected void sendResponse() {
68  
69      // Station
70      code = arguments.get("code");
71      if (code == null) {
72        error = "Error: you must specify a channel (code).";
73      } else {
74        code = code.replace('_', '$');
75        if (code.indexOf(";") != -1) {
76          error = "Error: illegal characters in channel (code).";
77        }
78      }
79  
80      // End Time
81      endTime = getEndTime(arguments.get("t2"));
82      if (endTime.isNaN()) {
83        error = error + "Error: could not parse end time (t2). Should be " + INPUT_DATE_FORMAT + ".";
84      }
85  
86      // Start Time
87      startTime = getStartTime(arguments.get("t1"), endTime, ONE_HOUR);
88      if (startTime.isNaN()) {
89        error = error + "Error: could not parse start time (t1). Should be " + INPUT_DATE_FORMAT
90            + " or -HH.";
91      }
92  
93      // TimeZone
94      timeZone = Util.stringToString(arguments.get("tz"), "UTC");
95      zone = TimeZone.getTimeZone(timeZone);
96  
97      // Minimum Gap Duration
98      minGapDuration = Util.stringToDouble(arguments.get("mgd"), 5);
99  
100     // Write Computer
101     writeComputer = Util.stringToInt(arguments.get("wc"), 0);
102 
103     // If Error
104     if (error.length() > 0) {
105       writeSimpleHTML(error);
106 
107     } else {
108 
109       // Finds Gaps
110       gaps = data.findGaps(code, startTime, endTime);
111       code = code.replace('$', '_');
112       startTimeS = Util.j2KToDateString(startTime, zone);
113       endTimeE = Util.j2KToDateString(endTime, zone);
114       totalTime = endTime - startTime;
115 
116       if (writeComputer == 1) {
117         writeComputer();
118       } else {
119         writeHuman();
120       }
121     }
122   }

253   // Write for Computers
254   void writeComputer() {
255 
256     final StringBuilder header = new StringBuilder();
257     final StringBuilder output = new StringBuilder();
258     output.append("# Gap Start\t\tGap End\t\t\tDuration\n");
259 
260     // Write Computer Data Gaps
261     gapCount = 0;
262     totalGapLength = 0;
263 
264     for (final double[] gap : gaps) {
265       gapLength = gap[1] - gap[0];
266       if (gapLength < minGapDuration) {
267         continue;
268       } else {
269         output.append(gap[0] + "\t" + gap[1] + "\t" + gapLength + "\n");
270         gapCount++;
271         totalGapLength = totalGapLength + gapLength;
272       }
273     }
274 
275     // Write Computer Winston Gaps
276     header.append("# Start Time: " + startTimeS + " (" + startTime + ")\n# End Time: " + endTimeE
277         + " (" + endTime + ")\n# Total Time: " + totalTime + "\n# Time Zone: " + timeZone
278         + "\n# Station: " + code + "\n# Minumum Gap Duration: " + minGapDuration + "\n# GapCount: "
279         + gapCount + "\n");
280 
281     writeSimpleText(header.toString() + output.toString());
282   }
