66      private String parseArguments() {
67          code = arguments.get("code");
68          if (code == null)
69              errorString += "Error: you must specify a channel (code).<br>";
70          else {
71              code = code.replace('_', '$');
72              if (code.indexOf(";") != -1)
73                  errorString += "Error: illegal characters in channel (code).<br>";
74          }
75  
76          timeZone = TimeZone.getTimeZone(Util.stringToString(arguments.get("tz"), "UTC"));
77  
78          endTime = getEndTime(arguments.get("t2"));
79          if (endTime == Double.NaN)
80              errorString += "Error: could not parse end time (t2). Should be " + INPUT_DATE_FORMAT + ".<br>";
81  
82          startTime = getStartTime(arguments.get("t1"), endTime, ONE_DAY);
83          if (startTime == Double.NaN)
84              errorString += "Error: could not parse start time (t1). Should be " + INPUT_DATE_FORMAT + " or -HH.M<br>";
85  
86          timeZoneOffset = timeZone.getOffset((long) Util.j2KToEW(endTime) * 1000) / 1000;
87  
88          width = Util.stringToInt(arguments.get("w"), DEFAULT_W);
89          height = Util.stringToInt(arguments.get("h"), DEFAULT_H);
90  
91          if (height * width <= 0 || height * width > wws.httpMaxSize())
92              errorString += "Error: product of width (w) and height (h) must be between 1 and " + wws.httpMaxSize()
93                      + ".<br>";
94  
95          detrend = Util.stringToBoolean(arguments.get("dt"), DEFAULT_DT);
96  
97          rsamPeriod = Util.stringToInt(arguments.get("rsamP"), DEFAULT_RSAMP);
98          
99          despike = Util.stringToBoolean(arguments.get("ds"), DEFAULT_DS);
100         despikePeriod = Util.stringToDouble(arguments.get("dsp"), 0);
101 
102         runningMedian = Util.stringToBoolean(arguments.get("rm"), false);
103         runningMedianPeriod = Util.stringToDouble(arguments.get("rmp"), 300);
104 
105         plotMax = Util.stringToDouble(arguments.get("max"), Double.MAX_VALUE);
106         plotMin = Util.stringToDouble(arguments.get("min"), Double.MIN_VALUE);
107 
108         outputData = Util.stringToBoolean(arguments.get("csv"), DEFAULT_CSV);
109         
110         return errorString;
111     }

113     private void getData() {
114         rsamData = null;
115         try {
116             
117             DownsamplingType dst;
118             int dsInt;
119             if (rsamPeriod < 2) {
120                 dst = DownsamplingType.NONE;
121                 dsInt = 0;
122             } else {
123                 dst = DownsamplingType.MEAN;
124                dsInt = rsamPeriod;
125             }
126             rsamData = data.getRSAMData(code, startTime, endTime, 0, dst, dsInt);
127             rsamData.adjustTime(timeZoneOffset);
128             
129             if (despike)
130                 rsamData.despike(1, despikePeriod);
131 
132             if (detrend)
133                 rsamData.detrend(1);
134 
135             if (runningMedian)
136                 rsamData.set2median(1, runningMedianPeriod);
137 
138 
139         } catch (UtilException e) {
140             writeSimpleHTML("Error: could not get RSAM data, check channel (code). e = " + e.toString());
141         }
142         if (rsamData == null || rsamData.rows() <= 0)
143             writeSimpleHTML("Error: could not get RSAM data, check channel (code). Empty result.");
144     }

146     private void sendPlot() {
147         HttpResponse response = new HttpResponse("image/png");
148         response.setVersion(request.getVersion());
149         if (wws.httpRefreshInterval() > 0)
150             response.setHeader("Refresh:", wws.httpRefreshInterval() + "; url=" + request.getResource());
151 
152         byte[] png;
153 
154         Plot plot = new Plot();
155         plot.setSize(width, height);
156         plot.setBackgroundColor(new Color(0.97f, 0.97f, 0.97f));
157 
158         MatrixRenderer mr = new MatrixRenderer(rsamData.getData(), false);
159         double max = Math.min(plotMax, rsamData.max(1) + rsamData.max(1) * .1);
160         double min = Math.max(plotMin, rsamData.min(1) - rsamData.max(1) * .1);
161 
162         mr.setExtents(startTime + timeZoneOffset, endTime + timeZoneOffset, min, max);
163         mr.setLocation(70, 35, width - 140, height - 70);
164         mr.createDefaultAxis();
165         mr.setXAxisToTime(8, true, true);
166 
167         String tzText = timeZone.getDisplayName(timeZone.inDaylightTime(Util.j2KToDate(endTime)), TimeZone.SHORT);
168         String bottomText = "(" + Util.j2KToDateString(startTime + timeZoneOffset, DISPLAY_DATE_FORMAT) + " to "
169                 + Util.j2KToDateString(endTime + timeZoneOffset, DISPLAY_DATE_FORMAT) + " " + tzText + ")";
170 
171         mr.getAxis().setBottomLabelAsText(bottomText);
172         mr.getAxis().setLeftLabelAsText("RSAM");
173         DefaultFrameDecorator.addLabel(mr, code.replace('$', ' '), Location.LEFT);
174         mr.createDefaultLineRenderers(Color.blue);
175         // mr.setExtents(startTime, endTime, gdm.min(1), gdm.max(1));
176         plot.addRenderer(mr);
177         try {
178             png = plot.getPNGBytes();
179             response.setLength(png.length);
180             netTools.writeString(response.getHeaderString(), socketChannel);
181             netTools.writeByteBuffer(ByteBuffer.wrap(png), socketChannel);
182         } catch (PlotException e) {
183             e.printStackTrace();
184             errorString = errorString + "Error: Can not create plot.";
185         }
186 
187     }
188 
189     private void sendData() {
190         String html = rsamData.toCSV();
191         HttpResponse response = new HttpResponse("text/csv; charset=utf-8");
192         String fileName = code + "-RSAM.csv";
193         response.setHeader("content-Disposition:", "attachment; filename='" + fileName + "'");
194         response.setLength(html.length());
195         response.setCode("200");
196         netTools.writeString(response.getHeaderString(), socketChannel);
197         netTools.writeString(html, socketChannel);
198     }
199 
200     protected void sendResponse() {
201         errorString = parseArguments();
202 
203         if (errorString.length() > 0) {
204             writeSimpleHTML(errorString);
205         } else {
206             getData();
207             if (outputData)
208                 sendData();
209             else
210                 sendPlot();
211         }
212     }
