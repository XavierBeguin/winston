22  public class GetSCNRawCommand extends BaseCommand {
23    private static final Logger LOGGER = LoggerFactory.getLogger(WWSClient.class);
24  
25    public GetSCNRawCommand(final NetTools nt, final WinstonDatabase db, final WWS wws) {
26      super(nt, db, wws);
27    }
28  
29    public void doCommand(final Object info, final SocketChannel channel) {
30      final String cmd = (String) info;
31  
32      final String[] ss = cmd.split(" ");
33      if (ss.length < 7)
34        return; // malformed command
35  
36      final String id = ss[1];
37      final String s = ss[2];
38      final String c = ss[3];
39      final String n = ss[4];
40      double t1 = Double.NaN;
41      double t2 = Double.NaN;
42      try {
43        t1 = J2kSec.fromEpoch(Ew.asEpoch(Double.parseDouble(ss[5])));
44        t1 = timeOrMaxDays(t1);
45  
46        t2 = J2kSec.fromEpoch(Ew.asEpoch(Double.parseDouble(ss[6])));
47        t2 = timeOrMaxDays(t2);
48      } catch (final Exception e) {
49      }
50  
51      if (id == null || s == null || c == null || n == null || Double.isNaN(t1) || Double.isNaN(t2))
52        return; // malformed command
53  
54      final int sid = emulator.getChannelID(s, c, n);
55      if (sid == -1) {
56        sendNoChannelResponse(id, 0, s, c, n, null, channel);
57        return;
58      }
59  
60      final double[] bounds = checkTimes(sid, t1, t2);
61      if (!allowTransaction(bounds)) {
62        final String error =
63            id + " " + sid + " " + s + " " + c + " " + n + " " + getError(bounds) + "\n";
64        netTools.writeString(error, channel);
65        return;
66      }
67  
68      final Object[] result = emulator.getWaveServerRaw(s, c, n, t1, t2);
69      int totalBytes = 0;
70      if (result != null) {
71        final String hdr = id + " " + (String) result[0] + "\n";
72        final int bytes = ((Integer) result[1]).intValue();
73        final List<?> items = (List<?>) result[2];
74        final ByteBuffer bb = ByteBuffer.allocate(bytes);
75        for (final Iterator<?> it = items.iterator(); it.hasNext();) {
76          bb.put((byte[]) it.next());
77        }
78        bb.flip();
79  
80        netTools.writeString(hdr, channel);
81        totalBytes = netTools.writeByteBuffer(bb, channel);
82      } else {
83        // must be a gap
84        netTools.writeString(id + " " + sid + " " + s + " " + c + " " + n + " FG s4\n", channel);
85      }
86      final String scn = s + "_" + c + "_" + n;
87      final String time = J2kSec.toDateString(t1) + " - " + J2kSec.toDateString(t2);
88      LOGGER.debug("GETSCNRAW {} : {}, {} bytes.", scn, time, totalBytes);
89    }
90  }
