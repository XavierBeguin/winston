21  abstract public class BaseCommand implements Command {
22  	protected final static int ONE_HOUR = 60 * 60;
23  	protected final static int ONE_DAY = 24 * ONE_HOUR;
24  
25  	protected NetTools netTools;
26  	protected WinstonDatabase winston;
27  	protected WWS wws;
28  	protected Data data;
29  	protected WaveServerEmulator emulator;
30  	protected int maxDays;
31  	protected DecimalFormat decimalFormat;
32  
33  	public BaseCommand(NetTools nt, WinstonDatabase db, WWS wws) {
34  		netTools = nt;
35  		winston = db;
36  		this.wws = wws;
37  		maxDays = wws.getMaxDays();
38  		emulator = new WaveServerEmulator(db);
39  		data = new Data(db);
40  		decimalFormat = (DecimalFormat) DecimalFormat.getInstance();
41  		decimalFormat.setMaximumFractionDigits(3);
42  		decimalFormat.setGroupingUsed(false);
43  	}

45  	protected void sendNoChannelResponse(String id, int pin, String s, String c, String n, String l,
46  			SocketChannel channel) {
47  		String loc = "";
48  		if (l != null)
49  			loc = " " + l;
50  		netTools.writeString(id + " " + id + " " + pin + " " + s + " " + c + " " + n + loc + " FN\n", channel);
51  	}
52  
53  	protected int writeByteBuffer(String id, ByteBuffer bb, boolean compress, SocketChannel channel) {
54  		if (bb == null) {
55  			netTools.writeString(id + " 0\n", channel);
56  			return 0;
57  		}
58  		if (compress)
59  			bb = ByteBuffer.wrap(Util.compress(bb.array()));
60  
61  		netTools.writeString(id + " " + bb.limit() + "\n", channel);
62  		return netTools.writeByteBuffer(bb, channel);
63  	}
64  
65  	protected String getError(double[] d) {
66  		if (d == null || d.length != 2)
67  			return "";
68  
69  		if (Double.isNaN(d[0]) && Double.isNaN(d[1]))
70  			return "FB";
71  
72  		if (Double.isNaN(d[0]))
73  			return "FL s4 " + Double.toString(Util.j2KToEW(d[1]));
74  
75  		if (Double.isNaN(d[1]))
76  			return "FR s4 " + Double.toString(Util.j2KToEW(d[0]));
77  
78  		return "OK";
79  	}
80  
81  	protected boolean allowTransaction(double[] d) {
82  		return !(d == null || d.length != 2 || Double.isNaN(d[0]) || Double.isNaN(d[1]));
83  	}
84  
85  	protected double[] checkTimes(int sid, double t1, double t2) {
86  		if (t1 >= t2)
87  			return new double[] { Double.NaN, Double.NaN };
88  
89  		double[] tb = data.getTimeSpan(sid);
90  
91  		if (t1 < tb[0])
92  			t1 = tb[0];
93  
94  		// only apply the later bounds check if there is an embargo otherwise we
95  		// have to deal
96  		// with other people's idea of what now is
97  		if (t2 > tb[1])
98  			t2 = tb[1];
99  
100 		if (t2 < tb[0])
101 			return new double[] { Double.NaN, tb[0] };
102 
103 		if (t1 > tb[1])
104 			return new double[] { tb[1], Double.NaN };
105 
106 		return new double[] { t1, t2 };
107 	}

109 	public int writeWaveAsAscii(Wave wave, int sid, String id, String s, String c, String n, String l, double t1,
110 			double t2, String fill, SocketChannel channel) {
111 		NumberFormat numberFormat = new DecimalFormat("#.######");
112 		String sts = null;
113 
114 		// find first sample time
115 		double ct = wave.getStartTime() - wave.getRegistrationOffset();
116 		double dt = 1 / wave.getSamplingRate();
117 		for (int i = 0; i < wave.numSamples(); i++) {
118 			if (ct >= (t1 - dt / 2))
119 				break;
120 			ct += dt;
121 		}
122 		sts = numberFormat.format(Util.j2KToEW(ct));
123 		ByteBuffer bb = ByteBuffer.allocate(wave.numSamples() * 13 + 256);
124 		bb.put(id.getBytes());
125 		bb.put((byte) ' ');
126 		bb.put(Integer.toString(sid).getBytes());
127 		bb.put((byte) ' ');
128 		bb.put(s.getBytes());
129 		bb.put((byte) ' ');
130 		bb.put(c.getBytes());
131 		bb.put((byte) ' ');
132 		bb.put(n.getBytes());
133 		if (l != null) {
134 			bb.put((byte) ' ');
135 			bb.put(l.getBytes());
136 		}
137 		bb.put(" F s4 ".getBytes());
138 		bb.put(sts.getBytes());
139 		bb.put((byte) ' ');
140 		bb.put(Double.toString(wave.getSamplingRate()).getBytes());
141 		bb.put(" ".getBytes());
142 		int sample;
143 		ct = wave.getStartTime();
144 		// int samples = 0;
145 		for (int i = 0; i < wave.numSamples(); i++) {
146 			if (ct >= (t1 - dt / 2)) {
147 				// samples++;
148 				sample = wave.buffer[i];
149 				if (sample == Wave.NO_DATA)
150 					bb.put(fill.getBytes());
151 				else
152 					bb.put(Integer.toString(wave.buffer[i]).getBytes());
153 				bb.put((byte) ' ');
154 			}
155 			ct += dt;
156 			if (ct >= t2)
157 				break;
158 		}
159 		bb.put((byte) '\n');
160 		bb.flip();
161 		return netTools.writeByteBuffer(bb, channel);
162 	}

164 	/**
165 	 * Apply maxDays to time
166 	 * 
167 	 * @param t
168 	 *            time
169 	 * @return greater of t or now less maxDays
170 	 */
171 	double timeOrMaxDays(double t) {
172 		if (maxDays == 0)
173 			return t;
174 		else
175 			return Math.max(t, Util.nowJ2K() - (maxDays * ONE_DAY));
176 	}
