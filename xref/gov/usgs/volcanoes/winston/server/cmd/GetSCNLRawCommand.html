15  /**
16   * Winston implementation of earthworm wave_serverV GETSCNLRAW command
17   *
18   * @author Dan Cervelli
19   */
20  public class GetSCNLRawCommand extends BaseCommand {
21    public GetSCNLRawCommand(final NetTools nt, final WinstonDatabase db, final WWS wws) {
22      super(nt, db, wws);
23    }
24  
25    public void doCommand(final Object info, final SocketChannel channel) {
26  
27      final CodeTimer ct = new CodeTimer("GetSCNLRaw");
28      final String cmd = (String) info;
29  
30      final String[] ss = cmd.split(" ");
31      if (ss.length < 8)
32        return; // malformed command
33  
34      final String id = ss[1];
35      final String s = ss[2];
36      final String c = ss[3];
37      final String n = ss[4];
38      final String l = ss[5];
39      double t1 = Double.NaN;
40      double t2 = Double.NaN;
41      try {
42        t1 = Util.ewToJ2K(Double.parseDouble(ss[6]));
43        t1 = timeOrMaxDays(t1);
44  
45        t2 = Util.ewToJ2K(Double.parseDouble(ss[7]));
46        t2 = timeOrMaxDays(t2);
47      } catch (final Exception e) {
48      }
49  
50      if (id == null || s == null || c == null || n == null || Double.isNaN(t1) || Double.isNaN(t2))
51        return; // malformed command
52  
53      final int sid = emulator.getChannelID(s, c, n, l);
54      if (sid == -1) {
55        sendNoChannelResponse(id, 0, s, c, n, l, channel);
56        return;
57      }
58  
59      final double[] bounds = checkTimes(sid, t1, t2);
60      if (!allowTransaction(bounds)) {
61        final String error =
62            id + " " + sid + " " + s + " " + c + " " + n + " " + l + " " + getError(bounds) + "\n";
63        netTools.writeString(error, channel);
64        return;
65      }
66  
67      final Object[] result = emulator.getWaveServerRaw(s, c, n, l, t1, t2);
68  
69      ct.stop();
70      if (wws.getSlowCommandTime() > 0 && ct.getRunTimeMillis() > wws.getSlowCommandTime() * .75)
71        wws.log(Level.INFO,
72            String.format(
73                "slow db query (%1.2f ms) GETSCNLRAW " + s + "$" + c + "$" + n + "$" + l + " " + t1
74                    + " -> " + t2 + " (" + decimalFormat.format(t2 - t1) + ") ",
75                ct.getRunTimeMillis()),
76            channel);
77  
78      int totalBytes = 0;
79      if (result != null) {
80        final String hdr = id + " " + (String) result[0] + "\n";
81        final int bytes = ((Integer) result[1]).intValue();
82        final List<?> items = (List<?>) result[2];
83        final ByteBuffer bb = ByteBuffer.allocate(bytes);
84        for (final Iterator<?> it = items.iterator(); it.hasNext();) {
85          bb.put((byte[]) it.next());
86        }
87        bb.flip();
88  
89        ct.start();
90        netTools.writeString(hdr, channel);
91        totalBytes = netTools.writeByteBuffer(bb, channel);
92        ct.stop();
93        if (wws.getSlowCommandTime() > 0 && ct.getRunTimeMillis() > wws.getSlowCommandTime() * .75)
94          wws.log(Level.INFO,
95              String.format(
96                  "slow network (%1.2f ms) GETSCNLRAW " + s + "$" + c + "$" + n + "$" + l + " " + t1
97                      + " -> " + t2 + " (" + decimalFormat.format(t2 - t1) + ") ",
98                  ct.getRunTimeMillis()),
99              channel);
100     } else {
101       // must be a gap
102       netTools.writeString(id + " " + sid + " " + s + " " + c + " " + n + " " + l + " FG s4\n",
103           channel);
104     }
105 
106     final String scnl = s + "_" + c + "_" + n + "_" + l;
107     final String time = Util.j2KToDateString(t1) + " - " + Util.j2KToDateString(t2);
108     wws.log(Level.FINER, "GETSCNLRAW " + scnl + " : " + time + ", " + totalBytes + " bytes.",
109         channel);
110   }
111 }
