18  public class GetSCNLCommand extends BaseCommand
19  {
20  	public GetSCNLCommand(NetTools nt, WinstonDatabase db, WWS wws)
21  	{
22  		super(nt, db, wws);
23  	}
24  	
25  	public void doCommand(Object info, SocketChannel channel)
26  	{
27  		int totalBytes = 0;
28  		WWSCommandString cmd = new WWSCommandString((String)info);
29  		if (!cmd.isLegalSCNLTT(9))
30  			return;
31  		
32  		int sid = emulator.getChannelID(cmd.getS(), cmd.getC(), cmd.getN(), cmd.getL());
33  		if (sid == -1)
34  		{
35  			sendNoChannelResponse(cmd.getID(), 0, cmd.getS(), cmd.getC(), cmd.getN(), cmd.getL(), channel);
36  			return;
37  		}
38  		
39  		double t1 = Util.ewToJ2K(cmd.getT1(true));
40  		t1 = timeOrMaxDays(t1);
41  		
42  		double t2 = Util.ewToJ2K(cmd.getT2(true));
43  		t2 = timeOrMaxDays(t2);
44  		
45  		double[] bounds = checkTimes(sid, t1, t2);
46  		if (!allowTransaction(bounds))
47  		{
48  			netTools.writeString(cmd.getEarthwormErrorString(sid, getError(bounds)), channel);
49  			return;
50  		}
51  		Wave wave = null;
52  		try{
53  			wave = data.getWave(sid, t1, t2, 0);
54  		} catch (UtilException e){
55  		}
56  		if (wave != null)
57  			totalBytes = writeWaveAsAscii(wave, sid, cmd.getID(), cmd.getS(), cmd.getC(), cmd.getN(), cmd.getL(), t1, t2, cmd.getString(8), channel);
58  		else
59  			netTools.writeString(cmd.getEarthwormErrorString(sid, "FG s4"), channel);
60  
61  		String scn = cmd.getS() + "_" + cmd.getC() + "_" + cmd.getN() + "_" + cmd.getL();
62  		String time = Util.j2KToDateString(t1) + " - " + Util.j2KToDateString(t2);
63  		wws.log(Level.FINER, "GETSCNL " + scn + " : " + time + ", " + totalBytes + " bytes.", channel);
64  	}
65  }
