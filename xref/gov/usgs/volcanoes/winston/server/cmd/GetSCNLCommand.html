21  public class GetSCNLCommand extends BaseCommand {
22    private static final Logger LOGGER = LoggerFactory.getLogger(GetSCNLCommand.class);
23  
24    public GetSCNLCommand(final NetTools nt, final WinstonDatabase db, final WWS wws) {
25      super(nt, db, wws);
26    }
27  
28    public void doCommand(final Object info, final SocketChannel channel) {
29      int totalBytes = 0;
30      final WWSCommandString cmd = new WWSCommandString((String) info);
31      if (!cmd.isLegalSCNLTT(9))
32        return;
33  
34      final int sid = emulator.getChannelID(cmd.getS(), cmd.getC(), cmd.getN(), cmd.getL());
35      if (sid == -1) {
36        sendNoChannelResponse(cmd.getID(), 0, cmd.getS(), cmd.getC(), cmd.getN(), cmd.getL(),
37            channel);
38        return;
39      }
40  
41      double t1 = J2kSec.fromEpoch(Ew.asEpoch(cmd.getT1(true)));
42      t1 = timeOrMaxDays(t1);
43  
44      double t2 = J2kSec.fromEpoch(Ew.asEpoch(cmd.getT2(true)));
45      t2 = timeOrMaxDays(t2);
46  
47      final double[] bounds = checkTimes(sid, t1, t2);
48      if (!allowTransaction(bounds)) {
49        netTools.writeString(cmd.getEarthwormErrorString(sid, getError(bounds)), channel);
50        return;
51      }
52      Wave wave = null;
53      try {
54        wave = data.getWave(sid, t1, t2, 0);
55      } catch (final UtilException e) {
56      }
57      if (wave != null)
58        totalBytes = writeWaveAsAscii(wave, sid, cmd.getID(), cmd.getS(), cmd.getC(), cmd.getN(),
59            cmd.getL(), t1, t2, cmd.getString(8), channel);
60      else
61        netTools.writeString(cmd.getEarthwormErrorString(sid, "FG s4"), channel);
62  
63      final String scn = cmd.getS() + "_" + cmd.getC() + "_" + cmd.getN() + "_" + cmd.getL();
64      final String time = J2kSec.toDateString(t1) + " - " + J2kSec.toDateString(t2);
65      LOGGER.debug("GETSCNL {} : {}, {} bytes.", scn, time, totalBytes);
66    }
67  }
