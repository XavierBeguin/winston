17  /**
18   * 
19   * Example command string:
20   * GETWAVERAW: GS AUI EHZ AV -- 284802563.447000 284802683.447000 1
21   * 
22   * Compress wave before returning
23   * Not supported by wave_serverV
24   * 
25   * @author Dan Cervelli
26   */
27  public class GetWaveRawCommand extends BaseCommand {
28  	public GetWaveRawCommand(NetTools nt, WinstonDatabase db, WWS wws) {
29  		super(nt, db, wws);
30  	}
31  
32  	public void doCommand(Object info, SocketChannel channel) {
33  		CodeTimer ct = new CodeTimer("GetWaveRaw");
34  
35  		WWSCommandString cmd = new WWSCommandString((String) info);
36  		if (!cmd.isLegalSCNLTT(9))
37  			return; // malformed command
38  
39  		double et = cmd.getT2(true);
40  		et = timeOrMaxDays(et);
41  
42  		double st = cmd.getT1(true);
43  		st = timeOrMaxDays(st);
44  
45  		et = Math.min(et, CurrentTime.getInstance().nowJ2K() - wws.getEmbargo());
46  		Wave wave = null;
47  		if (st < et) {
48  			try {
49  				wave = data.getWave(cmd.getWinstonSCNL(), st, et, 0);
50  			} catch (UtilException e) {
51  			}
52  		}
53  		ct.stop();
54  
55  		// Did it take too long to gather the data?
56  		if (wws.getSlowCommandTime() > 0 && ct.getRunTimeMillis() > wws.getSlowCommandTime() * .75)
57  			wws.log(Level.INFO,
58  					String.format("slow db query (%1.2f ms) GETWAVERAW " + cmd.getWinstonSCNL() + " " + st + " -> " + et
59  							+ " (" + decimalFormat.format(et - st) + ") ", ct.getRunTimeMillis()), channel);
60  
61  		ByteBuffer bb = null;
62  		if (wave != null && wave.numSamples() > 0)
63  			bb = (ByteBuffer) wave.toBinary().flip();
64  
65  		boolean compress = cmd.getInt(8) == 1;
66  
67  		ct.start();
68  		int bytes = writeByteBuffer(cmd.getID(), bb, compress, channel);
69  		ct.stop();
70  
71  		// Did it take too long to deliver the data?
72  		if (wws.getSlowCommandTime() > 0 && ct.getRunTimeMillis() > wws.getSlowCommandTime() * .75)
73  			wws.log(Level.INFO,
74  					String.format("slow network (%1.2f ms) GETWAVERAW " + cmd.getWinstonSCNL() + " " + st + " -> " + et
75  							+ " (" + decimalFormat.format(et - st) + ") ", ct.getRunTimeMillis()), channel);
76  
77  		String time = Util.j2KToDateString(st) + " - " + Util.j2KToDateString(et);
78  		wws.log(Level.FINER, "GETWAVERAW " + cmd.getWinstonSCNL() + ": " + time + ", " + bytes + " bytes. ("
79  				+ (String) info, channel);
80  	}
81  }
