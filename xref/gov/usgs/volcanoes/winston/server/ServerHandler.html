22  /**
23   * A class that handles WWS requests.
24   *
25   * TODO: send ERROR REQUEST on bad commands.
26   * TODO: compressed wave protocol.
27   *
28   * FYI, known bugs in wave_serverV that this program mimics:<br>
29   * -- SCN not found on MENUSCN replies with extra request ID<br>
30   * -- MENUSCN does not append \n properly<br>
31   *
32   *
33   * @author Dan Cervelli
34   */
35  public class ServerHandler extends CommandHandler {
36    private static final int PROTOCOL_VERSION = 3;
37  
38    private final WinstonDatabase winston;
39  
40    private static int instances = 0;
41    private final WWS wws;
42    private final NetTools netTools;
43  
44    public ServerHandler(final WWS s) {
45      super(s, "WWSHandler-" + instances++);
46  
47      wws = s;
48      netTools = new NetTools();
49      netTools.setServer(wws);
50      winston = new WinstonDatabase(wws.getWinstonDriver(), wws.getWinstonURL(),
51          wws.getWinstonPrefix(), wws.getWinstonStatementCacheCap());
52  
53      slowCommandTime = wws.slowCommandTime;
54      setupCommandHandlers();
55    }
56  
57    @Override
58    protected void setupCommandHandlers() {
59      addCommand("VERSION", new BaseCommand(netTools, winston, wws) {
60        public void doCommand(final Object info, final SocketChannel channel) {
61          netTools.writeString("PROTOCOL_VERSION: " + PROTOCOL_VERSION + "\n", channel);
62        }
63      });
64      addCommand("MENU", new MenuCommand(netTools, winston, wws));
65      addCommand("STATUS", new StatusCommand(netTools, winston, wws));
66      addCommand("GETSCNRAW", new GetSCNRawCommand(netTools, winston, wws));
67      addCommand("GETSCNLRAW", new GetSCNLRawCommand(netTools, winston, wws));
68      addCommand("GETSCN", new GetSCNCommand(netTools, winston, wws));
69      addCommand("GETSCNL", new GetSCNLCommand(netTools, winston, wws));
70      addCommand("GETSCNLHELIRAW", new GetSCNLHeliRawCommand(netTools, winston, wws));
71      addCommand("GETSCNLRSAMRAW", new GetSCNLRSAMRawCommand(netTools, winston, wws));
72      addCommand("GETCHANNELS", new GetChannelsCommand(netTools, winston, wws));
73      addCommand("GETWAVERAW", new GetWaveRawCommand(netTools, winston, wws));
74      addCommand("GETMETADATA", new GetMetadataCommand(netTools, winston, wws));
75      if (wws.isHttpAllowed()) {
76        addCommand("GET", new HttpCommand(netTools, winston, wws, this));
77        addCommand("POST", new HttpCommand(netTools, winston, wws, this));
78      }
79    }
80  }
