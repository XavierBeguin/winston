24  public class ImportSeed extends StaticImporter {
25  
26    public static final String JSAP_PROGRAM_NAME = "java gov.usgs.winston.in.ImportSeed";
27  
28    public static final String JSAP_EXPLANATION = "Import miniSEED\n" + "\n"
29        + "This program imports data from miniSEED volumes into a winston database\n";
30  
31    private static final Parameter[] JSAP_PARAMETERS = new Parameter[] {
32        new FlaggedOption("station", JSAP.STRING_PARSER, JSAP.NO_DEFAULT, JSAP.NOT_REQUIRED, 's',
33            "station", "Override station identifier code\n"),
34        new FlaggedOption("channel", JSAP.STRING_PARSER, JSAP.NO_DEFAULT, JSAP.NOT_REQUIRED, 'c',
35            "channel", "Override channel identifier\n"),
36        new FlaggedOption("network", JSAP.STRING_PARSER, JSAP.NO_DEFAULT, JSAP.NOT_REQUIRED, 'n',
37            "network", "Override network identifier\n"),
38        new FlaggedOption("location", JSAP.STRING_PARSER, JSAP.NO_DEFAULT, JSAP.NOT_REQUIRED, 'l',
39            "location", "Override location identifier\n"),
40        new FlaggedOption("rsamDelta", JSAP.INTEGER_PARSER, "10", JSAP.NOT_REQUIRED, 'r', "rsamDelta",
41            "Override location identifier\n"),
42        new FlaggedOption("rsamDuration", JSAP.INTEGER_PARSER, "60", JSAP.NOT_REQUIRED, 'd',
43            "rsamDuration", "Override location identifier\n"),
44        new UnflaggedOption("file", JSAP.STRING_PARSER, JSAP.NO_DEFAULT, JSAP.NOT_REQUIRED,
45            JSAP.GREEDY, "files to import.")};
46  
47    private String network;
48    private String station;
49    private String channel;
50    private String location;
51  
52    public ImportSeed() {
53      super();
54    }
55  
56    public ImportSeed(final JSAPResult config) {
57      this();
58      network = config.getString("network");
59      station = config.getString("station");
60      channel = config.getString("channel");
61      location = config.getString("location");
62      rsamDelta = config.getInt("rsamDelta");
63      rsamDuration = config.getInt("rsamDuration");
64    }
65  
66    @Override
67    public Map<String, List<Wave>> readFile(final String fn) throws IOException {
68      final Map<String, List<Wave>> map = new HashMap<String, List<Wave>>();
69  
70      final SeismicDataFile file = SeismicDataFile.getFile(fn, FileType.SEED);
71      file.setNetwork(network);
72      file.setStation(station);
73      file.setChannel(channel);
74      file.setLocation(location);
75  
76      file.read();
77      for (final String ch : file.getChannels()) {
78        final List<Wave> list = new ArrayList<Wave>();
79        list.add(file.getWave(ch));
80        map.put(ch, list);
81      }
82  
83      return map;
84    }
85  
86    public static JSAPResult getArguments(final String[] args) {
87      JSAPResult config = null;
88      try {
89        final SimpleJSAP jsap = new SimpleJSAP(JSAP_PROGRAM_NAME, JSAP_EXPLANATION, JSAP_PARAMETERS);
90  
91        config = jsap.parse(args);
92  
93        if (jsap.messagePrinted() || config.getStringArray("file").length == 0) {
94          // The following error message is useful for catching the case
95          // when args are missing, but help isn't printed.
96          if (!config.getBoolean("help"))
97            System.err.println("Try using the --help flag.");
98  
99          System.exit(1);
100       }
101     } catch (final Exception ex) {
102       ex.printStackTrace();
103       System.exit(1);
104     }
105     return config;
106   }
107 
108   public static void main(final String[] args) {
109     final JSAPResult config = getArguments(args);
110     System.out.printf("RSAM parameters: delta=%d, duration=%d.\n", config.getInt("rsamDelta"),
111         config.getInt("rsamDuration"));
112 
113     final ImportSeed is = new ImportSeed(config);
114 
115     final List<String> files = new ArrayList<String>();
116     for (final String file : config.getStringArray("file"))
117       files.add(file);
118 
119     process(files, is);
120   }
121 }
