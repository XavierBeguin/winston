20  /**
21   * If a span is added to the job it is an explicit guarantee that there is
22   * no existing data in that span. Therefore incoming TraceBufs only need to be
23   * between the given span to avoid overlappers.
24   *
25   * @author Dan Cervelli
26   */

54    public ImportWSJob(final WinstonDatabase w, final WaveServer ws, final ImportWS is) {
55      importWS = is;

58      spans = new ArrayList<double[]>(10);

95    private void getData(final double[] span) {
96      try {
97        final String[] ss = channel.split("\\$");
98        String loc = null;
99        if (ss.length == 4)
100         loc = ss[3];
101 
102       if (requestSCNL && loc == null)
103         loc = "--";
104 
105       final double t1 = span[0];
106       final double t2 = span[1];
107 
108       LOGGER.info("{}: downloading gap: [{} -> {}, {}]", channel, J2kSec.toDateString(span[0]),
109           J2kSec.toDateString(span[1]), Util.timeDifferenceToString(span[1] - span[0]));
110 
111       input.setRowParameters((int) chunkSize + 65, 60);
112 
113       double ct = t1 - chunkSize;
114       List<TraceBuf> tbs = null;
115       final CodeTimer timer = new CodeTimer("chunk");
116       int total = 0;
117       double totalInsTime = 0;
118       double totalDlTime = 0;
119       while (ct < t2) {
120         if (quit)
121           break;
122         ct += chunkSize;
123         final double ret = Math.min(ct + chunkSize + 5, t2 + 5);
124         final CodeTimer netTimer = new CodeTimer("net");
125         tbs = waveServer.getTraceBufs(ss[0], ss[1], ss[2], loc, J2kSec.asEpoch(ct - 5),
126             J2kSec.asEpoch(ret));
127         netTimer.stop();
128         totalDlTime += netTimer.getTotalTimeMillis();
129         if (tbs != null && tbs.size() > 0) {
130           final Iterator<TraceBuf> it = tbs.iterator();
131           double minTime = 1E300;
132           double maxTime = -1E300;
133           while (it.hasNext()) {
134             final TraceBuf tb = it.next();
135             minTime = Math.min(tb.getStartTimeJ2K(), minTime);
136             maxTime = Math.max(tb.getEndTimeJ2K(), maxTime);
137             if (tb.getEndTimeJ2K() < t1 || tb.getStartTimeJ2K() > t2) {
138               // these are totally outside range so can be dropped quietly.
139               it.remove();
140               continue;
141             }
142             if (tb.getStartTimeJ2K() - t1 < -0.0001 || tb.getEndTimeJ2K() - t2 > 0.0001) {
143               it.remove();
144               LOGGER.debug("Overlapping TraceBuf skipped. {} - {}", tb.getStartTimeJ2K() - t1,
145                   tb.getEndTimeJ2K() - t2);
146               continue;
147             }
148             tb.createBytes();
149           }
150           if (tbs.size() == 0)
151             continue;
152           try {
153             if (chunkDelay > 0) {
154               LOGGER.debug("{}: delaying for {}ms...", channel, chunkDelay);
155               Thread.sleep(chunkDelay);
156             }
157           } catch (final Exception e) {
158             e.printStackTrace();
159           }
160           final CodeTimer inputTimer = new CodeTimer("input");
161           final List<InputEW.InputResult> results =
162               input.inputTraceBufs(tbs, rsamEnable, rsamDelta, rsamDuration);
163           inputTimer.stop();
164           totalInsTime += inputTimer.getTotalTimeMillis();
165           LOGGER.debug("{}: {} tb ({}/{}ms), [{} -> {}, {}]", channel, tbs.size(),
166               netTimer.getRunTimeMillis(), inputTimer.getRunTimeMillis(),
167               J2kSec.toDateString(minTime), J2kSec.toDateString(maxTime),
168               Util.timeDifferenceToString(maxTime - minTime));
169 
170           // TODO: clean this up, unify with ImportEW
171           if (results.size() == 1) {
172             // TODO: handle errors
173             final InputEW.InputResult result = results.get(0);
174             LOGGER.warn("Error: {}", result.code);
175           } else {
176             for (int i = 0; i < results.size() - 2; i++) {
177               final InputEW.InputResult result = results.get(i);
178               final TraceBuf tb = result.traceBuf;
179               switch (result.code) {
180                 case SUCCESS_CREATED_TABLE:
181                   LOGGER.info("{}: day table created ({})", channel,
182                       J2kSec.format("yyyy-MM-dd", tb.getEndTimeJ2K() + 1));
183                 case SUCCESS:
184                   total++;
185                   LOGGER.debug("Insert: {}", tb.toString());
186                   break;
187                 case ERROR_DATABASE:
188                   // fixing
189                   LOGGER.debug("Database error: {}", tb.toString());
190                   break;
191                 case ERROR_UNKNOWN:
192                   LOGGER.warn("Unknown insert error: {}", tb.toString());
193                   break;
194                 case ERROR_CHANNEL:
195                 case ERROR_NULL_TRACEBUF:
196                   // these errors should never occur
197                   LOGGER.warn("Bad channel/null TraceBuf.");
198                   break;
199                 case ERROR_DUPLICATE:
200                   LOGGER.info("Duplicate TraceBuf: {}", tb.toString());
201                   break;
202                 case NO_CODE:
203                   // this should never occur
204                   LOGGER.warn("No error/success code: {}", tb.toString());
205                   break;
206                 case ERROR_HELICORDER:
207                   break;
208                 case ERROR_INPUT:
209                   break;
210                 case ERROR_NO_WINSTON:
211                   break;
212                 case ERROR_TIME_SPAN:
213                   break;
214                 case SUCCESS_HELICORDER:
215                   break;
216                 case SUCCESS_TIME_SPAN:
217                   break;
218                 default:
219                   break;
220               }
221             }
222           }
223         }
224       }
225       timer.stop();
226       importWS.addStats(total, totalDlTime, totalInsTime);
227       LOGGER.info("{}: gap {}, {} tbs inserted in {}ms ({}ms/tb)", channel,
228           (quit ? "interrupted" : "finished"), total, timer.getTotalTimeMillis(),
229           (total == 0 ? 0 : timer.getTotalTimeMillis() / total));
230     } catch (final Throwable t) {
231       t.printStackTrace();
232     }
233   }

235   private void getAllData() {
236     for (final double[] span : spans) {
237       if (!quit)
238         getData(span);
239     }
240   }

242   public double getSpansDuration() {
243     double duration = 0.0;
244     for (final double[] span : spans)
245       duration += span[1] - span[0];
246     return duration;
247   }

249   public void go() {
250     for (final double[] span : spans) {
251       LOGGER.info(String.format("%s: gap: [%s -> %s, %s]", channel, J2kSec.toDateString(span[0]),
252           J2kSec.toDateString(span[1]), Util.timeDifferenceToString(span[1] - span[0])));
253     }
254 
255     LOGGER.info(
256         "{}: starting job, total gaps: {} for a total duration of {}, Chunk: {}s, Delay: {}ms",
257         channel, spans.size(), Util.timeDifferenceToString(getSpansDuration()), chunkSize,
258         chunkDelay);
259 
260     if (!menu.channelExists(channel)) {
261       LOGGER.error("Channel does not exist on source WaveServer.");
262       return;
263     }
264 
265     if (!channels.channelExists(channel)) {
266       LOGGER.info("Creating new channel '{}' in Winston database.", channel);
267       channels.createChannel(channel);
268     }
269 
270     waveServer.connect();
271     getAllData();
272     waveServer.close();
273     spans.clear();
274   }
