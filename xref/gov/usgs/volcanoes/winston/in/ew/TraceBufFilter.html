4   import gov.usgs.util.ConfigFile;
5   import gov.usgs.util.Util;

7   import java.util.HashMap;
8   import java.util.Map;
9   import java.util.logging.Level;
10  
11  /**
12   * 
13   * $Log: not supported by cvs2svn $
14   * @author Dan Cervelli
15   */
16  abstract public class TraceBufFilter implements Comparable<TraceBufFilter>
17  {
18  	protected boolean keepRejects = false;
19  	protected Level logLevel = Level.FINEST;
20  	protected boolean accept = true;
21  	protected int order = 1;
22  	protected boolean terminal = true;
23  	private Map<String, String> metadata;
24  	
25  	// return true if tb matches, otherwise false
26  	public abstract boolean match(TraceBuf tb, Options options);
27  	
28  	public TraceBufFilter() 
29  	{}
30  	
31  	public void configure(ConfigFile cf)
32  	{
33  		if (cf == null)
34  			return;
35  		
36  		order = Util.stringToInt(cf.getString("order"), -1);
37  		
38  		String action = cf.getString("action");
39  		if (action == null)
40  		{
41  			accept = true;
42  			return;
43  		}
44  		
45  		if (action.toLowerCase().equals("reject"))
46  			accept = false;
47  		
48  		int log = Util.stringToInt(cf.getString("log"), 0);
49  		switch(log)
50  		{
51  			case 0:
52  				logLevel = Level.FINEST;
53  				break;
54  			case 1:
55  				logLevel = Level.FINE;
56  				break;
57  			case 2:
58  				logLevel = Level.WARNING;
59  				break;
60  		}
61  	}
62  	
63  	public void setKeepRejects(boolean b)
64  	{
65  		keepRejects = b;
66  	}
67  	
68  	public boolean keepRejects()
69  	{
70  		return keepRejects;
71  	}
72  	
73  	public Level getLogLevel()
74  	{
75  		return logLevel;
76  	}
77  	
78  	public boolean isAccept()
79  	{
80  		return accept;
81  	}
82  	
83  	public boolean isTerminal()
84  	{
85  		return terminal;
86  	}
87  	
88  	public Map<String, String> getMetadata() 
89  	{
90  		return metadata;
91  	}
92  	
93  	public void addMetadata(String n, String v)
94  	{
95  		if (metadata==null)
96  			metadata = new HashMap<String, String>();
97  		
98  		metadata.put(n, v);
99  	}
100 	
101 	public int compareTo(TraceBufFilter other)
102 	{
103 		return order - other.order;
104 	}
105 }
