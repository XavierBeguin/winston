12  /**
13   *
14   * $Log: not supported by cvs2svn $

16   * @author Dan Cervelli
17   */
18  public class ChannelStatus {
19    public enum SortOrder {
20      CHANNEL, LAST_INSERT_TIME, MIN_TIME, MAX_TIME, FAILURES, SUCCESSES;
21  
22      public static SortOrder parse(final char c) {
23        switch (Character.toUpperCase(c)) {
24          default:
25          case 'C':
26            return CHANNEL;
27          case 'L':
28            return LAST_INSERT_TIME;
29          case 'M':
30            return MIN_TIME;
31          case 'X':
32            return MAX_TIME;
33          case 'S':
34            return SUCCESSES;
35          case 'F':
36            return FAILURES;
37        }
38      }
39    }
40  
41    private final String channel;
42    private int successes = 0;
43    private int failures = 0;
44    private double lastTime;
45    private double lastBufTime = 0;
46    private double minBufTime = 1E300;
47    private double maxBufTime = -1E300;
48    private final SimpleDateFormat dateFormat;
49  
50    public ChannelStatus(final String c) {
51      dateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS");
52      dateFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
53      channel = c;
54    }
55  
56    public void process(final TraceBuf tb, final InputEW.InputResult.Code code) {
57      if (code == InputEW.InputResult.Code.SUCCESS
58          || code == InputEW.InputResult.Code.SUCCESS_CREATED_TABLE) {
59        successes++;
60        lastTime = J2kSec.now();
61      } else
62        failures++;
63  
64      if (tb != null) {
65        lastBufTime = tb.getStartTimeJ2K();
66        minBufTime = Math.min(lastBufTime, minBufTime);
67        maxBufTime = Math.max(lastBufTime, maxBufTime);
68      }
69    }
70  
71    public double timeSinceLast() {
72      final double now = J2kSec.now();
73      return now - lastTime;
74    }
75  
76    public static String getHeaderString() {
77      return "[C]hannel     [L]ast ins   [M]in time               Ma[x] time               [S]uccess/[F]ailure";
78    }
79  
80    @Override
81    public String toString() {
82      return String.format("%-13s %-12s %-24s %-24s %-12s", channel,
83          Time.secondsToString(timeSinceLast()), dateFormat.format(J2kSec.asDate(minBufTime)),
84          dateFormat.format(J2kSec.asDate(maxBufTime)), successes + "/" + failures);
85    }
86  
87    public static Comparator<ChannelStatus> getComparator(final SortOrder order, final boolean desc) {
88      return new Comparator<ChannelStatus>() {
89        public int compare(final ChannelStatus cs1, final ChannelStatus cs2) {
90          int cmp = 0;
91          switch (order) {
92            default:
93            case CHANNEL:
94              cmp = cs1.channel.compareTo(cs2.channel);
95              break;
96            case LAST_INSERT_TIME:
97              cmp = Double.compare(cs2.lastTime, cs1.lastTime);
98              break;
99            case SUCCESSES:
100             cmp = cs1.successes - cs2.successes;
101             break;
102           case FAILURES:
103             cmp = cs1.failures - cs2.failures;
104             break;
105           case MAX_TIME:
106             cmp = Double.compare(cs1.maxBufTime, cs2.maxBufTime);
107             break;
108           case MIN_TIME:
109             cmp = Double.compare(cs1.minBufTime, cs2.minBufTime);
110             break;
111         }
112         if (desc)
113           cmp = -cmp;
114 
115         return cmp;
116       }
117     };
118   }
