373   protected void processOptions() {
374     optionFilters = new ArrayList<OptionsFilter>();
375     final List<String> optionSets = config.getList("options");
376     for (final String options : optionSets) {
377       if (options.equals("Default"))
378         continue;
379       final ConfigFile tc = config.getSubConfig(options);
380       try {
381         final OptionsFilter filter = new OptionsFilter(options, tc, defaultOptions);
382         optionFilters.add(filter);
383       } catch (final Exception ex) {
384         LOGGER.error("Could not create options: {}", ex);
385       }
386     }
387     for (final OptionsFilter filter : optionFilters)
388       LOGGER.info("config, options, {}: {}", filter.getName(), filter);
389   }

391   protected void processFilters() {
392     traceBufFilters = new ArrayList<TraceBufFilter>();
393     final List<String> filters = config.getList("filter");
394     for (final String filterName : filters) {
395       final ConfigFile fc = config.getSubConfig(filterName);
396       try {
397         String filterClass = fc.getString("class");
398         if (filterClass.startsWith("gov.usgs.winston")) {
399           String newFilterClass = filterClass.replace("gov.usgs.winston", "gov.usgs.volcanoes.winston");
400           LOGGER.error("Defunct filter class found. I'll fix it this time, but this will cease to work in future versions.");
401           LOGGER.error("Update ImportEW config to reference {} in place of {}", newFilterClass, filterClass);
402           filterClass = newFilterClass;
403         }
404         final TraceBufFilter filter =
405             (TraceBufFilter) Class.forName(filterClass).newInstance();
406         filter.configure(fc);
407         traceBufFilters.add(filter);
408       } catch (final Exception ex) {
409         LOGGER.error("Could not create TraceBuf filter: {}", ex);
410       }
411     }
412     Collections.sort(traceBufFilters);
413     for (final TraceBufFilter filter : traceBufFilters)
414       LOGGER.info("config, filter: {}", filter);
415   }

417   /**
418    * Tests the TraceBuf against each filter, stopping at first match.
419    * 
420    * @param tb
421    *          the TraceBuf to test
422    * @return filter that matched, null if no match
423    */
424   protected TraceBufFilter matchFilter(final TraceBuf tb) {
425     final Options options = getOptions(tb);
426     for (final TraceBufFilter filter : traceBufFilters) {
427       if (filter.match(tb, options)) {
428 
429         final Map<String, String> m = filter.getMetadata();
430         if (m != null)
431           addMetadata(tb.toWinstonString(), m);
432 
433         if (filter.isTerminal())
434           return filter;
435       }
436 
437     }
438     return null;
439   }

451   /**
452    * This class is called by the {@link ImportGeneric} thread whenever a
453    * tracebuf has been received.
454    * 
455    * $Log: not supported by cvs2svn $ Revision 1.11 2007/05/03 23:25:54
456    * dcervelli OOM diagnosis.
457    * 
458    * @author Dan Cervelli
459    */
460   class TraceBufHandler implements MessageListener {
461     public void messageReceived(final Message msg) {
462       totalTraceBufs++;
463       final TraceBuf tb = (TraceBuf) msg;
464       LOGGER.debug("RX: {}", tb.toString());
465 
466       final TraceBufFilter matchingFilter = matchFilter(tb);
467       boolean accept = false;
468       if (matchingFilter != null) {
469         accept = matchingFilter.isAccept();
470 
471         // Don't consume cycles on String.format if the message is not
472         // going
473         // to be logged. Yes, it does matter. Running String.format for
474         // every
475         // single tracebuf adds up.
476         LOGGER.debug("{}: {}", matchingFilter, tb);
477       } else
478         LOGGER.debug("No matching filter, rejected: " + tb);
479 
480       if (accept) {
481         totalTraceBufsAccepted++;
482         addTraceBufToQueue(tb);
483       } else {
484         totalTraceBufsRejected++;
485 
486         // ack tracebufs that were rejected, so they will not be resent.
487         if (msg.sendAck)
488           importGeneric.sendAck(tb.seq);
489 
490         if (matchingFilter.keepRejects()) {
491           // TODO: deal with the rejected keepers
492         }
493       }
494     }
495   }
496 
497   protected void addTraceBufToQueue(final TraceBuf tb) {
498     final String channel = tb.toWinstonString();
499     ConcurrentLinkedQueue<TraceBuf> q = channelTraceBufs.get(channel);
500     if (q == null) {
501       q = new ConcurrentLinkedQueue<TraceBuf>();
502       channelTraceBufs.put(channel, q);
503     }
504 
505     final Options ip = getOptions(tb);
506 
507     q.add(tb);
508     while (q.size() > ip.maxBacklog) {
509       q.poll();
510       // TODO: improve logging of dropped tracebufs
511       totalTraceBufsDropped++;
512       if (totalTraceBufsDropped % 100 == 1) {
513         LOGGER.info("Overfull backlog, dropped TraceBuf");
514       }
515     }
516   }

518   private void cycle(final boolean force) {
519     // CodeTimer ct0 = new CodeTimer("init");
520     for (final String key : channelTraceBufs.keySet()) {
521       final ConcurrentLinkedQueue<TraceBuf> q = channelTraceBufs.get(key);
522       if (q.isEmpty())
523         continue;
524 
525       final Options ip = getOptions(q.peek());
526 
527       if (force || ip.thresholdExceeded(q.peek().getStartTimeJ2K(), q.size())) {
528 
529         importChannel(q);
530         if (channelMetadata.containsKey(key))
531           importMetadata(key, channelMetadata.get(key));
532       }
533     }
534     // ct0.stop();
535     // if (ct0.getRunTimeMillis() > 1000)
536     // System.out.println("Long cycle: " + ct0);
537   }

539   private Runnable getPurgeRunnable(final String code, final int maxDays) {
540     return new Runnable() {
541       public void run() {
542         try {
543           fixerInput.purgeTables(code, maxDays);
544           try {
545             Thread.sleep(dropTableDelay);
546           } catch (final Exception e) {
547           }
548         } catch (final OutOfMemoryError e) {
549           handleOutOfMemoryError(e);
550         }
551       }
552     };
553   }

555   private Runnable getRepairRunnable(final String database, final String table) {
556     if (underRepair.contains(table))
557       return null;
558 
559     final Double lastRepair = attemptedRepair.get(table);
560     if (lastRepair != null) {
561       if (J2kSec.now() - lastRepair.doubleValue() > repairRetryInterval)
562         attemptedRepair.remove(table);
563       else
564         return null;
565     }
566 
567     underRepair.add(table);
568     return new Runnable() {
569       public void run() {
570         try {
571           final boolean healthy = fixerAdmin.repairTable(database, table);
572           LOGGER.info("After repair attempt, table {} appears to {}.", table,
573               (healthy ? "be healthy" : "still be broken"));
574           attemptedRepair.put(table, J2kSec.now());
575           underRepair.remove(table);
576         } catch (final OutOfMemoryError e) {
577           handleOutOfMemoryError(e);
578         }
579       }
580     };
581   }

583   private void importChannel(final ConcurrentLinkedQueue<TraceBuf> q) {
584     if (q.isEmpty() || underRepair.contains("channels")) {
585       System.out.println("isempty: " + q.isEmpty());
586       System.out.println("underRepair: " + underRepair.contains("channels"));
587       return;
588     }
589     final TraceBuf tb = q.peek();
590     final String code = tb.toWinstonString();
591 
592     if (!existingChannels.contains(code) && !channels.channelExists(code)) {
593       LOGGER.info("Creating new channel '" + code + "' in Winston database.");
594       channels.createChannel(code);
595     }
596     existingChannels.add(code);
597 
598     final ArrayList<TraceBuf> tbs = new ArrayList<TraceBuf>(q.size());
599     while (!q.isEmpty()) {
600       final TraceBuf t = q.poll();
601       tbs.add(t);
602       if (t.sendAck)
603         importGeneric.sendAck(t.seq);
604 
605     }
606 
607     inputTimer.start();
608     // TODO: catch exceptions around here
609     final Options ip = getOptions(tbs.get(0));
610     final List<InputEW.InputResult> results =
611         input.inputTraceBufs(tbs, ip.rsamEnable, ip.rsamDelta, ip.rsamDuration);
612     inputTimer.stop();
613 
614     ChannelStatus status = channelStatus.get(code);
615     if (status == null) {
616       status = new ChannelStatus(code);
617       channelStatus.put(code, status);
618     }
619     // TODO: suppress repetitive MySQL exceptions
620     if (results.size() == 1) {
621       final InputEW.InputResult result = results.get(0);
622       switch (result.code) {
623         case ERROR_DATABASE:
624         case ERROR_NO_WINSTON:
625           // shouldn't happen because several database accesses must have
626           // already occurred by this point
627           break;
628         case ERROR_INPUT:
629           // should never happen, conditions checked before call
630           break;
631         case ERROR_TIME_SPAN:
632           LOGGER.warn("Time span error.");
633           final Runnable repairTask = getRepairRunnable("ROOT", "channels");
634           if (repairTask != null)
635             fixer.submit(repairTask);
636           break;
637         default:
638           LOGGER.warn("Error: " + result.code);
639       }
640     } else {
641       for (int i = 0; i < results.size() - 2; i++) {
642         final InputEW.InputResult result = results.get(i);
643         status.process(result.traceBuf, result.code);
644         boolean repair = false;
645         switch (result.code) {
646           case SUCCESS_CREATED_TABLE:
647             LOGGER.info("Day table created: " + tb.toWinstonString() + " "
648                 + winstonDateFormat.format(J2kSec.asDate(tb.getStartTimeJ2K())));
649             fixer.submit(getPurgeRunnable(code, ip.maxDays));
650           case SUCCESS:
651             attemptedRepair.remove(code);
652             totalTraceBufsWritten++;
653             LOGGER.debug("Insert: " + tb.toString());
654             break;
655           case ERROR_DATABASE:
656             totalTraceBufsFailed++;
657             repair = true;
658             LOGGER.warn("Database error: " + tb.toString());
659             break;
660           case ERROR_UNKNOWN:
661             totalTraceBufsFailed++;
662             repair = true;
663             LOGGER.warn("Unknown insert error: " + tb.toString());
664             break;
665           case ERROR_CHANNEL:
666           case ERROR_NULL_TRACEBUF:
667             totalTraceBufsFailed++;
668             // these errors should never occur
669             LOGGER.warn("Bad channel/null TraceBuf.");
670             break;
671           case ERROR_DUPLICATE:
672             totalTraceBufsFailed++;
673             LOGGER.warn("Duplicate TraceBuf: " + tb.toString());
674             break;
675           case NO_CODE:
676             // this should never occur
677             totalTraceBufsFailed++;
678             LOGGER.warn("No error/success code: " + tb.toString());
679             break;
680           case ERROR_HELICORDER:
681             break;
682           case ERROR_INPUT:
683             break;
684           case ERROR_NO_WINSTON:
685             break;
686           case ERROR_TIME_SPAN:
687             break;
688           case SUCCESS_HELICORDER:
689             break;
690           case SUCCESS_TIME_SPAN:
691             break;
692           default:
693             break;
694         }
695         if (repair) {
696           final String dt = winstonDateFormat.format(J2kSec.asDate(tb.getStartTimeJ2K()));
697           final Runnable repairTask = getRepairRunnable(code, code + "$$" + dt);
698           if (repairTask != null)
699             fixer.submit(repairTask);
700         }
701       }
702 
703       final InputEW.InputResult timeSpanResult = results.get(results.size() - 1);
704       if (timeSpanResult.code == InputEW.InputResult.Code.ERROR_TIME_SPAN) {
705         LOGGER.warn("Time span error.");
706         final Runnable repairTask = getRepairRunnable("ROOT", "channels");
707         if (repairTask != null)
708           fixer.submit(repairTask);
709       }
710       final InputEW.InputResult heliResult = results.get(results.size() - 2);
711       if (heliResult.code == InputEW.InputResult.Code.ERROR_HELICORDER) {
712         final String dt = winstonDateFormat.format(J2kSec.asDate(heliResult.failedHeliJ2K));
713         final String table = code + "$$H" + dt;
714         LOGGER.warn("Error writing helicorder data to table " + table + ".");
715         final Runnable repairTask = getRepairRunnable(code, table);
716         if (repairTask != null)
717           fixer.submit(repairTask);
718       }
719     }
720   }

734   @Override
735   public void run() {
736     // Register our interest in tracebuf messages.
737     final TraceBufHandler tbh = new TraceBufHandler();
738     importGeneric.addListener(MessageType.TYPE_TRACEBUF, tbh);
739     importGeneric.addListener(MessageType.TYPE_TRACEBUF2, tbh);
740 
741     boolean connected = false;
742     while (!connected) {
743       connected = importGeneric.connect();
744       if (!connected) {
745         try {
746           Thread.sleep(100);
747         } catch (final Exception e) {
748         }
749       }
750     }
751     while (!quit) {
752       try {
753         cycle(true);
754         Thread.sleep(10); // avoid busy-waiting when importing few
755                           // channels
756       } catch (final OutOfMemoryError e) {
757         handleOutOfMemoryError(e);
758       } catch (final Throwable e) {
759         LOGGER.error("Main loop exception: {}", e.getLocalizedMessage());
760         e.printStackTrace();
761       }
762     }
763     try {
764       cycle(true);
765     } catch (final Throwable e) {
766       LOGGER.error("Exception during final cycle: {}", e);
767     }
768   }

772     Options ip = channelOptions.get(channel);

798   public void printStatus() {
799     final Date now = new Date(CurrentTime.getInstance().now());
800     final long nowST = System.currentTimeMillis();
801     final double uptime = (now.getTime() - importStartTime.getTime()) / 1000.0;
802     final List<String> strings = new ArrayList<String>(4);
803 
804     strings.add("------- ImportEW --------");
805     strings.add("Status time: " + dateFormat.format(now));
806     strings.add("Start time:  " + dateFormat.format(importStartTime));
807     strings.add("Up time:     " + Time.secondsToString(uptime));
808 
809     long ht = importGeneric.getLastHeartbeatTime();
810     double dt = (double) (nowST - ht) / 1000;
811     if (ht == 0)
812       strings.add("Last HB RX:  (never)");
813     else
814       strings.add("Last HB RX:  " + dateFormat.format(new Date(ht)) + ", "
815           + Time.secondsToString(dt));
816     ht = importGeneric.getLastHeartbeatSentTime();
817     dt = (double) (nowST - ht) / 1000;
818     if (ht == 0)
819       strings.add("Last HB TX:  (never)");
820     else
821       strings.add("Last HB TX:  " + dateFormat.format(new Date(ht)) + ", "
822           + Time.secondsToString(dt));
823 
824     final Runtime rt = Runtime.getRuntime();
825     strings.add(String.format("Memory (used / max): %.1fMB / %.1fMB",
826         (rt.totalMemory() - rt.freeMemory()) / 1024.0 / 1024.0, rt.maxMemory() / 1024.0 / 1024.0));
827 
828     strings.add("---- TraceBufs");
829     strings.add("Accepted: " + totalTraceBufsAccepted);
830     strings.add("Written:  " + totalTraceBufsWritten);
831     strings.add("Failed:   " + totalTraceBufsFailed);
832     strings.add("Rejected: " + totalTraceBufsRejected);
833     strings.add("Dropped:  " + totalTraceBufsDropped);
834     final int pending = totalTraceBufsAccepted - totalTraceBufsWritten - totalTraceBufsFailed
835         - totalTraceBufsRejected - totalTraceBufsDropped;
836     strings.add("Pending:  " + pending);
837 
838     // by each filter
839     strings.add("---- Timing");
840     strings.add(String.format("Total input time:        %s",
841         Time.secondsToString(inputTimer.getTotalTimeMillis() / 1000)));
842     strings.add(String.format("Input time per TraceBuf: %.2fms",
843         inputTimer.getTotalTimeMillis() / totalTraceBufsWritten));
844 
845     for (final String s : strings)
846       System.out.println(s);
847   }

849   public void printChannels(final String s) {
850     char col = 'C';
851     if (s.length() > 1)
852       col = s.charAt(1);
853     final boolean desc = s.endsWith("-");
854

857     final List<ChannelStatus> channels = new ArrayList<ChannelStatus>(channelStatus.size());
858     channels.addAll(channelStatus.values());

866     System.out.println(ChannelStatus.getHeaderString());

870    * Find and parse the command line arguments.
871    * 
872    * @param args
873    *          The command line arguments.
874    */
875   public static JSAPResult getArguments(final String[] args) {
876     JSAPResult config = null;
877     try {
878       final SimpleJSAP jsap = new SimpleJSAP(JSAP_PROGRAM_NAME,
879           JSAP_EXPLANATION_PREFACE + DEFAULT_JSAP_EXPLANATION, DEFAULT_JSAP_PARAMETERS);
880 
881       config = jsap.parse(args);
882 
883       if (jsap.messagePrinted()) {
884         // The following error message is useful for catching the case
885         // when args are missing, but help isn't printed.
886         if (!config.getBoolean("help")) {
887           System.err.println("Try using the --help flag.");
888         }
889         System.exit(1);
890       }
891     } catch (final Exception ex) {
892       ex.printStackTrace();
893       System.exit(1);
894     }
895     return config;
896   }

911   /**
912    * Manage the console input.
913    * 
914    * @param im
915    *          The Winston importer.
916    */
917   public static void consoleInputManager(final ImportEW im) {
918     boolean acceptCommands = true;
919     final BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
920     LOGGER.info("Enter ? for console commands.");
921     while (acceptCommands) {
922       try {
923         String s = null;
924         try {
925           s = in.readLine();
926         } catch (final IOException ioex) {
927           LOGGER.error("IOException encountered while attempting to read console input. {}", ioex);
928         }
929 
930         if (s != null) {
931           s = s.toLowerCase().trim();
932           if (s.equals("q")) {
933             im.quit();
934             try {
935               im.join();
936             } catch (final Throwable e) {
937               LOGGER.error("Failed to quit cleanly. {}");
938             } finally {
939               im.printStatus();
940             }
941             System.exit(0);
942           } else if (s.equals("s"))
943             im.printStatus();
944           else if (s.startsWith("c"))
945             im.printChannels(s);
946           else if (s.equals("0"))
947             Log.setLevel(Level.OFF);
948           else if (s.equals("1"))
949             Log.setLevel(Level.ERROR);
950           else if (s.equals("2"))
951             Log.setLevel(Level.INFO);
952           else if (s.equals("3"))
953             Log.setLevel(Level.TRACE);
954           else if (s.equals("i")) {
955             acceptCommands = false;
956             LOGGER.error("No longer accepting console commands.");
957           } else if (s.equals("?"))
958             ImportEW.printKeys();
959           else
960             ImportEW.printKeys();
961         }
962       } catch (final OutOfMemoryError e) {
963         im.handleOutOfMemoryError(e);
964       }
965     }
966   }
