366     protected void processOptions() {
367         optionFilters = new ArrayList<OptionsFilter>();
368         List<String> optionSets = config.getList("options");
369         for (String options : optionSets) {
370             if (options.equals("Default"))
371                 continue;
372             ConfigFile tc = config.getSubConfig(options);
373             try {
374                 OptionsFilter filter = new OptionsFilter(options, tc, defaultOptions);
375                 optionFilters.add(filter);
376             } catch (Exception ex) {
377                 logger.log(Level.SEVERE, "Could not create options: ", ex);
378             }
379         }
380         for (OptionsFilter filter : optionFilters)
381             logger.info("config, options, " + filter.getName() + ": " + filter);
382     }

384     protected void processFilters() {
385         traceBufFilters = new ArrayList<TraceBufFilter>();
386         List<String> filters = config.getList("filter");
387         for (String filterName : filters) {
388             ConfigFile fc = config.getSubConfig(filterName);
389             try {
390                 TraceBufFilter filter = (TraceBufFilter) Class.forName(fc.getString("class")).newInstance();
391                 filter.configure(fc);
392                 traceBufFilters.add(filter);
393             } catch (Exception ex) {
394                 logger.log(Level.SEVERE, "Could not create TraceBuf filter: ", ex);
395             }
396         }
397         Collections.sort(traceBufFilters);
398         for (TraceBufFilter filter : traceBufFilters)
399             logger.info("config, filter: " + filter);
400     }

402     /**
403      * Tests the TraceBuf against each filter, stopping at first match.
404      * 
405      * @param tb
406      *            the TraceBuf to test
407      * @return filter that matched, null if no match
408      */
409     protected TraceBufFilter matchFilter(TraceBuf tb) {
410         Options options = getOptions(tb);
411         for (TraceBufFilter filter : traceBufFilters) {
412             if (filter.match(tb, options)) {
413 
414                 Map<String, String> m = filter.getMetadata();
415                 if (m != null)
416                     addMetadata(tb.toWinstonString(), m);
417 
418                 if (filter.isTerminal())
419                     return filter;
420             }
421 
422         }
423         return null;
424     }

436     /**
437      * This class is called by the {@link ImportGeneric} thread whenever a
438      * tracebuf has been received.
439      * 
440      * $Log: not supported by cvs2svn $ Revision 1.11 2007/05/03 23:25:54
441      * dcervelli OOM diagnosis.
442      * 
443      * @author Dan Cervelli
444      */
445     class TraceBufHandler implements MessageListener {
446         public void messageReceived(Message msg) {
447             totalTraceBufs++;
448             TraceBuf tb = (TraceBuf) msg;
449             logger.finest("RX: " + tb.toString());
450 
451             TraceBufFilter matchingFilter = matchFilter(tb);
452             boolean accept = false;
453             if (matchingFilter != null) {
454                 accept = matchingFilter.isAccept();
455 
456                 // Don't consume cycles on String.format if the message is not
457                 // going
458                 // to be logged. Yes, it does matter. Running String.format for
459                 // every
460                 // single tracebuf adds up.
461                 if (logger.isLoggable(matchingFilter.getLogLevel()))
462                     logger.log(matchingFilter.getLogLevel(), String.format("%s: %s", matchingFilter, tb.toString()));
463             } else
464                 logger.finest("No matching filter, rejected: " + tb);
465 
466             if (accept) {
467                 totalTraceBufsAccepted++;
468                 addTraceBufToQueue(tb);
469             } else {
470                 totalTraceBufsRejected++;
471 
472                 // ack tracebufs that were rejected, so they will not be resent.
473                 if (msg.sendAck)
474                     importGeneric.sendAck(tb.seq);
475 
476                 if (matchingFilter.keepRejects()) {
477                     // TODO: deal with the rejected keepers
478                 }
479             }
480         }
481     }

483     protected void addTraceBufToQueue(TraceBuf tb) {
484         String channel = tb.toWinstonString();
485         ConcurrentLinkedQueue<TraceBuf> q = channelTraceBufs.get(channel);
486         if (q == null) {
487             q = new ConcurrentLinkedQueue<TraceBuf>();
488             channelTraceBufs.put(channel, q);
489         }
490 
491         Options ip = getOptions(tb);
492 
493         q.add(tb);
494         while (q.size() > ip.maxBacklog) {
495             q.poll();
496             // TODO: improve logging of dropped tracebufs
497             totalTraceBufsDropped++;
498             if (totalTraceBufsDropped % 100 == 1) {
499                 logger.fine("Overfull backlog, dropped TraceBuf");
500             }
501         }
502     }

504     private void cycle(boolean force) {
505         // CodeTimer ct0 = new CodeTimer("init");
506         for (String key : channelTraceBufs.keySet()) {
507             ConcurrentLinkedQueue<TraceBuf> q = channelTraceBufs.get(key);
508             if (q.isEmpty())
509                 continue;
510 
511             Options ip = getOptions(q.peek());
512 
513             if (force || ip.thresholdExceeded(q.peek().getStartTimeJ2K(), q.size())) {
514 
515                 importChannel(q);
516                 if (channelMetadata.containsKey(key))
517                     importMetadata(key, channelMetadata.get(key));
518             }
519         }
520         // ct0.stop();
521         // if (ct0.getRunTimeMillis() > 1000)
522         // System.out.println("Long cycle: " + ct0);
523     }

525     private Runnable getPurgeRunnable(final String code, final int maxDays) {
526         return new Runnable() {
527             public void run() {
528                 try {
529                     fixerInput.purgeTables(code, maxDays);
530                     try {
531                         Thread.sleep(dropTableDelay);
532                     } catch (Exception e) {
533                     }
534                 } catch (OutOfMemoryError e) {
535                     handleOutOfMemoryError(e);
536                 }
537             }
538         };
539     }

541     private Runnable getRepairRunnable(final String database, final String table) {
542         if (underRepair.contains(table))
543             return null;
544 
545         Double lastRepair = attemptedRepair.get(table);
546         if (lastRepair != null) {
547             if (CurrentTime.getInstance().nowJ2K() - lastRepair.doubleValue() > repairRetryInterval)
548                 attemptedRepair.remove(table);
549             else
550                 return null;
551         }
552 
553         underRepair.add(table);
554         return new Runnable() {
555             public void run() {
556                 try {
557                     boolean healthy = fixerAdmin.repairTable(database, table);
558                     logger.info(String.format("After repair attempt, table %s appears to %s.", table,
559                             (healthy ? "be healthy" : "still be broken")));
560                     attemptedRepair.put(table, CurrentTime.getInstance().nowJ2K());
561                     underRepair.remove(table);
562                 } catch (OutOfMemoryError e) {
563                     handleOutOfMemoryError(e);
564                 }
565             }
566         };
567     }

569     private void importChannel(final ConcurrentLinkedQueue<TraceBuf> q) {
570         if (q.isEmpty() || underRepair.contains("channels")) {
571             System.out.println("isempty: " + q.isEmpty());
572             System.out.println("underRepair: " + underRepair.contains("channels"));
573             return;
574         }
575         TraceBuf tb = q.peek();
576         String code = tb.toWinstonString();
577 
578         if (!existingChannels.contains(code) && !channels.channelExists(code)) {
579             logger.info("Creating new channel '" + code + "' in Winston database.");
580             channels.createChannel(code);
581         }
582         existingChannels.add(code);
583 
584         ArrayList<TraceBuf> tbs = new ArrayList<TraceBuf>(q.size());
585         while (!q.isEmpty()) {
586             TraceBuf t = q.poll();
587             tbs.add(t);
588             if (t.sendAck)
589                 importGeneric.sendAck(t.seq);
590 
591         }
592 
593         inputTimer.start();
594         // TODO: catch exceptions around here
595         Options ip = getOptions(tbs.get(0));
596         List<InputEW.InputResult> results = input.inputTraceBufs(tbs, ip.rsamEnable, ip.rsamDelta, ip.rsamDuration);
597         inputTimer.stop();
598 
599         ChannelStatus status = channelStatus.get(code);
600         if (status == null) {
601             status = new ChannelStatus(code);
602             channelStatus.put(code, status);
603         }
604         // TODO: suppress repetitive MySQL exceptions
605         if (results.size() == 1) {
606             InputEW.InputResult result = results.get(0);
607             switch (result.code) {
608             case ERROR_DATABASE:
609             case ERROR_NO_WINSTON:
610                 // shouldn't happen because several database accesses must have
611                 // already occurred by this point
612                 break;
613             case ERROR_INPUT:
614                 // should never happen, conditions checked before call
615                 break;
616             case ERROR_TIME_SPAN:
617                 logger.warning("Time span error.");
618                 Runnable repairTask = getRepairRunnable("ROOT", "channels");
619                 if (repairTask != null)
620                     fixer.submit(repairTask);
621                 break;
622             default:
623                 logger.warning("Error: " + result.code);
624             }
625         } else {
626             for (int i = 0; i < results.size() - 2; i++) {
627                 InputEW.InputResult result = results.get(i);
628                 status.process(result.traceBuf, result.code);
629                 boolean repair = false;
630                 switch (result.code) {
631                 case SUCCESS_CREATED_TABLE:
632                     logger.fine("Day table created: " + tb.toWinstonString() + " "
633                             + winstonDateFormat.format(Util.j2KToDate(tb.getStartTimeJ2K())));
634                     fixer.submit(getPurgeRunnable(code, ip.maxDays));
635                 case SUCCESS:
636                     attemptedRepair.remove(code);
637                     totalTraceBufsWritten++;
638                     logger.finest("Insert: " + tb.toString());
639                     break;
640                 case ERROR_DATABASE:
641                     totalTraceBufsFailed++;
642                     repair = true;
643                     logger.warning("Database error: " + tb.toString());
644                     break;
645                 case ERROR_UNKNOWN:
646                     totalTraceBufsFailed++;
647                     repair = true;
648                     logger.warning("Unknown insert error: " + tb.toString());
649                     break;
650                 case ERROR_CHANNEL:
651                 case ERROR_NULL_TRACEBUF:
652                     totalTraceBufsFailed++;
653                     // these errors should never occur
654                     logger.warning("Bad channel/null TraceBuf.");
655                     break;
656                 case ERROR_DUPLICATE:
657                     totalTraceBufsFailed++;
658                     logger.warning("Duplicate TraceBuf: " + tb.toString());
659                     break;
660                 case NO_CODE:
661                     // this should never occur
662                     totalTraceBufsFailed++;
663                     logger.warning("No error/success code: " + tb.toString());
664                     break;
665                 case ERROR_HELICORDER:
666                     break;
667                 case ERROR_INPUT:
668                     break;
669                 case ERROR_NO_WINSTON:
670                     break;
671                 case ERROR_TIME_SPAN:
672                     break;
673                 case SUCCESS_HELICORDER:
674                     break;
675                 case SUCCESS_TIME_SPAN:
676                     break;
677                 default:
678                     break;
679                 }
680                 if (repair) {
681                     String dt = winstonDateFormat.format(Util.j2KToDate(tb.getStartTimeJ2K()));
682                     Runnable repairTask = getRepairRunnable(code, code + "$$" + dt);
683                     if (repairTask != null)
684                         fixer.submit(repairTask);
685                 }
686             }
687 
688             InputEW.InputResult timeSpanResult = results.get(results.size() - 1);
689             if (timeSpanResult.code == InputEW.InputResult.Code.ERROR_TIME_SPAN) {
690                 logger.warning("Time span error.");
691                 Runnable repairTask = getRepairRunnable("ROOT", "channels");
692                 if (repairTask != null)
693                     fixer.submit(repairTask);
694             }
695             InputEW.InputResult heliResult = results.get(results.size() - 2);
696             if (heliResult.code == InputEW.InputResult.Code.ERROR_HELICORDER) {
697                 String dt = winstonDateFormat.format(Util.j2KToDate(heliResult.failedHeliJ2K));
698                 String table = code + "$$H" + dt;
699                 logger.warning("Error writing helicorder data to table " + table + ".");
700                 Runnable repairTask = getRepairRunnable(code, table);
701                 if (repairTask != null)
702                     fixer.submit(repairTask);
703             }
704         }
705     }

719     public void run() {
720         // Register our interest in tracebuf messages.
721         TraceBufHandler tbh = new TraceBufHandler();
722         importGeneric.addListener(MessageType.TYPE_TRACEBUF, tbh);
723         importGeneric.addListener(MessageType.TYPE_TRACEBUF2, tbh);
724 
725         boolean connected = false;
726         while (!connected) {
727             connected = importGeneric.connect();
728             if (!connected) {
729                 try {
730                     Thread.sleep(100);
731                 } catch (Exception e) {
732                 }
733             }
734         }
735         while (!quit) {
736             try {
737                 cycle(true);
738                 Thread.sleep(10); // avoid busy-waiting when importing few
739                                   // channels
740             } catch (OutOfMemoryError e) {
741                 handleOutOfMemoryError(e);
742             } catch (Throwable e) {
743                 logger.log(Level.SEVERE, "Main loop exception: ", e.getLocalizedMessage());
744                 e.printStackTrace();
745             }
746         }
747         try {
748             cycle(true);
749         } catch (Throwable e) {
750             logger.log(Level.SEVERE, "Exception during final cycle: ", e);
751         }
752     }

754     protected Options getOptions(TraceBuf tb) {
755         String channel = tb.toWinstonString();
756         Options ip = channelOptions.get(channel);
757         if (ip == null) {
758             ip = defaultOptions;
759             for (OptionsFilter tf : optionFilters) {
760                 if (tf.match(tb)) {
761                     ip = tf.getOptions();
762                     break;
763                 }
764             }
765             channelOptions.put(channel, ip);
766         }
767         return ip;
768     }

787     public void printStatus() {
788         Date now = CurrentTime.getInstance().nowDate();
789         long nowST = System.currentTimeMillis();
790         double uptime = (double) (now.getTime() - importStartTime.getTime()) / 1000.0;
791         List<String> strings = new ArrayList<String>(4);
792 
793         strings.add("------- ImportEW --------");
794         strings.add("Status time: " + dateFormat.format(now));
795         strings.add("Start time:  " + dateFormat.format(importStartTime));
796         strings.add("Up time:     " + Util.timeDifferenceToString(uptime));
797 
798         long ht = importGeneric.getLastHeartbeatTime();
799         double dt = (double) (nowST - ht) / 1000;
800         if (ht == 0)
801             strings.add("Last HB RX:  (never)");
802         else
803             strings.add("Last HB RX:  " + dateFormat.format(new Date(ht)) + ", " + Util.timeDifferenceToString(dt));
804         ht = importGeneric.getLastHeartbeatSentTime();
805         dt = (double) (nowST - ht) / 1000;
806         if (ht == 0)
807             strings.add("Last HB TX:  (never)");
808         else
809             strings.add("Last HB TX:  " + dateFormat.format(new Date(ht)) + ", " + Util.timeDifferenceToString(dt));
810 
811         Runtime rt = Runtime.getRuntime();
812         strings.add(String.format("Memory (used / max): %.1fMB / %.1fMB",
813                 (rt.totalMemory() - rt.freeMemory()) / 1024.0 / 1024.0, rt.maxMemory() / 1024.0 / 1024.0));
814 
815         strings.add("---- TraceBufs");
816         strings.add("Accepted: " + totalTraceBufsAccepted);
817         strings.add("Written:  " + totalTraceBufsWritten);
818         strings.add("Failed:   " + totalTraceBufsFailed);
819         strings.add("Rejected: " + totalTraceBufsRejected);
820         strings.add("Dropped:  " + totalTraceBufsDropped);
821         int pending = totalTraceBufsAccepted - totalTraceBufsWritten - totalTraceBufsFailed - totalTraceBufsRejected
822                 - totalTraceBufsDropped;
823         strings.add("Pending:  " + pending);
824 
825         // by each filter
826         strings.add("---- Timing");
827         strings.add(String.format("Total input time:        %s",
828                 Util.timeDifferenceToString(inputTimer.getTotalTimeMillis() / 1000)));
829         strings.add(String.format("Input time per TraceBuf: %.2fms", inputTimer.getTotalTimeMillis()
830                 / totalTraceBufsWritten));
831 
832         for (String s : strings)
833             System.out.println(s);
834     }

897     /**
898      * Manage the console input.
899      * 
900      * @param im
901      *            The Winston importer.
902      */
903     public static void consoleInputManager(ImportEW im) {
904         im.logger.entering(im.getClass().getName(), "consoleInputManager");
905         boolean acceptCommands = true;
906         BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
907         im.logger.info("Enter ? for console commands.");
908         while (acceptCommands) {
909             try {
910                 String s = null;
911                 try {
912                     s = in.readLine();
913                 } catch (IOException ioex) {
914                     im.logger
915                             .log(Level.SEVERE, "IOException encountered while attempting to read console input.", ioex);
916                 }
917 
918                 if (s != null) {
919                     s = s.toLowerCase().trim();
920                     if (s.equals("q")) {
921                         im.quit();
922                         try {
923                             im.join();
924                         } catch (Throwable e) {
925                             im.logger.log(Level.SEVERE, "Failed to quit cleanly.", e);
926                         } finally {
927                             im.printStatus();
928                         }
929                         System.exit(0);
930                     } else if (s.equals("s"))
931                         im.printStatus();
932                     else if (s.startsWith("c"))
933                         im.printChannels(s);
934                     else if (s.equals("0"))
935                         im.setLogLevel(Level.OFF);
936                     else if (s.equals("1"))
937                         im.setLogLevel(Level.WARNING);
938                     else if (s.equals("2"))
939                         im.setLogLevel(Level.FINE);
940                     else if (s.equals("3"))
941                         im.setLogLevel(Level.ALL);
942                     else if (s.equals("i")) {
943                         acceptCommands = false;
944                         im.logger.warning("No longer accepting console commands.");
945                     } else if (s.equals("?"))
946                         ImportEW.printKeys();
947                     else
948                         ImportEW.printKeys();
949                 }
950             } catch (OutOfMemoryError e) {
951                 im.handleOutOfMemoryError(e);
952             }
953         }
954         im.logger.exiting(im.getClass().getName(), "consoleInputManager");
955     }

957     public static void main(String[] args) {
958         JSAPResult config = getArguments(args);
959 
960         String fn = Util.stringToString(config.getString("configFilename"), DEFAULT_CONFIG_FILENAME);
961 
962         Level logLevel = Level.parse(DEFAULT_LOG_LEVEL);
963 
964         if (config.getString("logLevel") != null) {
965             try {
966                 logLevel = Level.parse(config.getString("logLevel"));
967             } catch (IllegalArgumentException ex) {
968                 System.err.println("Invalid log level: " + config.getString("logLevel"));
969                 System.err.println("Using default log level: " + logLevel);
970             }
971         } else {
972             if (config.getBoolean("logoff"))
973                 logLevel = Level.OFF;
974 
975             if (config.getBoolean("lognormal"))
976                 logLevel = Level.WARNING;
977 
978             if (config.getBoolean("loghigh"))
979                 logLevel = Level.FINE;
980 
981             if (config.getBoolean("logall"))
982                 logLevel = Level.ALL;
983 
984         }
985 
986         ImportEW im = new ImportEW(fn);
987         im.setLogLevel(logLevel);
988 
989         // Start the importer. start() automatically calls run()
990         im.start();
991 
992         // Decide if we're accepting commands based on the args parsed by jsap
993         // and pass the importer to the consoleInputManager if we are.
994         if (!(config.getBoolean("noinput")))
995             consoleInputManager(im);
996     }
