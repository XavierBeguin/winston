13  /**
14   * Import station metadata from hypoinverse station file
15   *
16   * Format taken from:
17   * http://folkworm.ceri.memphis.edu/ew-doc/USER_GUIDE/hypoinv_sta.html
18   *
19   * @author tparker
20   *
21   */
22  public class ImportHypoinverse extends AbstractMetadataImporter {
23    private static final Logger LOGGER = LoggerFactory.getLogger(ImportHypoinverse.class);
24  
25    public static final String me = ImportHypoinverse.class.getName();
26  
27    public ImportHypoinverse(final String configFile) {
28      super(configFile);
29    }
30  
31    @Override
32    public List<Instrument> readMetadata(final String fn) {
33      LOGGER.info("Reading {}", fn);
34  
35      final List<Instrument> list = new LinkedList<Instrument>();
36      try {
37        String record;
38        final BufferedReader in = new BufferedReader(new FileReader(fn));
39        record = in.readLine();
40  
41        while (record != null) {
42          final Instrument inst = new Instrument();
43          inst.setName(record.substring(0, 5).trim());
44  
45          double lat = Double.parseDouble(record.substring(15, 17));
46          lat += Double.parseDouble(record.substring(18, 25).trim()) / 60;
47          if (record.substring(25, 26).equals("S"))
48            lat *= -1;
49          inst.setLatitude(lat);
50  
51          double lon = Double.parseDouble(record.substring(26, 29).trim());
52          lon += Double.parseDouble(record.substring(30, 36).trim()) / 60;
53          if (record.substring(37, 38).equals("W"))
54            lon *= -1;
55          inst.setLongitude(lon);;
56  
57          final int height = Integer.parseInt(record.substring(38, 42).trim());
58          inst.setHeight(height);
59  
60          list.add(inst);
61          record = in.readLine();
62        }
63        in.close();
64      } catch (final Exception e) {
65        e.printStackTrace();
66      }
67      return list;
68    }
69  
70    /**
71     * @param args
72     */
73    public static void main(final String[] args) {
74      String configFile;
75      String dataless;
76  
77      if (args.length == 0) {
78        System.err.println("Usage: ImportDataless [-c <winston.config>] <dataless>");
79        System.exit(1);
80      }
81  
82      if (args[0].equals("-c")) {
83        configFile = args[1];
84        dataless = args[2];
85      } else {
86        configFile = DEFAULT_CONFIG_FILE;
87        dataless = args[0];
88      }
89  
90      final ImportHypoinverse imp = new ImportHypoinverse(configFile);
91      imp.updateInstruments(dataless);
92    }
