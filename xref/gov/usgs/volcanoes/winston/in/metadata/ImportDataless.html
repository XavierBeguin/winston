25  	public List<Instrument> readMetadata(String fn)
26  	{
27  		LOGGER.fine("Reading " + fn);
28  		
29  		List<Instrument> list = new LinkedList<Instrument>();
30  		try
31  		{
32  			DataInputStream ls = new DataInputStream(new BufferedInputStream(
33  	                new FileInputStream(fn)));
34  
35  			ImportDirector importDirector = new SeedImportDirector();
36  			SeedObjectBuilder objectBuilder = new SeedObjectBuilder();
37  			importDirector.assignBuilder(objectBuilder);  // register the builder with the director
38  			// begin reading the stream with the construct command
39  			importDirector.construct(ls);  // construct SEED objects silently
40  			SeedObjectContainer container = (SeedObjectContainer)importDirector.getBuilder().getContainer();
41  			
42  			Object object;
43  			container.iterate();
44  			while ((object = container.getNext()) != null)
45  			{
46  				Blockette b = (Blockette)object;
47  				if (b.getType() != 50)
48  					continue;
49  				
50  				Instrument inst = new Instrument();
51  				inst.setLatitude((Double)b.getFieldVal(4));
52  				inst.setLongitude((Double)b.getFieldVal(5));
53  				inst.setHeight((Double)b.getFieldVal(6));
54  				inst.setName((String)b.getFieldVal(3));
55  				inst.setDescription((String)b.getFieldVal(9));
56  				
57  				LOGGER.fine("found " + inst.toString());
58  				list.add(inst);
59  			}
60  		}
61  		catch (Exception e)
62  		{
63  			e.printStackTrace();
64  		}
65  		return list;
66  	}
67  	
68  	/**
69  	 * @param args
70  	 */
71  	public static void main(String[] args) {
72  		String configFile;
73  		String dataless;
74  		
75  		if (args.length == 0) {
76  			System.err.println("Usage: ImportDataless [-c <winston.config>] <dataless>");
77  			System.exit(1);
78  		}
79  
80  		if (args[0].equals("-c")) {
81  			configFile = args[1];
82  			dataless = args[2];
83  		} else {
84  			configFile = DEFAULT_CONFIG_FILE;
85  			dataless = args[0];
86  		}
87  		
88  		ImportDataless imp = new ImportDataless(configFile);
89  		imp.updateInstruments(dataless);
90  	}
