7   package gov.usgs.volcanoes.winston.legacyServer;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  import com.martiansoftware.jsap.JSAPResult;
13  import com.martiansoftware.jsap.Parameter;
14  import com.martiansoftware.jsap.Switch;
15  
16  import gov.usgs.volcanoes.core.args.Args;
17  import gov.usgs.volcanoes.core.args.Arguments;
18  import gov.usgs.volcanoes.core.args.decorator.ConfigFileArg;
19  import gov.usgs.volcanoes.core.args.decorator.VerboseArg;
20  
21  /**
22   * Argument processor for Pensive.
23   *
24   * @author Tom Parker
25   */
26  public class WWSArgs {
27    private static final Logger LOGGER = LoggerFactory.getLogger(WWSArgs.class);
28  
29    private static final String DEFAULT_CONFIG_FILENAME = "WWS.config";
30  
31    private static final String PROGRAM_NAME = "java -jar gov.usgs.volcanoes.winston.server.WWS";
32    private static final String EXPLANATION = "I am the Winston wave server\n";
33    
34    /** format of time on cmd line */
35    public static final String INPUT_TIME_FORMAT = "yyyyMMddHHmm";
36  
37    private static final Parameter[] PARAMETERS = new Parameter[] {new Switch("noInput", 'i', "noinput", "Do not poll keyboard for input.")};
38  
39    /** if true be verbose */
40    public final boolean isVerbose;
41    
42    /** if true don't poll console for input */
43    public final boolean isNoInput;
44  
45    /** my config file. */
46    public final String configFileName;
47  
48    /**
49     * Class constructor.
50     * @param commandLineArgs the command line arguments
51     * @throws Exception when things go wrong
52     */
53    public WWSArgs(final String[] commandLineArgs) throws Exception {
54      
55      Arguments args = null;
56      args = new Args(PROGRAM_NAME, EXPLANATION, PARAMETERS);
57      args = new ConfigFileArg(DEFAULT_CONFIG_FILENAME, args);
58      args = new VerboseArg(args);
59  
60      JSAPResult jsapResult = null;
61      jsapResult = args.parse(commandLineArgs);
62  
63      isVerbose = jsapResult.getBoolean("verbose");
64      configFileName = jsapResult.getString("config-filename");
65      isNoInput = jsapResult.getBoolean("noInput");
66      LOGGER.debug("Setting: verbose={}", isVerbose);
67    }
68  }
