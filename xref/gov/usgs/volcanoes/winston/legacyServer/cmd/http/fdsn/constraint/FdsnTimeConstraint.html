14  abstract public class FdsnTimeConstraint implements FdsnConstraint {
15    protected static final String FAR_IN_FUTURE = "2070-01-01T00:00:00.0000";
16    protected static final String FAR_IN_PAST = "1970-01-01T00:00:00.0000";
17  
18    private FdsnDateFormat fdsnDate;
19    
20    protected FdsnTimeConstraint() {
21      fdsnDate = new FdsnDateFormat();
22    }
23    
24    protected double dateStringToDouble(final String s1, final String s2) throws ParseException {
25      String s = StringUtils.stringToString(s1, s2);
26  
27      if (s.indexOf('T') == -1)
28        s += "T00:00:00.0000";
29  
30      final int dot = s.indexOf('.');
31      if (dot == -1)
32        s += ".0000";
33      else if (dot + 5 > s.length())
34        s = s.substring(0, dot + 5);
35      else
36        while (dot + 5 < s.length())
37          s += "0";
38  
39      return J2kSec.fromDate(fdsnDate.parse(s));
40    }
41  }
