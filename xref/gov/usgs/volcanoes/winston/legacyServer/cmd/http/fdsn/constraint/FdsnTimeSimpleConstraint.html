8   import gov.usgs.volcanoes.winston.legacyServer.cmd.http.fdsn.FdsnException;

10  /**
11   * @author Tom Parker
12   *
13   */
14  public class FdsnTimeSimpleConstraint extends FdsnTimeConstraint {
15  
16  
17    public final double endTime; // j2ksec
18    public final double startTime; // j2ksec
19  
20    public FdsnTimeSimpleConstraint(final String startTime, final String endTime)
21        throws FdsnException {
22      super();
23      try {
24        this.startTime = dateStringToDouble(startTime, FAR_IN_PAST);
25        this.endTime = dateStringToDouble(endTime, FAR_IN_FUTURE);
26      } catch (final ParseException e) {
27        throw new FdsnException(400, "Can't parse time constraint: " + this);
28      }
29    }
30  
31    public boolean matches(final Channel chan) {
32      final double end = chan.getMaxTime();
33      if (!Double.isNaN(end) && chan.getMaxTime() > endTime) {
34        return false;
35      }
36  
37      final double start = chan.getMinTime();
38      if (!Double.isNaN(start) && chan.getMinTime() < startTime) {
39        return false;
40      }
41  
42      return true;
43    }
44  
45    public boolean matches(final TraceBuf buf) {
46      final double end = buf.lastSampleTime() + buf.samplingPeriod();
47      if (end > endTime) {
48        return false;
49      }
50  
51      final double start = buf.firstSampleTime();
52      if (start < startTime) {
53        return false;
54      }
55  
56      return true;
57    }
58  
59    @Override
60    public String toString() {
61      return "FdsnTimeSimpleConstraint: " + J2kSec.toDateString(startTime) + ":"
62          + J2kSec.toDateString(endTime);
63    }
64  
65  
66  }
