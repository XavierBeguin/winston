32    @Override
33    protected void sendResponse() {
34      // validate input. Write error and return if bad.
35      final int sortCol = StringUtils.stringToInt(arguments.get("ob"), DEFAULT_OB);
36      if (sortCol < 1 || sortCol > 8) {
37        writeSimpleHTML("Error: could not parse ob = " + arguments.get("ob"));
38        return;
39      }
40  
41      final String o = StringUtils.stringToString(arguments.get("so"), DEFAULT_SO);
42      final char order = o.charAt(0);
43      if (order != 'a' && order != 'd') {
44        writeSimpleHTML("Error: could not parse so = " + arguments.get("so"));
45        return;
46      }
47  
48      final String tz = StringUtils.stringToString(arguments.get("tz"), DEFAULT_TZ);
49      final TimeZone timeZone = TimeZone.getTimeZone(tz);
50  
51      // write header
52      final String[] colTitle = {null, "Pin", "S", "C", "N", "L", "Earliest", "Most Recent", "Type"};
53      colTitle[sortCol] += order == 'a' ? " &#9652;" : " &#9662;";
54  
55      final char[] colOrd = new char[colTitle.length];
56      Arrays.fill(colOrd, 'a');
57      if (order == 'a')
58        colOrd[sortCol] = 'd';
59  
60      final StringBuilder output = new StringBuilder();
61  
62      output.append("<table CELLPADDING=\"5\"><tr>");
63      for (int i = 1; i < colTitle.length; i++)
64        output.append(
65            "<th><a href=\"?ob=" + i + "&so=" + colOrd[i] + "\">" + colTitle[i] + "</a></th>");
66  
67      output.append("</tr>");
68  
69      // get and sort menu
70      final List<String> list = emulator.getWaveServerMenu(true, 0, 0, maxDays);
71      final String[][] menu = new String[list.size()][8];
72      int i = 0;
73      for (final String s : list)
74        menu[i++] = s.split("\\s");
75  
76      Arrays.sort(menu, getMenuComparator(sortCol, order));
77  
78      // display menu items
79      for (final String[] line : menu) {
80        if (line.length < 8) {
81          output.append("can't parse line, skipping. " + Arrays.toString(line));
82          continue;
83        }
84  
85        SimpleDateFormat dateF = new SimpleDateFormat(Time.STANDARD_TIME_FORMAT);
86        dateF.setTimeZone(timeZone);
87        
88        final double start = Double.parseDouble(line[6]);
89        final double end = Double.parseDouble(line[7]);
90  
91        output.append("<tr>");
92        output.append("<td>" + line[1] + "</td>");
93        output.append("<td>" + line[2] + "</td>");
94        output.append("<td>" + line[3] + "</td>");
95        output.append("<td>" + line[4] + "</td>");
96        output.append("<td>" + line[5] + "</td>");
97        output.append("<td>" + dateF.format(Ew.asEpoch(start)) + "</td>");
98        output.append("<td>" + dateF.format(Ew.asEpoch(end)) + "</td>");
99        output.append("<td>" + line[8] + "</td>");
100       output.append("</tr>\n");
101     }
102 
103     output.append("</table>");
104     writeSimpleHTML(output.toString());
105   }

107   private Comparator<String[]> getMenuComparator(final int sortCol, final char order) {
108     return new Comparator<String[]>() {
109       public int compare(final String[] e1, final String[] e2) {
110         // numeric columns
111         if (sortCol == 1 || sortCol == 6 || sortCol == 7) {
112           final double d1 = Double.parseDouble(e1[sortCol]);
113           final double d2 = Double.parseDouble(e2[sortCol]);
114 
115           // Do this the hard way to avoid an int overflow.
116           // Yes, this bug was encountered in a running system.
117           final double d = d1 - d2;
118           if (d == 0)
119             return 0;
120 
121           final int i = (d < 0 ? -1 : 1);
122 
123           if (order == 'a')
124             return i;
125           else
126             return -i;
127         }
128         // textual columns
129         else {
130           if (order == 'a')
131             return e1[sortCol].compareTo(e2[sortCol]);
132           else
133             return e2[sortCol].compareTo(e1[sortCol]);
134         }
135       }
136     };
137   }
